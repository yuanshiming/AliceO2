67  void process(soa::Filtered<soa::Join<aod::HfCandProng3, aod::HFSelXicToPKPiCandidate>> const& candidates)
68  {
69  for (auto& candidate : candidates) {
70  if (!(candidate.hfflag() & 1 << DecayType::XicToPKPi)) {
71  continue;
72  }
73  if (cutYCandMax >= 0. && std::abs(YXic(candidate)) > cutYCandMax) {
74  continue;
75  }
76  if (candidate.isSelXicToPKPi() >= d_selectionFlagXic) {
77  registry.fill(HIST("hmass"), InvMassXicToPKPi(candidate), candidate.pt());
78  }
79  if (candidate.isSelXicToPiKP() >= d_selectionFlagXic) {
80  registry.fill(HIST("hmass"), InvMassXicToPiKP(candidate), candidate.pt());
81  }
82  registry.fill(HIST("hptcand"), candidate.pt());
83  registry.fill(HIST("hptprong0"), candidate.ptProng0());
84  registry.fill(HIST("hptprong1"), candidate.ptProng1());
85  registry.fill(HIST("hptprong2"), candidate.ptProng2());
86  registry.fill(HIST("hdeclength"), candidate.decayLength(), candidate.pt());
87  registry.fill(HIST("hd0Prong0"), candidate.impactParameter0(), candidate.pt());
88  registry.fill(HIST("hd0Prong1"), candidate.impactParameter1(), candidate.pt());
89  registry.fill(HIST("hd0Prong2"), candidate.impactParameter2(), candidate.pt());
90  registry.fill(HIST("hCt"), CtXic(candidate), candidate.pt());
91  registry.fill(HIST("hCPA"), candidate.cpa(), candidate.pt());
92  registry.fill(HIST("hEta"), candidate.eta(), candidate.pt());
93  registry.fill(HIST("hselectionstatus"), candidate.isSelXicToPKPi(), candidate.pt());
94  registry.fill(HIST("hselectionstatus"), candidate.isSelXicToPiKP(), candidate.pt());
95  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter0(), candidate.pt());
96  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter1(), candidate.pt());
97  registry.fill(HIST("hImpParErr"), candidate.errorImpactParameter2(), candidate.pt());
98  registry.fill(HIST("hDecLenErr"), candidate.errorDecayLength(), candidate.pt());
99  registry.fill(HIST("hDecLenErr"), candidate.chi2PCA(), candidate.pt());
100  }
101  }
102 };

123  void process(soa::Filtered<soa::Join<aod::HfCandProng3, aod::HFSelXicToPKPiCandidate, aod::HfCandProng3MCRec>> const& candidates,
124  soa::Join<aod::McParticles, aod::HfCandProng3MCGen> const& particlesMC, aod::BigTracksMC const& tracks)
125  {
126  // MC rec.
127  //Printf("MC Candidates: %d", candidates.size());
128  for (auto& candidate : candidates) {
129  if (!(candidate.hfflag() & 1 << DecayType::XicToPKPi)) {
130  continue;
131  }
132  if (cutYCandMax >= 0. && std::abs(YXic(candidate)) > cutYCandMax) {
133  continue;
134  }
135  if (std::abs(candidate.flagMCMatchRec()) == 1 << DecayType::XicToPKPi) {
136  // Get the corresponding MC particle.
137  auto indexMother = RecoDecay::getMother(particlesMC, candidate.index0_as<aod::BigTracksMC>().mcParticle_as<soa::Join<aod::McParticles, aod::HfCandProng3MCGen>>(), pdg::Code::kXiCPlus, true);
138  auto particleMother = particlesMC.iteratorAt(indexMother);
139  registry.fill(HIST("hPtGenSig"), particleMother.pt()); // gen. level pT
140  registry.fill(HIST("hPtRecSig"), candidate.pt()); // rec. level pT
141  registry.fill(HIST("hCPARecSig"), candidate.cpa());
142  registry.fill(HIST("hEtaRecSig"), candidate.eta());
143  } else {
144  registry.fill(HIST("hPtRecBg"), candidate.pt());
145  registry.fill(HIST("hCPARecBg"), candidate.cpa());
146  registry.fill(HIST("hEtaRecBg"), candidate.eta());
147  }
148  }
149  // MC gen.
150  //Printf("MC Particles: %d", particlesMC.size());
151  for (auto& particle : particlesMC) {
152  if (std::abs(particle.flagMCMatchGen()) == 1 << DecayType::XicToPKPi) {
153  if (cutYCandMax >= 0. && std::abs(RecoDecay::Y(array{particle.px(), particle.py(), particle.pz()}, RecoDecay::getMassPDG(particle.pdgCode()))) > cutYCandMax) {
154  continue;
155  }
156  registry.fill(HIST("hPtGen"), particle.pt());
157  registry.fill(HIST("hEtaGen"), particle.eta());
158  }
159  }
160  }
161 };
