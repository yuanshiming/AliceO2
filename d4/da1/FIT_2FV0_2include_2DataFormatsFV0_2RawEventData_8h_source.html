288  std::vector<char> to_vector(bool tcm)
289  {
290  constexpr int CRUWordSize = 16;
291 
292  std::vector<char> result(size() * CRUWordSize);
293  char* out = result.data();
294  if (!tcm) {
295  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
296  out += EventHeader::PayloadSize;
297  LOG(DEBUG) << " Write PM header: nWords: " << (int)mEventHeader.nGBTWords
298  << " orbit: " << int(mEventHeader.orbit)
299  << " BC: " << int(mEventHeader.bc)
300  << " size: " << result.size();
301  printHexEventHeader();
302  out += CRUWordSize - EventHeader::PayloadSize; // Padding enabled
303 
304  for (uint64_t i = 0; i < mEventHeader.nGBTWords; ++i) {
305  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
306  out += EventData::PayloadSizeFirstWord;
307  LOG(DEBUG) << " 1st word: Ch: " << std::setw(2) << mEventData[2 * i].channelID
308  << " charge: " << std::setw(4) << mEventData[2 * i].charge
309  << " time: " << std::setw(4) << mEventData[2 * i].time;
310  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
311  out += EventData::PayloadSizeSecondWord;
312  LOG(DEBUG) << " 2nd word: Ch: " << std::setw(2) << mEventData[2 * i + 1].channelID
313  << " charge: " << std::setw(4) << mEventData[2 * i + 1].charge
314  << " time: " << std::setw(4) << mEventData[2 * i + 1].time;
315 
316  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
317  printHexEventData(i);
318  }
319  } else {
320  // TCM data
321  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
322  out += EventHeader::PayloadSize;
323  LOG(DEBUG) << " Write TCM header: nWords: " << (int)mEventHeader.nGBTWords
324  << " orbit: " << int(mEventHeader.orbit)
325  << " BC: " << int(mEventHeader.bc)
326  << " size: " << result.size();
327  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
328  out += sizeof(TCMdata);
329  // TODO: No TCM payload printing until the meaning of trigger bits and other flags is clarified
330  }
331  return result;
332  }
