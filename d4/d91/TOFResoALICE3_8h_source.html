58 float TOFResoALICE3Param(const float& momentum, const float& momentumError, const float& evtimereso, const float& length, const float& mass, const Parameters& parameters)
59 {
60  if (momentum <= 0) {
61  return -999;
62  }
63 
64  const float p2 = momentum * momentum;
65  const float Lc = length / 0.0299792458f;
66  const float mass2 = mass * mass;
67  const float ep = momentumError * momentum;
68  // const float ep = momentumError * p2;
69  const float etexp = Lc * mass2 / p2 / sqrt(mass2 + p2) * ep;
70  return sqrt(etexp * etexp + parameters[0] * parameters[0] + evtimereso * evtimereso);
71 }
