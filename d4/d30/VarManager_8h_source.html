316 template <typename T, typename U, typename V>
317 auto VarManager::getRotatedCovMatrixXX(const T& matrix, U phi, V theta)
318 {
319  auto cp = std::cos(phi);
320  auto sp = std::sin(phi);
321  auto ct = std::cos(theta);
322  auto st = std::sin(theta);
323  return matrix[0] * cp * cp * ct * ct // covXX
324  + matrix[1] * 2. * cp * sp * ct * ct // covXY
325  + matrix[2] * sp * sp * ct * ct // covYY
326  + matrix[3] * 2. * cp * ct * st // covXZ
327  + matrix[4] * 2. * sp * ct * st // covYZ
328  + matrix[5] * st * st; // covZZ
329 }

344  values[kTimestamp] = event.timestamp();

346

348  if (fgUsedVars[kIsINT7]) {
349  values[kIsINT7] = (event.alias()[kINT7] > 0);

365  }

460 template <uint32_t fillMap, typename T>
461 void VarManager::FillTrack(T const& track, float* values)
462 {
463  if (!values) {
464  values = fgValues;
465  }
466 
467  // Quantities based on the basic table (contains just kine information and filter bits)
468  if constexpr ((fillMap & Track) > 0 || (fillMap & Muon) > 0 || (fillMap & ReducedTrack) > 0 || (fillMap & ReducedMuon) > 0) {
469  values[kPt] = track.pt();
470  if (fgUsedVars[kPx]) {
471  values[kPx] = track.px();
472  }
473  if (fgUsedVars[kPy]) {
474  values[kPy] = track.py();
475  }
476  if (fgUsedVars[kPz]) {
477  values[kPz] = track.pz();
478  }
479  values[kEta] = track.eta();
480  values[kPhi] = track.phi();
481  values[kCharge] = track.sign();
482 
483  if constexpr ((fillMap & ReducedTrack) > 0 && !((fillMap & Pair) > 0)) {
484  values[kIsGlobalTrack] = track.filteringFlags() & (uint64_t(1) << 0);
485  values[kIsGlobalTrackSDD] = track.filteringFlags() & (uint64_t(1) << 1);
486  }
487  }
488 
489  // Quantities based on the barrel tables
490  if constexpr ((fillMap & TrackExtra) > 0 || (fillMap & ReducedTrackBarrel) > 0) {
491  values[kPin] = track.tpcInnerParam();
492  if (fgUsedVars[kIsITSrefit]) {
493  values[kIsITSrefit] = (track.flags() & o2::aod::track::ITSrefit) > 0;
494  }
495  if (fgUsedVars[kIsTPCrefit]) {
496  values[kIsTPCrefit] = (track.flags() & o2::aod::track::TPCrefit) > 0;
497  }
498  if (fgUsedVars[kIsGoldenChi2]) {
499  values[kIsGoldenChi2] = (track.flags() & o2::aod::track::GoldenChi2) > 0;
500  }
501  if (fgUsedVars[kIsSPDfirst]) {
502  values[kIsSPDfirst] = (track.itsClusterMap() & uint8_t(1)) > 0;
503  }
504  if (fgUsedVars[kIsSPDboth]) {
505  values[kIsSPDboth] = (track.itsClusterMap() & uint8_t(3)) > 0;
506  }
507  if (fgUsedVars[kIsSPDany]) {
508  values[kIsSPDany] = (track.itsClusterMap() & uint8_t(1)) || (track.itsClusterMap() & uint8_t(2));
509  }
510  values[kITSchi2] = track.itsChi2NCl();
511  values[kTPCncls] = track.tpcNClsFound();
512  values[kTPCchi2] = track.tpcChi2NCl();
513  values[kTrackLength] = track.length();
514  values[kTPCnclsCR] = track.tpcNClsCrossedRows();
515  values[kTRDPattern] = track.trdPattern();
516 
517  if constexpr ((fillMap & TrackExtra) > 0) {
518  if (fgUsedVars[kITSncls]) {
519  values[kITSncls] = track.itsNCls(); // dynamic column
520  }
521  }
522  if constexpr ((fillMap & ReducedTrackBarrel) > 0) {
523  if (fgUsedVars[kITSncls]) {
524  values[kITSncls] = 0.0;
525  for (int i = 0; i < 6; ++i) {
526  values[kITSncls] += ((track.itsClusterMap() & (1 << i)) ? 1 : 0);
527  }
528  }
529  values[kTrackDCAxy] = track.dcaXY();
530  values[kTrackDCAz] = track.dcaZ();
531  }
532  }
533 
534  // Quantities based on the barrel track selection table
535  if constexpr ((fillMap & TrackDCA) > 0) {
536  values[kTrackDCAxy] = track.dcaXY();
537  values[kTrackDCAz] = track.dcaZ();
538  }
539 
540  // Quantities based on the barrel track selection table
541  if constexpr ((fillMap & TrackSelection) > 0) {
542  values[kIsGlobalTrack] = track.isGlobalTrack();
543  values[kIsGlobalTrackSDD] = track.isGlobalTrackSDD();
544  }
545 
546  // Quantities based on the barrel covariance tables
547  if constexpr ((fillMap & TrackCov) > 0 || (fillMap & ReducedTrackBarrelCov) > 0) {
548  values[kTrackCYY] = track.cYY();
549  values[kTrackCZZ] = track.cZZ();
550  values[kTrackCSnpSnp] = track.cSnpSnp();
551  values[kTrackCTglTgl] = track.cTglTgl();
552  values[kTrackC1Pt21Pt2] = track.c1Pt21Pt2();
553  }
554 
555  // Quantities based on the barrel PID tables
556  if constexpr ((fillMap & TrackPID) > 0 || (fillMap & ReducedTrackBarrelPID) > 0) {
557  values[kTPCnSigmaEl] = track.tpcNSigmaEl();
558  values[kTPCnSigmaMu] = track.tpcNSigmaMu();
559  values[kTPCnSigmaPi] = track.tpcNSigmaPi();
560  values[kTPCnSigmaKa] = track.tpcNSigmaKa();
561  values[kTPCnSigmaPr] = track.tpcNSigmaPr();
562  values[kTOFnSigmaEl] = track.tofNSigmaEl();
563  values[kTOFnSigmaMu] = track.tofNSigmaMu();
564  values[kTOFnSigmaPi] = track.tofNSigmaPi();
565  values[kTOFnSigmaKa] = track.tofNSigmaKa();
566  values[kTOFnSigmaPr] = track.tofNSigmaPr();
567  values[kTPCsignal] = track.tpcSignal();
568  values[kTRDsignal] = track.trdSignal();
569  values[kTOFbeta] = track.beta();
570  if (fgUsedVars[kTPCsignalRandomized] || fgUsedVars[kTPCnSigmaElRandomized] || fgUsedVars[kTPCnSigmaPiRandomized] || fgUsedVars[kTPCnSigmaPrRandomized]) {
571  // NOTE: this is needed temporarilly for the study of the impact of TPC pid degradation on the quarkonium triggers in high lumi pp
572  // This study involves a degradation from a dE/dx resolution of 5% to one of 6% (20% worsening)
573  // For this we smear the dE/dx and n-sigmas using a gaus distribution with a width of 3.3%
574  // which is approx the needed amount to get dE/dx to a resolution of 6%
575  double randomX = gRandom->Gaus(0.0, 0.033);
576  values[kTPCsignalRandomized] = values[kTPCsignal] * (1.0 + randomX);
577  values[kTPCsignalRandomizedDelta] = values[kTPCsignal] * randomX;
578  values[kTPCnSigmaElRandomized] = values[kTPCnSigmaEl] * (1.0 + randomX);
579  values[kTPCnSigmaElRandomizedDelta] = values[kTPCnSigmaEl] * randomX;

668  values[kVertexingLxyErr] = -999.;
669  values[kVertexingLxyzErr] = -999.;
670  return;
671  }
672 
673  const auto& secondaryVertex = fgFitterTwoProng.getPCACandidate();

675  auto covMatrixPCA = fgFitterTwoProng.calcPCACovMatrix().Array();

682  std::array<float, 3> pvec1;
683  trackParVar0.getPxPyPzGlo(pvec0);

702  double theta = std::atan2(secondaryVertex[2] - collision.posZ(),
703  std::sqrt((secondaryVertex[0] - collision.posX()) * (secondaryVertex[0] - collision.posX()) +
704  (secondaryVertex[1] - collision.posY()) * (secondaryVertex[1] - collision.posY())));

707  values[kVertexingLxyErr] = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));

710  (collision.posY() - secondaryVertex[1]) * (collision.posY() - secondaryVertex[1]);
711  values[kVertexingLxyz] = values[kVertexingLxy] + (collision.posZ() - secondaryVertex[2]) * (collision.posZ() - secondaryVertex[2]);
712  values[kVertexingLxy] = std::sqrt(values[kVertexingLxy]);
713  values[kVertexingLxyz] = std::sqrt(values[kVertexingLxyz]);
