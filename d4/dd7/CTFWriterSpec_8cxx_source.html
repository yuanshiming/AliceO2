53 template <typename T>
54 void appendToTree(TTree& tree, const std::string brname, T& ptr)
55 {
56  auto* br = tree.GetBranch(brname.c_str());
57  auto* pptr = &ptr;
58  if (br) {
59  br->SetAddress(&pptr);
60  } else {
61  br = tree.Branch(brname.c_str(), &pptr);
62  }
63  br->Fill();
64  br->ResetAddress();
65 }

115 // process data of particular detector
116 template <typename C>
117 void CTFWriterSpec::processDet(o2::framework::ProcessingContext& pc, DetID det, CTFHeader& header, TTree* tree)
118 {
119  if (!isPresent(det) || !pc.inputs().isValid(det.getName())) {
120  return;
121  }
122  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
123  const auto ctfImage = C::getImage(ctfBuffer.data());
124  ctfImage.print(o2::utils::Str::concat_string(det.getName(), ": "));
125  if (mWriteCTF) {
126  ctfImage.appendToTree(*tree, det.getName());
127  header.detectors.set(det);
128  }
129  if (mCreateDict) {
130  if (!mFreqsAccumulation[det].size()) {
131  mFreqsAccumulation[det].resize(C::getNBlocks());
132  mFreqsMetaData[det].resize(C::getNBlocks());
133  }
134  if (!mHeaders[det]) { // store 1st header
135  mHeaders[det] = ctfImage.cloneHeader();
136  }
137  for (int ib = 0; ib < C::getNBlocks(); ib++) {
138  const auto& bl = ctfImage.getBlock(ib);
139  if (bl.getNDict()) {
140  auto& freq = mFreqsAccumulation[det][ib];
141  auto& mdSave = mFreqsMetaData[det][ib];
142  const auto& md = ctfImage.getMetadata(ib);
143  freq.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min, md.max);
144  mdSave = o2::ctf::Metadata{0, 0, md.coderType, md.streamSize, md.probabilityBits, md.opt, freq.getMinSymbol(), freq.getMaxSymbol(), (int)freq.size(), 0, 0};
145  }
146  }
147  }
148 }

150 // store dictionary of a particular detector
151 template <typename C>
152 void CTFWriterSpec::storeDictionary(DetID det, CTFHeader& header)
153 {
154  if (!isPresent(det) || !mFreqsAccumulation[det].size()) {
155  return;
156  }
157  prepareDictionaryTreeAndFile(det);
158  // create vector whose data contains dictionary in CTF format (EncodedBlock)
159  auto dictBlocks = C::createDictionaryBlocks(mFreqsAccumulation[det], mFreqsMetaData[det]);
160  auto& h = C::get(dictBlocks.data())->getHeader();
161  h = *reinterpret_cast<typename std::remove_reference<decltype(h)>::type*>(mHeaders[det].get());
162  C::get(dictBlocks.data())->print(o2::utils::Str::concat_string("Storing dictionary for ", det.getName(), ": "));
163  C::get(dictBlocks.data())->appendToTree(*mDictTreeOut.get(), det.getName()); // cast to EncodedBlock
164  // mFreqsAccumulation[det].clear();
165  // mFreqsMetaData[det].clear();
166  if (mDictPerDetector) {
167  header.detectors.reset();
168  }
169  header.detectors.set(det);
170  if (mDictPerDetector) {
171  closeDictionaryTreeAndFile(header);
172  }
173 }

175 CTFWriterSpec::CTFWriterSpec(DetID::mask_t dm, uint64_t r, bool doCTF, bool doDict, bool dictPerDet)
176  : mDets(dm), mRun(r), mWriteCTF(doCTF), mCreateDict(doDict), mDictPerDetector(dictPerDet)
177 {
178  mTimer.Stop();
179  mTimer.Reset();
180 
181  if (doDict) { // make sure that there is no local dictonary
182  for (int id = 0; id < DetID::nDetectors; id++) {
183  DetID det(id);
184  if (isPresent(det)) {
185  auto dictName = dictionaryFileName(det.getName());
186  if (std::filesystem::exists(dictName)) {
187  throw std::runtime_error(o2::utils::Str::concat_string("CTF dictionary creation is requested but ", dictName, " already exists, remove it!"));
188  }
189  if (!mDictPerDetector) {
190  break; // no point in checking further
191  }
192  }
193  }
194  }

204 void CTFWriterSpec::run(ProcessingContext& pc)
205 {
206  auto cput = mTimer.CpuTime();
207  mTimer.Start(false);
208  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getByPos(0));
209 
210  std::unique_ptr<TFile> fileOut;
211  std::unique_ptr<TTree> treeOut;
212  if (mWriteCTF) {
213  fileOut.reset(TFile::Open(o2::utils::Str::concat_string(mCTFDir, o2::base::NameConf::getCTFFileName(dh->runNumber, dh->firstTForbit, dh->tfCounter)).c_str(), "recreate"));
214  treeOut = std::make_unique<TTree>(std::string(o2::base::NameConf::CTFTREENAME).c_str(), "O2 CTF tree");
215  }
216 
217  // create header
218  CTFHeader header{mRun, dh->firstTForbit};
219 
220  processDet<o2::itsmft::CTF>(pc, DetID::ITS, header, treeOut.get());
221  processDet<o2::itsmft::CTF>(pc, DetID::MFT, header, treeOut.get());
222  processDet<o2::tpc::CTF>(pc, DetID::TPC, header, treeOut.get());
223  processDet<o2::trd::CTF>(pc, DetID::TRD, header, treeOut.get());
224  processDet<o2::tof::CTF>(pc, DetID::TOF, header, treeOut.get());
225  processDet<o2::ft0::CTF>(pc, DetID::FT0, header, treeOut.get());
226  processDet<o2::fv0::CTF>(pc, DetID::FV0, header, treeOut.get());
227  processDet<o2::fdd::CTF>(pc, DetID::FDD, header, treeOut.get());
228  processDet<o2::mid::CTF>(pc, DetID::MID, header, treeOut.get());
229  processDet<o2::mch::CTF>(pc, DetID::MCH, header, treeOut.get());
230  processDet<o2::emcal::CTF>(pc, DetID::EMC, header, treeOut.get());
231  processDet<o2::phos::CTF>(pc, DetID::PHS, header, treeOut.get());
232  processDet<o2::cpv::CTF>(pc, DetID::CPV, header, treeOut.get());
233  processDet<o2::zdc::CTF>(pc, DetID::ZDC, header, treeOut.get());
234  processDet<o2::hmpid::CTF>(pc, DetID::HMP, header, treeOut.get());
235 
236  mTimer.Stop();
237 
238  if (mWriteCTF) {
239  appendToTree(*treeOut.get(), "CTFHeader", header);
240  treeOut->SetEntries(1);
241  treeOut->Write();
242  treeOut.reset();
243  fileOut->Close();
244  LOG(INFO) << "TF#" << mNTF << ": wrote " << fileOut->GetName() << " with CTF{" << header << "} in " << mTimer.CpuTime() - cput << " s";
245  } else {
246  LOG(INFO) << "TF#" << mNTF << " CTF writing is disabled";
247  }
248  mNTF++;
249  if (mCreateDict && mSaveDictAfter > 0 && (mNTF % mSaveDictAfter) == 0) {
250  storeDictionaries();
251  }
252 }

265 void CTFWriterSpec::prepareDictionaryTreeAndFile(DetID det)
266 {
267  if (mDictPerDetector) {
268  if (mDictTreeOut) {
269  mDictTreeOut->SetEntries(1);
270  mDictTreeOut->Write();
271  mDictTreeOut.reset();
272  mDictFileOut.reset();
273  }
274  }
275  if (!mDictTreeOut) {
276  mDictFileOut.reset(TFile::Open(dictionaryFileName(det.getName()).c_str(), "recreate"));
277  mDictTreeOut = std::make_unique<TTree>(std::string(o2::base::NameConf::CTFDICT).c_str(), "O2 CTF dictionary");
278  }

330 DataProcessorSpec getCTFWriterSpec(DetID::mask_t dets, uint64_t run, bool doCTF, bool doDict, bool dictPerDet)
331 {
332  std::vector<InputSpec> inputs;
333  LOG(INFO) << "Detectors list:";
334  for (auto id = DetID::First; id <= DetID::Last; id++) {
335  if (dets[id]) {
336  inputs.emplace_back(DetID::getName(id), DetID::getDataOrigin(id), "CTFDATA", 0, Lifetime::Timeframe);
337  LOG(INFO) << "Det " << DetID::getName(id) << " added";
338  }
339  }
340  return DataProcessorSpec{
341  "ctf-writer",
342  inputs,
343  Outputs{},
344  AlgorithmSpec{adaptFromTask<CTFWriterSpec>(dets, run, doCTF, doDict, dictPerDet)},
345  Options{{"save-dict-after", VariantType::Int, -1, {"In dictionary generation mode save it dictionary after certain number of TFs processed"}},
346  {"ctf-dict-dir", VariantType::String, "none", {"CTF dictionary directory"}},
347  {"output-dir", VariantType::String, "none", {"CTF output directory"}}}};
348 }
