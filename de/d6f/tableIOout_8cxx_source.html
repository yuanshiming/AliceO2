40 struct ProduceMinMaxPt {
41  Produces<aod::MinMaxPt> minmaxpt;
42  float minpt;
43  float maxpt;
44 
45  void process(aod::Collision const& collision, aod::Tracks const& tracks)
46  {
47  // reset minpt / maxpt
48  minpt = 1.E3;
49  maxpt = 0.;
50 
51  // loop over tracks of the collision
52  LOGF(info, "Collision %d number of tracks %d", collision.index(), tracks.size());
53  for (auto& track : tracks) {
54  if (track.pt() < minpt) {
55  minpt = track.pt();
56  }
57  if (track.pt() > maxpt) {
58  maxpt = track.pt();
59  }
60  }
61  LOGF(info, " ptmin %f ptmax %f", minpt, maxpt);
62 
63  // update table minmax
64  minmaxpt(minpt, maxpt);
65  }
66 };
