33  template <typename T>
34  void makelogaxis(T h)
35  {
36  // return;
37  const int nbins = h->GetNbinsX();
38  double binp[nbins + 1];
39  double max = h->GetXaxis()->GetBinUpEdge(nbins);
40  double min = h->GetXaxis()->GetBinLowEdge(1);
41  if (min <= 0) {
42  min = 0.00001;
43  }
44  double lmin = TMath::Log10(min);
45  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
46  for (int i = 0; i < nbins; i++) {
47  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
48  }
49  binp[nbins] = max + 1;
50  h->GetXaxis()->Set(nbins, binp);
51  }

65  using TrackCandidates = soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection>;
66  void process(const TrackCandidates& tracks,
67  const aod::HMPIDs& hmpids,
68  const aod::Collisions& colls)
69  {
70  for (const auto& t : hmpids) {
71 
72  if (t.track_as<TrackCandidates>().isGlobalTrack() != (uint8_t) true) {
73  continue;
74  }
75  if (abs(t.track_as<TrackCandidates>().dcaXY()) > maxDCA) {
76  continue;
77  }
78  histos.fill(HIST("distance/nonselected"), t.hmpidDistance());
79  histos.fill(HIST("qmip/nonselected"), t.hmpidQMip());
80  histos.fill(HIST("nphotons/nonselected"), t.hmpidNPhotons());
81  if (t.hmpidDistance() > 5.f) {
82  continue;
83  }
84  if (t.hmpidQMip() < 120.f) {
85  continue;
86  }
87  histos.fill(HIST("distance/selected"), t.hmpidDistance());
88  histos.fill(HIST("qmip/selected"), t.hmpidQMip());
89  histos.fill(HIST("nphotons/selected"), t.hmpidNPhotons());
90  histos.fill(HIST("qa/signalvsP"), t.track_as<TrackCandidates>().p(), t.hmpidSignal());
91  }
92  }
