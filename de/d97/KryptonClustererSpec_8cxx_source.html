45 class KrBoxClusterFinderDevice : public o2::framework::Task
46 {
47  public:
48  KrBoxClusterFinderDevice(int lane, const std::vector<int>& sectors) : mClusters(36), mLane{lane}, mSectors(sectors), mClusterFinder{std::make_unique<KrBoxClusterFinder>()} {}
49 
50  void init(o2::framework::InitContext& ic) final
51  {
52  // set up ADC value filling
53  mRawReader.createReader("");
54 
55  mRawReader.setLinkZSCallback([this](int cru, int rowInSector, int padInRow, int timeBin, float adcValue) -> bool {
56  const int sector = cru / 10;
57  if ((mLastSector > -1) && (sector != mLastSector)) {
58  LOGP(debug, "analysing sector {} ({})", mLastSector, sector);
59  mClusterFinder->findLocalMaxima(true);
60  LOGP(info, "found {} clusters in sector {}", mClusterFinder->getClusters().size(), mLastSector);
61  std::swap(mClusters[mLastSector], mClusterFinder->getClusters());
62  mClusterFinder->resetADCMap();
63  mClusterFinder->resetClusters();
64  }
65 
66  mClusterFinder->fillADCValue(cru, rowInSector, padInRow, timeBin, adcValue);
67 
68  mLastSector = sector;
69  return true;
70  });
71 
72  mMaxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
73  mForceQuit = ic.options().get<bool>("force-quit");
74  }
75 
76  void run(o2::framework::ProcessingContext& pc) final
77  {
78  // in case the maximum number of events was reached don't do further processing
79  if (mReadyToQuit) {
80  return;
81  }
82 
83  std::for_each(mClusters.begin(), mClusters.end(), [](auto& cl) { cl.clear(); });
84 
85  auto& reader = mRawReader.getReaders()[0];
86  mActiveSectors = calib_processing_helper::processRawData(pc.inputs(), reader, false, mSectors);
87 
88  // analyse the final sector
89  if (mLastSector > -1) {
90  LOGP(debug, "analysing sector {}", mLastSector);
91  mClusterFinder->findLocalMaxima(true);
92  if (mClusterFinder->getClusters().size()) {
93  LOGP(info, "found {} clusters in sector {}", mClusterFinder->getClusters().size(), mLastSector);
94  std::swap(mClusters[mLastSector], mClusterFinder->getClusters());
95  }
96  }
97  mClusterFinder->resetADCMap();
98  mClusterFinder->resetClusters();
99 
100  ++mProcessedTFs;
101  LOGP(info, "Number of processed time frames: {} ({})", mProcessedTFs, mMaxEvents);
102 
103  snapshotClusters(pc.outputs());
104 
105  // TODO: is this still needed?
106  if (mMaxEvents && (mProcessedTFs >= mMaxEvents)) {
107  LOGP(info, "Maximm number of time frames reached ({}), no more processing will be done", mMaxEvents);
108  mReadyToQuit = true;
109  if (mForceQuit) {
110  pc.services().get<ControlService>().endOfStream();
111  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
112  } else {
113  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
114  }
115  }
116  }
117 
118  void endOfStream(o2::framework::EndOfStreamContext& ec) final
119  {
120  LOGP(info, "endOfStream");
121  if (mActiveSectors) {
122  snapshotClusters(ec.outputs());
123  }
124  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
125  }
126 
127  private:
128  std::vector<std::vector<KrCluster>> mClusters;
129  std::unique_ptr<KrBoxClusterFinder> mClusterFinder;
130  int mLastSector{-1};
131  rawreader::RawReaderCRUManager mRawReader;
132  int mLane{0};
133  std::vector<int> mSectors{};
134  uint32_t mMaxEvents{0};
135  uint32_t mProcessedTFs{0};
136  bool mReadyToQuit{false};
137  bool mCalibDumped{false};
138  bool mForceQuit{false};
139  uint64_t mActiveSectors{0};

141  //____________________________________________________________________________
142  void snapshotClusters(DataAllocator& output)
143  {
144  for (const int sector : mSectors) {
145  o2::tpc::TPCSectorHeader header{sector};
146  header.activeSectors = mActiveSectors;
147  // digit for now are transported per sector, not per lane
148  output.snapshot(Output{"TPC", "KRCLUSTERS", static_cast<SubSpecificationType>(sector), Lifetime::Timeframe, header},
149  mClusters[sector]);
150  }
151  mActiveSectors = 0;
152  }
153 };

155 o2::framework::DataProcessorSpec getKryptonClustererSpec(const std::string inputSpec, int ilane, std::vector<int> const& sectors)
156 {
157  using device = o2::tpc::KrBoxClusterFinderDevice;
158 
159  std::vector<OutputSpec> outputs;
160  for (auto isector : sectors) {
161  outputs.emplace_back("TPC", "KRCLUSTERS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
162  }
163 
164  return DataProcessorSpec{
165  fmt::format("tpc-krypton-clusterer-{}", ilane),
166  select(inputSpec.data()),
167  outputs,
168  AlgorithmSpec{adaptFromTask<device>(ilane, sectors)},
169  Options{
170  {"max-events", VariantType::Int, 0, {"maximum number of events to process"}},
171  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
172  } // end Options
173  }; // end DataProcessorSpec
