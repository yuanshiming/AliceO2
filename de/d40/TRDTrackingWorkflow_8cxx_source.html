31 framework::WorkflowSpec getTRDTrackingWorkflow(bool disableRootInp, bool disableRootOut, GTrackID::mask_t srcTRD)
32 {
33  framework::WorkflowSpec specs;
34  bool useMC = false;
35  if (!disableRootInp) {
36  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, srcTRD)) {
37  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(useMC));
38  }
39  if (GTrackID::includesSource(GTrackID::Source::TPC, srcTRD)) {
40  specs.emplace_back(o2::tpc::getTPCTrackReaderSpec(useMC));
41  }
42  specs.emplace_back(o2::trd::getTRDTrackletReaderSpec(useMC, false));
43  }
44 
45  specs.emplace_back(o2::trd::getTRDTrackletTransformerSpec());
46  specs.emplace_back(o2::trd::getTRDGlobalTrackingSpec(useMC, srcTRD));
47 
48  if (!disableRootOut) {
49  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, srcTRD)) {
50  specs.emplace_back(o2::trd::getTRDGlobalTrackWriterSpec(useMC));
51  }
52  if (GTrackID::includesSource(GTrackID::Source::TPC, srcTRD)) {
53  specs.emplace_back(o2::trd::getTRDTPCTrackWriterSpec(useMC));
54  }
55  }
56  return specs;
57 }
