76  template <o2::track::PID::ID id>
77  float sigma(Trks::iterator track)
78  {
79  return o2::pid::tof::TOFResoALICE3ParamTrack<id>(track.collision(), track, resoParameters);
80  }
81  template <o2::track::PID::ID id>
82  float nsigma(Trks::iterator track)
83  {
84  // return (track.tofSignal() - tof::ExpTimes<Coll::iterator, Trks::iterator, id>::GetExpectedSignal(track.collision(), track)) / sigma<id>(track);
85  return (track.tofSignal() - track.collision().collisionTime() * 1000.f - tof::ExpTimes<Coll::iterator, Trks::iterator, id>::GetExpectedSignal(track.collision(), track)) / sigma<id>(track);
86  }
87  void process(Coll const& collisions, Trks const& tracks)
88  {
89  tablePIDEl.reserve(tracks.size());
90  tablePIDMu.reserve(tracks.size());
91  tablePIDPi.reserve(tracks.size());
92  tablePIDKa.reserve(tracks.size());
93  tablePIDPr.reserve(tracks.size());
94  tablePIDDe.reserve(tracks.size());
95  tablePIDTr.reserve(tracks.size());
96  tablePIDHe.reserve(tracks.size());
97  tablePIDAl.reserve(tracks.size());
98  for (auto const& trk : tracks) {
99  tablePIDEl(sigma<PID::Electron>(trk), nsigma<PID::Electron>(trk));
100  tablePIDMu(sigma<PID::Muon>(trk), nsigma<PID::Muon>(trk));
101  tablePIDPi(sigma<PID::Pion>(trk), nsigma<PID::Pion>(trk));
102  tablePIDKa(sigma<PID::Kaon>(trk), nsigma<PID::Kaon>(trk));
103  tablePIDPr(sigma<PID::Proton>(trk), nsigma<PID::Proton>(trk));
104  tablePIDDe(sigma<PID::Deuteron>(trk), nsigma<PID::Deuteron>(trk));
105  tablePIDTr(sigma<PID::Triton>(trk), nsigma<PID::Triton>(trk));
106  tablePIDHe(sigma<PID::Helium3>(trk), nsigma<PID::Helium3>(trk));
107  tablePIDAl(sigma<PID::Alpha>(trk), nsigma<PID::Alpha>(trk));
108  }

145  template <typename T>
146  void makelogaxis(T h)
147  {
148  if (logAxis == 0) {
149  return;
150  }
151  const int nbins = h->GetNbinsX();
152  double binp[nbins + 1];
153  double max = h->GetXaxis()->GetBinUpEdge(nbins);
154  double min = h->GetXaxis()->GetBinLowEdge(1);
155  if (min <= 0) {
156  min = 0.00001;
157  }
158  double lmin = TMath::Log10(min);
159  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
160  for (int i = 0; i < nbins; i++) {
161  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
162  }
163  binp[nbins] = max + 1;
164  h->GetXaxis()->Set(nbins, binp);
165  }

214  void fillParticleHistos(const T& t, const float& tof, const float& exp_diff, const float& expsigma, const float& nsigma)
215  {
216  histos.fill(HIST(hexpected[i]), t.p(), tof - exp_diff);
217  histos.fill(HIST(hexpected_diff[i]), t.p(), exp_diff);

222  void process(aod::Collision const& collision,
223  soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksCov,
224  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
225  aod::pidTOFFullKa, aod::pidTOFFullPr, aod::pidTOFFullDe,
226  aod::pidTOFFullTr, aod::pidTOFFullHe, aod::pidTOFFullAl,
227  aod::TrackSelection> const& tracks)
228  {
229  // Computing Multiplicity first
230  int mult = 0;
231  for (auto t : tracks) {
232  //
233  if (t.tofSignal() < 0) { // Skipping tracks without TOF
234  continue;
235  }
236  mult++;
237  }
238  if (mult < MinMult) { // Cutting on low multiplicity events
239  return;
240  }
241  const float collisionTime_ps = collision.collisionTime() * 1000.f;
242  histos.fill(HIST("event/vertexz"), collision.posZ());
243  histos.fill(HIST("event/colltime"), collisionTime_ps);
244  histos.fill(HIST("event/tofmultiplicity"), mult);
245  histos.fill(HIST("event/colltimereso"), mult, collision.collisionTimeRes() * 1000.f);
246  for (auto t : tracks) {
247  //
248  if (t.tofSignal() < 0) { // Skipping tracks without TOF
249  continue;
250  }
251  if (!t.isGlobalTrack()) {
252  continue;
253  }
254 
255  const float tof = t.tofSignal() - collisionTime_ps;
256 
257  //
258  histos.fill(HIST("event/tofsignal"), t.p(), t.tofSignal());
259  histos.fill(HIST("event/eta"), t.eta());
260  histos.fill(HIST("event/length"), t.length());
261  histos.fill(HIST("event/pt"), t.pt());
262  histos.fill(HIST("event/ptreso"), t.p(), t.sigma1Pt() * t.pt() * t.pt());
263  //
264  fillParticleHistos<0>(t, tof, t.tofExpSignalDiffEl(), t.tofExpSigmaEl(), t.tofNSigmaEl());
265  fillParticleHistos<1>(t, tof, t.tofExpSignalDiffMu(), t.tofExpSigmaMu(), t.tofNSigmaMu());
266  fillParticleHistos<2>(t, tof, t.tofExpSignalDiffPi(), t.tofExpSigmaPi(), t.tofNSigmaPi());
267  fillParticleHistos<3>(t, tof, t.tofExpSignalDiffKa(), t.tofExpSigmaKa(), t.tofNSigmaKa());
268  fillParticleHistos<4>(t, tof, t.tofExpSignalDiffPr(), t.tofExpSigmaPr(), t.tofNSigmaPr());
269  fillParticleHistos<5>(t, tof, t.tofExpSignalDiffDe(), t.tofExpSigmaDe(), t.tofNSigmaDe());
270  fillParticleHistos<6>(t, tof, t.tofExpSignalDiffTr(), t.tofExpSigmaTr(), t.tofNSigmaTr());
271  fillParticleHistos<7>(t, tof, t.tofExpSignalDiffHe(), t.tofExpSigmaHe(), t.tofNSigmaHe());
272  fillParticleHistos<8>(t, tof, t.tofExpSignalDiffAl(), t.tofExpSigmaAl(), t.tofNSigmaAl());
273  }
274  }
