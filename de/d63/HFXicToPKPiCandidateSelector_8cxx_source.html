69  template <typename T>
70  bool selectionTopol(const T& candidate)
71  {
72  auto candpT = candidate.pt();
73  int pTBin = findBin(pTBins, candpT);
74  if (pTBin == -1) {
75  return false;
76  }
77 
78  // check that the candidate pT is within the analysis range
79  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
80  return false;
81  }
82 
83  // cosine of pointing angle
84  if (candidate.cpa() <= cuts->get(pTBin, "cos pointing angle")) {
85  return false;
86  }
87 
88  // candidate DCA
89  /* if (candidate.chi2PCA() > cuts->get(pTBin, "DCA")) {
90  return false;
91  }*/
92 
93  if (candidate.decayLength() <= cuts->get(pTBin, "decay length")) {
94  return false;
95  }
96  return true;
97  }

105  template <typename T1, typename T2>
106  bool selectionTopolConjugate(const T1& candidate, const T2& trackProton, const T2& trackKaon, const T2& trackPion)
107  {
108 
109  auto candpT = candidate.pt();
110  int pTBin = findBin(pTBins, candpT);
111  if (pTBin == -1) {
112  return false;
113  }
114 
115  // cut on daughter pT
116  if (trackProton.pt() < cuts->get(pTBin, "pT p") || trackKaon.pt() < cuts->get(pTBin, "pT K") || trackPion.pt() < cuts->get(pTBin, "pT Pi")) {
117  return false;
118  }
119 
120  if (trackProton.globalIndex() == candidate.index0Id()) {
121  if (std::abs(InvMassXicToPKPi(candidate) - RecoDecay::getMassPDG(pdg::Code::kXiCPlus)) > cuts->get(pTBin, "m")) {
122  return false;
123  }
124  } else {
125  if (std::abs(InvMassXicToPiKP(candidate) - RecoDecay::getMassPDG(pdg::Code::kXiCPlus)) > cuts->get(pTBin, "m")) {
126  return false;
127  }
128  }
129 
130  return true;
131  }

133  void process(aod::HfCandProng3 const& candidates, aod::BigTracksPID const&)
134  {
135  TrackSelectorPID selectorPion(kPiPlus);
136  selectorPion.setRangePtTPC(d_pidTPCMinpT, d_pidTPCMaxpT);
137  selectorPion.setRangeNSigmaTPC(-d_nSigmaTPC, d_nSigmaTPC);
138  selectorPion.setRangeNSigmaTPCCondTOF(-d_nSigmaTPCCombined, d_nSigmaTPCCombined);
139  selectorPion.setRangePtTOF(d_pidTOFMinpT, d_pidTOFMaxpT);
140  selectorPion.setRangeNSigmaTOF(-d_nSigmaTOF, d_nSigmaTOF);
141  selectorPion.setRangeNSigmaTOFCondTPC(-d_nSigmaTOFCombined, d_nSigmaTOFCombined);
142 
143  TrackSelectorPID selectorKaon(selectorPion);
144  selectorKaon.setPDG(kKPlus);
145 
146  TrackSelectorPID selectorProton(selectorPion);
147  selectorProton.setPDG(kProton);
148 
149  // looping over 3-prong candidates
150  for (auto& candidate : candidates) {
151 
152  // final selection flag: 0 - rejected, 1 - accepted
153  auto statusXicToPKPi = 0;
154  auto statusXicToPiKP = 0;
155 
156  if (!(candidate.hfflag() & 1 << DecayType::XicToPKPi)) {
157  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
158  continue;
159  }
160 
161  auto trackPos1 = candidate.index0_as<aod::BigTracksPID>(); // positive daughter (negative for the antiparticles)
162  auto trackNeg = candidate.index1_as<aod::BigTracksPID>(); // negative daughter (positive for the antiparticles)
163  auto trackPos2 = candidate.index2_as<aod::BigTracksPID>(); // positive daughter (negative for the antiparticles)
164 
165  /*
166  // daughter track validity selection
167  if (!daughterSelection(trackPos1) || !daughterSelection(trackNeg) || !daughterSelection(trackPos2)) {
168  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
169  continue;
170  }
171  */
172 
173  // implement filter bit 4 cut - should be done before this task at the track selection level
174 
175  // conjugate-independent topological selection
176  if (!selectionTopol(candidate)) {
177  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
178  continue;
179  }
180 
181  // conjugate-dependent topplogical selection for Xic
182 
183  bool topolXicToPKPi = selectionTopolConjugate(candidate, trackPos1, trackNeg, trackPos2);
184  bool topolXicToPiKP = selectionTopolConjugate(candidate, trackPos2, trackNeg, trackPos1);
185 
186  if (!topolXicToPKPi && !topolXicToPiKP) {
187  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
188  continue;
189  }
190 
191  auto pidXicToPKPi = -1;
192  auto pidXicToPiKP = -1;
193 
194  if (!d_FilterPID) {
195  // PID non applied
196  pidXicToPKPi = 1;
197  pidXicToPiKP = 1;
198  } else {
199  // track-level PID selection
200  int pidTrackPos1Proton = selectorProton.getStatusTrackPIDAll(trackPos1);
201  int pidTrackPos2Proton = selectorProton.getStatusTrackPIDAll(trackPos2);
202  int pidTrackPos1Pion = selectorPion.getStatusTrackPIDAll(trackPos1);
203  int pidTrackPos2Pion = selectorPion.getStatusTrackPIDAll(trackPos2);
204  int pidTrackNegKaon = selectorKaon.getStatusTrackPIDAll(trackNeg);
205 
206  if (pidTrackPos1Proton == TrackSelectorPID::Status::PIDAccepted &&
207  pidTrackNegKaon == TrackSelectorPID::Status::PIDAccepted &&
208  pidTrackPos2Pion == TrackSelectorPID::Status::PIDAccepted) {
209  pidXicToPKPi = 1; // accept LcpKpi
210  } else if (pidTrackPos1Proton == TrackSelectorPID::Status::PIDRejected ||
211  pidTrackNegKaon == TrackSelectorPID::Status::PIDRejected ||
212  pidTrackPos2Pion == TrackSelectorPID::Status::PIDRejected) {
213  pidXicToPKPi = 0; // exclude LcpKpi
214  }
215  if (pidTrackPos2Proton == TrackSelectorPID::Status::PIDAccepted &&
216  pidTrackNegKaon == TrackSelectorPID::Status::PIDAccepted &&
217  pidTrackPos1Pion == TrackSelectorPID::Status::PIDAccepted) {
218  pidXicToPiKP = 1; // accept LcpiKp
219  } else if (pidTrackPos1Pion == TrackSelectorPID::Status::PIDRejected ||
220  pidTrackNegKaon == TrackSelectorPID::Status::PIDRejected ||
221  pidTrackPos2Proton == TrackSelectorPID::Status::PIDRejected) {
222  pidXicToPiKP = 0; // exclude LcpiKp
223  }
224  }
225 
226  if (pidXicToPKPi == 0 && pidXicToPiKP == 0) {
227  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
228  continue;
229  }
230 
231  if ((pidXicToPKPi == -1 || pidXicToPKPi == 1) && topolXicToPKPi) {
232  statusXicToPKPi = 1; // identified as Xic->pKpi
233  }
234  if ((pidXicToPiKP == -1 || pidXicToPiKP == 1) && topolXicToPiKP) {
235  statusXicToPiKP = 1; // identified as Xic->piKp
236  }
237 
238  hfSelXicToPKPiCandidate(statusXicToPKPi, statusXicToPiKP);
239  }
240  }
