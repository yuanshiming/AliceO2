59 void AlignableDetectorTRD::defineVolumes()
60 {
61  // define TRD volumes
62  //
63  const int kNSect = 18, kNStacks = 5, kNLayers = 6;
64  AlignableSensorTRD* chamb = 0;
65  //
66  int labDet = getDetLabel();
67  // AddVolume( volTRD = new AlignableVolume("TRD") ); // no main volume, why?
68  AlignableVolume* sect[kNSect] = {0};
69  //
70  for (int ilr = 0; ilr < kNLayers; ilr++) { // layer
71  for (int ich = 0; ich < kNStacks * kNSect; ich++) { // chamber
72  int isector = ich / AliTRDgeometry::Nstack();
73  int istack = ich % AliTRDgeometry::Nstack();
74  //int lid = AliTRDgeometry::GetDetector(ilr,istack,isector);
75  int iid = labDet + (1 + ilr) * 10000 + (1 + isector) * 100 + (1 + istack);
76  const char* symname = Form("TRD/sm%02d/st%d/pl%d", isector, istack, ilr);
77  if (!gGeoManager->GetAlignableEntry(symname))
78  continue;
79  uint16_t vid = AliGeomManager::LayerToVolUID(AliGeomManager::kTRD1 + ilr, ich);
80  addVolume(chamb = new AlignableSensorTRD(symname, vid, iid /*lid*/, isector));
81  iid = labDet + (1 + isector) * 100;
82  if (!sect[isector])
83  sect[isector] = new AlignableVolume(Form("TRD/sm%02d", isector), iid);
84  chamb->setParent(sect[isector]);
85  } // chamber
86  } // layer
87  //
88  for (int isc = 0; isc < kNSect; isc++) {
89  if (sect[isc])
90  addVolume(sect[isc]);
91  }
92  //
93 }

122 void AlignableDetectorTRD::writePedeInfo(FILE* parOut, const Option_t* opt) const
123 {
124  // contribute to params and constraints template files for PEDE
125  AlignableDetector::writePedeInfo(parOut, opt);
126  //
127  // write calibration parameters
128  enum { kOff,
129  kOn,
130  kOnOn };
131  const char* comment[3] = {" ", "! ", "!!"};
132  const char* kKeyParam = "parameter";
133  //
134  fprintf(parOut, "%s%s %s\t %d calibraction params for %s\n", comment[kOff], kKeyParam, comment[kOnOn],
135  getNCalibDOFs(), GetName());
136  //
137  for (int ip = 0; ip < getNCalibDOFs(); ip++) {
138  int cmt = isCondDOF(ip) ? kOff : kOn;
139  fprintf(parOut, "%s %9d %+e %+e\t%s %s p%d\n", comment[cmt], getParLab(ip),
140  getParVal(ip), getParErr(ip), comment[kOnOn], isFreeDOF(ip) ? " " : "FX", ip);
141  }
142  //
143 }
