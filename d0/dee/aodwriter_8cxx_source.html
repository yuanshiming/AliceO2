123 struct aodWriter {
124  Produces<aod::Uno> table_uno;
125  Produces<aod::Due> table_due;
126  Produces<aod::Tre> table_tre;
127 
128  void init(InitContext&)
129  {
130  cnt = 0;
131  }
132 
133  void process(aod::Tracks const& tracks)
134  {
135  for (auto& track : tracks) {
136  bool ok = (cnt % 2) == 0;
137  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
138  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
139  double mom = track.tgl();
140  double pt = track.signed1Pt();
141  int id = (int)cnt;
142 
143  table_uno(phi, eta, mom);
144  table_due(ok, phi, eta, mom, pt);
145  table_tre(phi, eta, mom, id);
146  //LOGF(INFO, "Values (%i): (%i %f, %f, %f, %f, %i)", cnt, ok, eta, phi, mom, pt, id);
147  cnt++;
148  }
149 
150  LOGF(INFO, "ATask Processed %i data points from Tracks", cnt);
151  }
152 
153  size_t cnt = 0;
154 };
