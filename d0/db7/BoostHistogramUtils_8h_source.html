176 template <typename T, typename Iterator, typename AxisIterator>
177 fitResult<T, 4> fitGaus(Iterator first, Iterator last, AxisIterator axisfirst)
178 {
179  TLinearFitter fitter(3, "pol2");
180  TMatrixD mat(3, 3);
181  Double_t kTol = mat.GetTol();
182  fitter.StoreData(kFALSE);
183  fitter.ClearPoints();
184  TVectorD par(3);
185  TVectorD sigma(3);
186  TMatrixD A(3, 3);
187  TMatrixD b(3, 1);
188  T rms = TMath::RMS(first, last);
189  T xMax = std::max_element(first, last);
190  T xMin = std::min_element(first, last);
191  auto nbins = last - first;
192  const double binWidth = double(xMax - xMin) / double(nbins);
193 
194  Float_t meanCOG = 0;
195  Float_t rms2COG = 0;
196  Float_t sumCOG = 0;
197 
198  Float_t entries = 0;
199  Int_t nfilled = 0;
200 
201  for (auto iter = first, axisiter = axisfirst; iter != last; iter++, axisiter++) {
202  entries += *iter;
203  if (*iter > 0) {
204  nfilled++;
205  }
206  }
207 
208  // TODO: Check why this is needed
209  if (xMax < 4) {
210  throw FitGausError_t::FIT_ERROR;
211  }
212  if (entries < 12) {
213  throw FitGausError_t::FIT_ERROR;
214  }
215 
216  if (rms < kTol) {
217  throw FitGausError_t::FIT_ERROR;
218  }
219 
220  fitResult<T, 4> result;
221  result.setParameter<3>(entries);
222 
223  int ibin = 0;
224  Int_t npoints = 0;
225  for (auto iter = first, axisiter = axisfirst; iter != last; iter++, axisiter++) {
226  if (nbins > 1) {
227  Double_t x = (*axisiter + *(axisiter + 1)) / 2.;
228  Double_t y = *iter;
229  Double_t ey = std::sqrt(y);
230  fitter.AddPoint(&x, y, ey);
231  if (npoints < 3) {
232  A(npoints, 0) = 1;
233  A(npoints, 1) = x;
234  A(npoints, 2) = x * x;
235  b(npoints, 0) = y;
236  meanCOG += x * nbins;
237  rms2COG += x * nbins * x;
238  sumCOG += nbins;
239  }
240  npoints++;
241  }
242  }
243 
244  Double_t chi2 = 0;
245  if (npoints >= 3) {
246  if (npoints == 3) {
247  //analytic calculation of the parameters for three points
248  A.Invert();
249  TMatrixD res(1, 3);
250  res.Mult(A, b);
251  par[0] = res(0, 0);
252  par[1] = res(0, 1);
253  par[2] = res(0, 2);
254  chi2 = -3.;
255  } else {
256  // use fitter for more than three points
257  fitter.Eval();
258  fitter.GetParameters(par);
259  fitter.GetCovarianceMatrix(mat);
260  result.setChi2(fitter.GetChisquare() / Double_t(npoints));
261  }
262  if (TMath::Abs(par[1]) < kTol) {
263  throw FitGausError_t::FIT_ERROR;
264  ;
265  }
266  if (TMath::Abs(par[2]) < kTol) {
267  throw FitGausError_t::FIT_ERROR;
268  ;
269  }
270 
271  T param1 = T(par[1] / (-2. * par[2]));
272  result.setParameter<1>(param1);
273  result.setParameter<2>(T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2]))));
274  auto lnparam0 = par[0] + par[1] * param1 + par[2] * param1 * param1;
275  if (lnparam0 > 307) {
276  throw FitGausError_t::FIT_ERROR;
277  ;
278  }
279  result.setParameter<0>(TMath::Exp(lnparam0));
280 
281  return result;
282  }
283 
284  if (npoints == 2) {
285  //use center of gravity for 2 points
286  meanCOG /= sumCOG;
287  rms2COG /= sumCOG;
288  result.setParameter<0>(xMax);
289  result.setParameter<1>(meanCOG);
290  result.setParameter<2>(TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG)));
291  result.setChi2(-2);
292  }
293  if (npoints == 1) {
294  meanCOG /= sumCOG;
295  result.setParameter<0>(xMax);
296  result.setParameter<1>(meanCOG);
297  result.setParameter<2>(binWidth / TMath::Sqrt(12));
298  result.setChi2(-1);
299  }
300  return result;
301 }
