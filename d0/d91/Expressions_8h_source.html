65 template <typename T>
66 constexpr auto selectArrowType()
67 {
68  if constexpr (std::is_same_v<T, int>) {
69  return atype::INT32;
70  } else if constexpr (std::is_same_v<T, bool>) {
71  return atype::BOOL;
72  } else if constexpr (std::is_same_v<T, float>) {
73  return atype::FLOAT;
74  } else if constexpr (std::is_same_v<T, double>) {
75  return atype::DOUBLE;
76  } else if constexpr (std::is_same_v<T, uint8_t>) {
77  return atype::UINT8;
78  } else if constexpr (std::is_same_v<T, int8_t>) {
79  return atype::INT8;
80  } else if constexpr (std::is_same_v<T, uint16_t>) {
81  return atype::UINT16;
82  } else if constexpr (std::is_same_v<T, int16_t>) {
83  return atype::INT16;
84  } else if constexpr (std::is_same_v<T, int64_t>) {
85  return atype::INT64;
86  } else if constexpr (std::is_same_v<T, uint32_t>) {
87  return atype::UINT32;
88  } else if constexpr (std::is_same_v<T, uint64_t>) {
89  return atype::UINT64;
90  } else {
91  return atype::NA;
92  }
93  O2_BUILTIN_UNREACHABLE();
94 }

128 struct PlaceholderNode : LiteralNode {
129  template <typename T>
130  PlaceholderNode(Configurable<T> v) : LiteralNode{v.value}, name{v.name}
131  {
132  if constexpr (variant_trait_v<typename std::decay<T>::type> != VariantType::Unknown) {
133  retrieve = [](InitContext& context, std::string const& name) { return LiteralNode::var_t{context.options().get<T>(name.c_str())}; };
134  } else {
135  runtime_error("Unknown parameter used in expression.");
136  }
137  }
138 
139  void reset(InitContext& context)
140  {
141  value = retrieve(context, name);
142  }
143 
144  std::string name;
145  LiteralNode::var_t (*retrieve)(InitContext&, std::string const& name);
146 };

370 template <typename... C>
371 std::shared_ptr<gandiva::Projector> createProjectors(framework::pack<C...>, gandiva::SchemaPtr schema)
372 {
373  std::shared_ptr<gandiva::Projector> projector;
374  auto s = gandiva::Projector::Make(
375  schema,
376  {makeExpression(
377  framework::expressions::createExpressionTree(
378  framework::expressions::createOperations(C::Projector()),
379  schema),
380  C::asArrowField())...},
381  &projector);
382  if (s.ok()) {
383  return projector;
384  } else {
385  throw o2::framework::runtime_error_f("Failed to create projector: %s", s.ToString().c_str());
