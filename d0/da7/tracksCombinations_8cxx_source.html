44 struct HashTask {
45  std::vector<float> xBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
46  std::vector<float> yBins{-1.5f, -1.0f, -0.5f, 0.0f, 0.5f, 1.0f, 1.5f};
47  Produces<aod::Hashes> hashes;
48 
49  // Calculate hash for an element based on 2 properties and their bins.
50  int getHash(const std::vector<float>& xBins, const std::vector<float>& yBins, float colX, float colY)
51  {
52  for (int i = 0; i < xBins.size(); i++) {
53  if (colX < xBins[i]) {
54  for (int j = 0; j < yBins.size(); j++) {
55  if (colY < yBins[j]) {
56  return i + j * (xBins.size() + 1);
57  }
58  }
59  // overflow for yBins only
60  return i + yBins.size() * (xBins.size() + 1);
61  }
62  }
63 
64  // overflow for xBins only
65  for (int j = 0; j < yBins.size(); j++) {
66  if (colY < yBins[j]) {
67  return xBins.size() + j * (xBins.size() + 1);
68  }
69  }
70 
71  // overflow for both bins
72  return (yBins.size() + 1) * (xBins.size() + 1) - 1;
73  }
74 
75  void process(aod::Tracks const& tracks)
76  {
77  for (auto& track : tracks) {
78  int hash = getHash(xBins, yBins, track.x(), track.y());
79  LOGF(info, "Track: %d (%f, %f, %f) hash: %d", track.index(), track.x(), track.y(), track.z(), hash);
80  hashes(hash);
81  }
82  }
83 };
