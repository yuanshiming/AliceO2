85 void Trap2CRU::sortDataToLinks()
86 {
87  //sort the digits array TODO refactor this intoa vector index sort and possibly generalise past merely digits.
88  auto sortstart = std::chrono::high_resolution_clock::now();
89  //build indexes
90  // digits first
91  mDigitsIndex.resize(mDigits.size());
92  std::iota(mDigitsIndex.begin(), mDigitsIndex.end(), 0);
93 
94  for (auto& trig : mDigitTriggerRecords) {
95  if (trig.getNumberOfDigits() != 0) {
96  LOG(debug) << " sorting digits from : " << trig.getFirstDigit() << " till " << trig.getFirstDigit() + trig.getNumberOfDigits();
97  std::stable_sort(mDigitsIndex.begin() + trig.getFirstDigit(), mDigitsIndex.begin() + trig.getNumberOfDigits() + trig.getFirstDigit(), //,digitcompare);
98  [this](const uint32_t i, const uint32_t j) {
99  uint32_t hcida = mDigits[i].getDetector() * 2 + (mDigits[i].getROB() % 2);
100  uint32_t hcidb = mDigits[j].getDetector() * 2 + (mDigits[j].getROB() % 2);
101  if(hcida!=hcidb){return hcida<hcidb;}
102  if(mDigits[i].getROB() != mDigits[j].getROB()){return (mDigits[i].getROB() < mDigits[j].getROB());}
103  return (mDigits[i].getMCM() < mDigits[j].getMCM()); });
104  }
105  }
106  for (auto& trig : mTrackletTriggerRecords) {
107  if (trig.getNumberOfTracklets() > 0) {
108  LOG(debug) << " sorting digits from : " << trig.getFirstTracklet() << " till " << trig.getFirstTracklet() + trig.getNumberOfTracklets();
109  std::stable_sort(std::begin(mTracklets) + trig.getFirstTracklet(), std::begin(mTracklets) + trig.getNumberOfTracklets() + trig.getFirstTracklet(),
110  [this](auto&& t1, auto&& t2) {
111  if (t1.getHCID() != t2.getHCID()) {
112  return t1.getHCID() < t2.getHCID();
113  }
114  if (t1.getPadRow() != t2.getPadRow()) {
115  return t1.getPadRow() < t2.getPadRow();
116  }
117  return t1.getMCM() < t2.getMCM();
118  });
119  }
120  }
121 
122  std::chrono::duration<double> duration = std::chrono::high_resolution_clock::now() - sortstart;
123  LOG(debug) << "TRD Digit/Tracklet Sorting took " << duration.count() << " s";
124 }

148  //
149  mRawData.reserve(1024 * 1024); //TODO take out the hardcoded 1MB its supposed to come in from the options
150  LOG(debug) << "Trap2CRU::readTrapData";
151  // data comes in index by event (triggerrecord)
152  // first 15 links go to cru0a, second 15 links go to cru0b, 3rd 15 links go to cru1a ... first 90 links to flp0 and then repeat for 12 flp
153  // then do next event
154 
155  // lets register our links
156  std::string prefix = mOutputDir;

158  prefix += '/';
159  }
160 
161  for (int link = 0; link < o2::trd::constants::NHALFCRU; link++) {
162  // FeeID *was* 0xFEED, now is indicates the cru Supermodule, side (A/C) and endpoint. See RawData.cxx for details.

170 
171  std::string outFileLink;
172  std::string outPrefix = "trd";
173  std::string outSuffix = ".raw";
174  std::string trdside = mEndPointID ? "_h" : "_l"; // the side of cru lower or higher
175  // filename structure of trd_cru_[CRU#]_[upper/lower].raw
176  std::string whichrun = mUseTrackletHCHeader ? "run3" : "run2";
177  if (mFilePer == "all") { // single file for all links
178  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, outSuffix);
179  } else if (mFilePer == "sm") {
180  int sm = link / 4;
181  std::stringstream ss;
182  ss << std::setw(2) << std::setfill('0') << sm;
183  std::string supermodule = ss.str();
184  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, "_sm_", supermodule, outSuffix);

191  }
192 
193  //std::string outputFilelink = o2::utils::Str::concat_string(prefix, "trd_cru_", std::to_string(mCruID), "_", trdside, "_", whichrun, ".raw");

197 
198  openInputFiles();
199 
200  if (mTrackletsTree->GetEntries() != mDigitsTree->GetEntries()) {
201  LOG(fatal) << "Entry counts in mTrackletsTree and Digits Tree dont match " << mTrackletsTree->GetEntries() << "!=" << mDigitsTree->GetEntries();

205  for (int entry = 0; entry < mTrackletsTree->GetEntries(); entry++) {
206  mTrackletsTree->GetEntry(entry);
207  mDigitsTree->GetEntry(entry);
208  //migrate digit trigger information into the tracklettrigger (historical)
209  mergetriggerDigitRanges(); // merge data (if needed) from the digits trigger record to the tracklets trigger record (different files)
210 
211  //TODO figure out if want to care about unaligned trees.
212  sortDataToLinks();
213  // each entry is a timeframe
214  uint32_t linkcount = 0;
215  for (auto tracklettrigger : mTrackletTriggerRecords) {
216  // for (auto digitstrigger : mDigitTriggerRecords) {
217  // //get the event limits from TriggerRecord;
218  // if (digitstrigger.getBCData() == tracklettrigger.getBCData()) {
219  // convertTrapData(tracklettrigger, digitstrigger, triggercount);
220  // }

228 void Trap2CRU::linkSizePadding(uint32_t linksize, uint32_t& crudatasize, uint32_t& padding)
229 {
230 
231  // if zero the whole 256 bit must be padded (empty link)
232  // crudatasize is the size to be stored in the cruheader, i.e. units of 256bits.
233  // linksize is the incoming link size from the linkrecord,
234  // padding is of course the amount of padding in 32bit words.
235  uint32_t rem = 0;
236  if (linksize != 0) {
237  //data, so figure out padding cru word, the other case is simple, full padding if size=0
238  rem = linksize % 8;
239  if (rem != 0) {
240  crudatasize = linksize / 8 + 1;
241  padding = 8 - rem;
242  } else {
243 
244  crudatasize = linksize / 8; // 32 bit word to 256 bit word.
245  padding = 0;
246  }
247  LOG(debug) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
248  } else {
249  //linksize is zero so no data, pad fully.
250  crudatasize = 1;
251  padding = 8;
252  LOG(debug) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
253  }
254  LOG(debug) << __func__ << " leaving linkSizePadding : CRUDATASIZE : " << crudatasize;
255 }

294 int Trap2CRU::buildDigitRawData(const int digitindex, const std::array<int64_t, o2::trd::constants::NADCMCM>& localParsedDigits, const uint32_t triggerrecordcount)
295 {
296  LOG(debug) << __func__ << " digitindex:" << digitindex << " and triggercount of" << triggerrecordcount;
297  //this is not zero suppressed.
298  int digitswritten = 0;
299  // Digit
300  DigitMCMHeader header;
301  DigitMCMData data;
302  int startdet = mDigits[digitindex].getDetector();
303  int startrob = mDigits[digitindex].getROB();
304  int startmcm = mDigits[digitindex].getMCM();
305  int digitcounter = 0;
306  header.res = 12; //1100
307  header.mcm = startmcm;
308  header.rob = startrob;
309  header.yearflag = 1; // >2007
310  header.eventcount = triggerrecordcount;
311  header.yearflag = 1;
312  memcpy(mRawDataPtr, (char*)&header, 4); // 32 bits -- 4 bytes.
313  LOG(debug) << "writing data to digit stream of " << std::hex << header.word;
314  mRawDataPtr += 4;
315  for (auto digitindex : localParsedDigits) {
316  LOG(debug) << "digit indexof : " << digitindex; //<<
317  if (digitindex != -1) {
318  Digit* d = &mDigits[digitindex];
319  ArrayADC adcdata = d->getADC();
320  //write these 2 now as we only have it now.
321  header.mcm = d->getMCM();
322  header.rob = d->getROB();
323 
324  for (int timebin = 0; timebin < o2::trd::constants::TIMEBINS; timebin += 3) {
325  LOG(debug) << "ADC values : " << timebin << " = " << adcdata[timebin] << ":" << adcdata[timebin + 1] << ":" << adcdata[timebin + 2];
326  data.x = adcdata[timebin];
327  data.y = adcdata[timebin + 1];
328  data.z = adcdata[timebin + 2];
329  data.c = 0;
330  LOG(debug) << "writing data to digit stream of " << std::hex << data.word;
331  memcpy(mRawDataPtr, (char*)&data, 4); // 32 bits -- 4 bytes.
332  mRawDataPtr += 4;
333  }
334  digitswritten++;
335  } else {
336  //blank digit;//10 words of 0 for the 30 timebins, its done this way incase we want to or need to change c.
337  for (int timebin = 0; timebin < o2::trd::constants::TIMEBINS; timebin += 3) {
338  data.x = 0;
339  data.y = 0;
340  data.z = 0;
341  data.c = 0;
342  LOG(debug) << "writing data to digit stream of " << std::hex << data.word;
343  memcpy(mRawDataPtr, (char*)&data, 4); // 32 bits -- 4 bytes.
344  mRawDataPtr += 4;
345  }
346  digitswritten++;
347  }
348  }
349  LOG(debug) << __func__ << " leaving with digitswritten of: " << digitswritten << " for det:rob:mcm of " << startdet << ":" << startrob << ":" << startmcm;
350  return digitswritten;
351 }

353 int Trap2CRU::buildTrackletRawData(const int trackletindex, const int linkid)
354 {
355  TrackletMCMHeader header;
356  bool destroytracklets = false;
357  std::array<TrackletMCMData, 3> trackletdata;
358  header.col = mTracklets[trackletindex].getColumn();
359  header.padrow = mTracklets[trackletindex].getPadRow();
360  header.onea = 1;
361  header.oneb = 1;
362  int trackletcounter = 0;
363  LOG(debug) << header;
364  while (linkid == mTracklets[trackletindex + trackletcounter].getHCID() && header.col == mTracklets[trackletindex + trackletcounter].getColumn() && header.padrow == mTracklets[trackletindex + trackletcounter].getPadRow()) {
365  buildTrackletMCMData(trackletdata[trackletcounter], mTracklets[trackletindex + trackletcounter].getSlope(),
366  mTracklets[trackletindex + trackletcounter].getPosition(), mTracklets[trackletindex + trackletcounter].getQ0(),
367  mTracklets[trackletindex + trackletcounter].getQ1(), mTracklets[trackletindex + trackletcounter].getQ2());
368  int headerqpart = ((mTracklets[trackletindex + trackletcounter].getQ2()) << 2) + ((mTracklets[trackletindex + trackletcounter].getQ1()) >> 5);
369  switch (trackletcounter) {
370  case 0:
371  header.pid0 = headerqpart;
372  break;
373  case 1:
374  header.pid1 = headerqpart;
375  break;
376  case 2:

382  break;

385  }
386  //now copy the mcmheader and mcmdata.

389  mRawDataPtr += sizeof(TrackletMCMHeader);

393  }
394  } else {
395  LOG(warn) << "something wrong with these tracklets, there are too many. You might want to take a closer look. Rejecting for now, and moving on.";
396  }
397  return trackletcounter;
398 }

471 void Trap2CRU::convertTrapData(o2::trd::TriggerRecord const& triggerrecord, const int& triggercount)
472 {
473  // LOG(info) << "starting :" << __func__ ; //<<
474  //build a HalfCRUHeader for this event/cru/endpoint
475  //loop over cru's
476  // loop over all half chambers, thankfully they data is sorted.
477  // check if current chamber has a link
478  // if not blank, else fill in data from link records
479  // dump data to rawwriter
480  //finished for event. this method is only called per event.
481  // char* traprawdataptr = (char*)&mTrapRawData[0];
482  std::array<int64_t, 21> localParsedDigitsindex; // store the index of the digits of an mcm
483  int rawwords = 0;
484  char* rawdataptratstart;
485  std::vector<char> rawdatavector(1024 * 1024 * 2); // sum of link sizes + padding in units of bytes and some space for the header (512 bytes).
486  LOG(debug) << "BUNCH CROSSING : " << triggerrecord.getBCData().bc << " with orbit : " << triggerrecord.getBCData().orbit;
487  for (int halfcru = 0; halfcru < o2::trd::constants::NHALFCRU; halfcru++) {
488  int halfcruwordswritten = 0;
489  int supermodule = halfcru / 4; // 2 cru per supermodule.
490  int endpoint = halfcru / 2; // 2 pci end points per cru, 15 links each
491  int side = halfcru % 2 ? 1 : 0;
492  mFeeID = buildTRDFeeID(supermodule, side, endpoint);
493  mCruID = halfcru / 2;
494  mLinkID = o2::trd::constants::TRDLINKID;
495  mEndPointID = halfcru % 2 ? 1 : 0;
496  //15 links per half cru or cru end point.
497  memset(&mRawData[0], 0, sizeof(mRawData[0]) * mRawData.size()); // zero the rawdata storage
498  int numberofdetectors = o2::trd::constants::MAXCHAMBER;
499  HalfCRUHeader halfcruheader;
500  //now write the cruheader at the head of all the data for this halfcru.
501  buildHalfCRUHeader(halfcruheader, triggerrecord.getBCData().bc, halfcru);
