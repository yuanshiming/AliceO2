31 struct Alice3TrackSelectionTask {
32  Produces<aod::TrackSelection> filterTable;
33  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::QAObject};
34 
35  Configurable<float> MaxDCAxy{"MaxDCAxy", 0.1f, "Maximum DCAxy"};
36  Configurable<float> MinEta{"MinEta", -0.8f, "Minimum eta in range"};
37  Configurable<float> MaxEta{"MaxEta", 0.8f, "Maximum eta in range"};
38 
39  void init(InitContext&)
40  {
41  histos.add("selection", "Selection process;Check;Entries", HistType::kTH1F, {{10, -0.5, 9.5}});
42  histos.get<TH1>(HIST("selection"))->GetXaxis()->SetBinLabel(1, "Tracks read");
43  histos.get<TH1>(HIST("selection"))->GetXaxis()->SetBinLabel(2, "DCAxy");
44  histos.get<TH1>(HIST("selection"))->GetXaxis()->SetBinLabel(3, "Eta");
45  histos.add("dcaXY/selected", "Selected;DCA_{xy} (cm);Entries", HistType::kTH1F, {{100, -5, 5}});
46  histos.add("dcaXY/nonselected", "Not selected;DCA_{xy} (cm);Entries", HistType::kTH1F, {{100, -5, 5}});
47  histos.add("eta/selected", "Selected;#eta;Entries", HistType::kTH1F, {{100, -2, 2}});
48  histos.add("eta/nonselected", "Not selected;#eta;Entries", HistType::kTH1F, {{100, -2, 2}});
49  }
50 
51  void process(soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
52  {
53  filterTable.reserve(tracks.size());
54  for (auto& track : tracks) {
55  histos.fill(HIST("selection"), 0);
56  histos.fill(HIST("dcaXY/nonselected"), track.dcaXY());
57  histos.fill(HIST("eta/nonselected"), track.eta());
58 
59  uint8_t sel = true;
60  if (abs(track.dcaXY()) > MaxDCAxy) {
61  histos.fill(HIST("selection"), 1);
62  sel = false;
63  }
64  if (track.eta() < MinEta || track.eta() > MaxEta) {
65  histos.fill(HIST("selection"), 2);
66  sel = false;
67  }
68  if (sel) {
69  histos.fill(HIST("dcaXY/selected"), track.dcaXY());
70  histos.fill(HIST("eta/selected"), track.eta());
71  }
72 
73  filterTable(sel, sel);
74  }
75  }
76 };
