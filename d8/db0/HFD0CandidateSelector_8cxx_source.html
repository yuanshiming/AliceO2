67  template <typename T>
68  bool selectionTopol(const T& candidate)
69  {
70  auto candpT = candidate.pt();
71  auto pTBin = findBin(pTBins, candpT);
72  if (pTBin == -1) {
73  return false;
74  }
75 
76  // check that the candidate pT is within the analysis range
77  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
78  return false;
79  }
80  // product of daughter impact parameters
81  if (candidate.impactParameterProduct() > cuts->get(pTBin, "d0d0")) {
82  return false;
83  }
84  // cosine of pointing angle
85  if (candidate.cpa() < cuts->get(pTBin, "cos pointing angle")) {
86  return false;
87  }
88  // cosine of pointing angle XY
89  if (candidate.cpaXY() < cuts->get(pTBin, "cos pointing angle xy")) {
90  return false;
91  }
92  // normalised decay length in XY plane
93  if (candidate.decayLengthXYNormalised() < cuts->get(pTBin, "normalized decay length XY")) {
94  return false;
95  }
96  // candidate DCA
97  //if (candidate.chi2PCA() > cuts[pTBin][1]) return false;
98 
99  // decay exponentail law, with tau = beta*gamma*ctau
100  // decay length > ctau retains (1-1/e)
101  if (std::abs(candidate.impactParameterNormalised0()) < 0.5 || std::abs(candidate.impactParameterNormalised1()) < 0.5) {
102  return false;
103  }
104  double decayLengthCut = std::min((candidate.p() * 0.0066) + 0.01, 0.06);
105  if (candidate.decayLength() * candidate.decayLength() < decayLengthCut * decayLengthCut) {
106  return false;
107  }
108  if (candidate.decayLengthNormalised() * candidate.decayLengthNormalised() < 1.0) {
109  //return false; // add back when getter fixed
110  }
111  return true;
112  }

120  template <typename T1, typename T2>
121  bool selectionTopolConjugate(const T1& candidate, const T2& trackPion, const T2& trackKaon)
122  {
123  auto candpT = candidate.pt();
124  auto pTBin = findBin(pTBins, candpT);
125  if (pTBin == -1) {
126  return false;
127  }
128 
129  // invariant-mass cut
130  if (trackPion.sign() > 0) {
131  if (std::abs(InvMassD0(candidate) - RecoDecay::getMassPDG(pdg::Code::kD0)) > cuts->get(pTBin, "m")) {
132  return false;
133  }
134  } else {
135  if (std::abs(InvMassD0bar(candidate) - RecoDecay::getMassPDG(pdg::Code::kD0)) > cuts->get(pTBin, "m")) {
136  return false;
137  }
138  }
139 
140  // cut on daughter pT
141  if (trackPion.pt() < cuts->get(pTBin, "pT Pi") || trackKaon.pt() < cuts->get(pTBin, "pT K")) {
142  return false;
143  }
144 
145  // cut on daughter DCA - need to add secondary vertex constraint here
146  if (std::abs(trackPion.dcaPrim0()) > cuts->get(pTBin, "d0pi") || std::abs(trackKaon.dcaPrim0()) > cuts->get(pTBin, "d0K")) {
147  return false;
148  }
149 
150  // cut on cos(theta*)
151  if (trackPion.sign() > 0) {
152  if (std::abs(CosThetaStarD0(candidate)) > cuts->get(pTBin, "cos theta*")) {
153  return false;
154  }
155  } else {
156  if (std::abs(CosThetaStarD0bar(candidate)) > cuts->get(pTBin, "cos theta*")) {
157  return false;
158  }
159  }
160 
161  return true;
162  }

164  void process(aod::HfCandProng2 const& candidates, aod::BigTracksPID const&)
165  {
166  TrackSelectorPID selectorPion(kPiPlus);
167  selectorPion.setRangePtTPC(d_pidTPCMinpT, d_pidTPCMaxpT);
168  selectorPion.setRangeNSigmaTPC(-d_nSigmaTPC, d_nSigmaTPC);
169  selectorPion.setRangeNSigmaTPCCondTOF(-d_nSigmaTPCCombined, d_nSigmaTPCCombined);
170  selectorPion.setRangePtTOF(d_pidTOFMinpT, d_pidTOFMaxpT);
171  selectorPion.setRangeNSigmaTOF(-d_nSigmaTOF, d_nSigmaTOF);
172  selectorPion.setRangeNSigmaTOFCondTPC(-d_nSigmaTOFCombined, d_nSigmaTOFCombined);
173 
174  TrackSelectorPID selectorKaon(selectorPion);
175  selectorKaon.setPDG(kKPlus);
176 
177  // looping over 2-prong candidates
178  for (auto& candidate : candidates) {
179 
180  // final selection flag: 0 - rejected, 1 - accepted
181  int statusD0 = 0;
182  int statusD0bar = 0;
183 
184  if (!(candidate.hfflag() & 1 << DecayType::D0ToPiK)) {
185  hfSelD0Candidate(statusD0, statusD0bar);
186  continue;
187  }
188 
189  auto trackPos = candidate.index0_as<aod::BigTracksPID>(); // positive daughter
190  auto trackNeg = candidate.index1_as<aod::BigTracksPID>(); // negative daughter
191 
192  /*
193  if (!daughterSelection(trackPos) || !daughterSelection(trackNeg)) {
194  hfSelD0Candidate(statusD0, statusD0bar);
195  continue;
196  }
197  */
198 
199  // conjugate-independent topological selection
200  if (!selectionTopol(candidate)) {
201  hfSelD0Candidate(statusD0, statusD0bar);
202  continue;
203  }
204 
205  // implement filter bit 4 cut - should be done before this task at the track selection level
206  // need to add special cuts (additional cuts on decay length and d0 norm)
207 
208  // conjugate-dependent topological selection for D0
209  bool topolD0 = selectionTopolConjugate(candidate, trackPos, trackNeg);
210  // conjugate-dependent topological selection for D0bar
211  bool topolD0bar = selectionTopolConjugate(candidate, trackNeg, trackPos);
212 
213  if (!topolD0 && !topolD0bar) {
214  hfSelD0Candidate(statusD0, statusD0bar);
215  continue;
216  }
217 
218  // track-level PID selection
219  int pidTrackPosKaon = selectorKaon.getStatusTrackPIDAll(trackPos);
220  int pidTrackPosPion = selectorPion.getStatusTrackPIDAll(trackPos);
221  int pidTrackNegKaon = selectorKaon.getStatusTrackPIDAll(trackNeg);
222  int pidTrackNegPion = selectorPion.getStatusTrackPIDAll(trackNeg);
223 
224  int pidD0 = -1;
225  int pidD0bar = -1;
226 
227  if (pidTrackPosPion == TrackSelectorPID::Status::PIDAccepted &&
228  pidTrackNegKaon == TrackSelectorPID::Status::PIDAccepted) {
229  pidD0 = 1; // accept D0
230  } else if (pidTrackPosPion == TrackSelectorPID::Status::PIDRejected ||
231  pidTrackNegKaon == TrackSelectorPID::Status::PIDRejected ||
232  pidTrackNegPion == TrackSelectorPID::Status::PIDAccepted ||
233  pidTrackPosKaon == TrackSelectorPID::Status::PIDAccepted) {
234  pidD0 = 0; // exclude D0
235  }
236 
237  if (pidTrackNegPion == TrackSelectorPID::Status::PIDAccepted &&
238  pidTrackPosKaon == TrackSelectorPID::Status::PIDAccepted) {
239  pidD0bar = 1; // accept D0bar
240  } else if (pidTrackPosPion == TrackSelectorPID::Status::PIDAccepted ||
241  pidTrackNegKaon == TrackSelectorPID::Status::PIDAccepted ||
242  pidTrackNegPion == TrackSelectorPID::Status::PIDRejected ||
243  pidTrackPosKaon == TrackSelectorPID::Status::PIDRejected) {
244  pidD0bar = 0; // exclude D0bar
245  }
246 
247  if (pidD0 == 0 && pidD0bar == 0) {
248  hfSelD0Candidate(statusD0, statusD0bar);
249  continue;
250  }
251 
252  if ((pidD0 == -1 || pidD0 == 1) && topolD0) {
253  statusD0 = 1; // identified as D0
254  }
255  if ((pidD0bar == -1 || pidD0bar == 1) && topolD0bar) {
256  statusD0bar = 1; // identified as D0bar
257  }
258 
259  hfSelD0Candidate(statusD0, statusD0bar);
260  }
261  }
