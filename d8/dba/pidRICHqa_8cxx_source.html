64  template <typename T>
65  void makelogaxis(T h)
66  {
67  if (logAxis == 0) {
68  return;
69  }
70  const int nbins = h->GetNbinsX();
71  double binp[nbins + 1];
72  double max = h->GetXaxis()->GetBinUpEdge(nbins);
73  double min = h->GetXaxis()->GetBinLowEdge(1);
74  if (min <= 0) {
75  min = 0.00001;
76  }
77  double lmin = TMath::Log10(min);
78  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
79  for (int i = 0; i < nbins; i++) {
80  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
81  }
82  binp[nbins] = max + 1;
83  h->GetXaxis()->Set(nbins, binp);
84  }

123  using Trks = soa::Join<aod::Tracks, aod::RICHTracksIndex, aod::TracksExtra>;
124  void process(const Trks& tracks,
125  const aod::McTrackLabels& labels,
126  const aod::RICHs&,
127  const aod::McParticles& mcParticles,
128  const aod::Collisions& colls)
129  {
130  for (const auto& col : colls) {
131  histos.fill(HIST("event/vertexz"), col.posZ());
132  }
133  for (const auto& track : tracks) {
134  if (!track.has_rich()) {
135  continue;
136  }
137  if (track.length() < minLength) {
138  continue;
139  }
140  if (track.length() > maxLength) {
141  continue;
142  }
143  if (track.eta() > maxEta || track.eta() < minEta) {
144  continue;
145  }
146  const auto mcParticle = labels.iteratorAt(track.globalIndex()).mcParticle();
147  if (pdgCode != 0 && abs(mcParticle.pdgCode()) != pdgCode) {
148  continue;
149  }
150  if (useOnlyPhysicsPrimary == 1 && !MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Selecting primaries
151  histos.fill(HIST("p/Sec"), track.p());
152  continue;
153  }
154  histos.fill(HIST("p/Prim"), track.p());
155  histos.fill(HIST("p/Unselected"), track.p());
156  histos.fill(HIST("pt/Unselected"), track.pt());
157  histos.fill(HIST("qa/signal"), track.rich().richSignal());
158  histos.fill(HIST("qa/signalerror"), track.rich().richSignalError());
159  histos.fill(HIST("qa/signalvsP"), track.p(), track.rich().richSignal());
160  histos.fill(HIST("qa/deltaEl"), track.p(), track.rich().richDeltaEl());
161  histos.fill(HIST("qa/deltaMu"), track.p(), track.rich().richDeltaMu());
162  histos.fill(HIST("qa/deltaPi"), track.p(), track.rich().richDeltaPi());
163  histos.fill(HIST("qa/deltaKa"), track.p(), track.rich().richDeltaKa());
164  histos.fill(HIST("qa/deltaPr"), track.p(), track.rich().richDeltaPr());
165  histos.fill(HIST("qa/nsigmaEl"), track.p(), track.rich().richNsigmaEl());
166  histos.fill(HIST("qa/nsigmaMu"), track.p(), track.rich().richNsigmaMu());
167  histos.fill(HIST("qa/nsigmaPi"), track.p(), track.rich().richNsigmaPi());
168  histos.fill(HIST("qa/nsigmaKa"), track.p(), track.rich().richNsigmaKa());
169  histos.fill(HIST("qa/nsigmaPr"), track.p(), track.rich().richNsigmaPr());
170  }
