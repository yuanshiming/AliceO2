231 AnalysisCut* o2::aod::dqcuts::GetAnalysisCut(const char* cutName)
232 {
233  //
234  // define here cuts which are likely to be used often
235  //
236  AnalysisCut* cut = new AnalysisCut(cutName, cutName);
237  std::string nameStr = cutName;
238 
239  if (!nameStr.compare("eventStandard")) {
240  cut->AddCut(VarManager::kVtxZ, -10.0, 10.0);
241  cut->AddCut(VarManager::kIsINT7, 0.5, 1.5);
242  return cut;
243  }
244 
245  if (!nameStr.compare("eventStandardNoINT7")) {
246  cut->AddCut(VarManager::kVtxZ, -10.0, 10.0);
247  return cut;
248  }
249 
250  if (!nameStr.compare("int7vtxZ5")) {
251  cut->AddCut(VarManager::kVtxZ, -5.0, 5.0);
252  cut->AddCut(VarManager::kIsINT7, 0.5, 1.5);
253  return cut;
254  }
255 
256  if (!nameStr.compare("jpsiStandardKine")) {
257  cut->AddCut(VarManager::kPt, 1.0, 1000.0);
258  cut->AddCut(VarManager::kEta, -0.9, 0.9);
259  return cut;
260  }
261 
262  if (!nameStr.compare("lmeeStandardKine")) {
263  cut->AddCut(VarManager::kPt, 0.2, 10.0);
264  cut->AddCut(VarManager::kEta, -0.8, 0.8);
265  return cut;
266  }
267 
268  if (!nameStr.compare("lmeeLowBKine")) {
269  cut->AddCut(VarManager::kPt, 0.075, 10.0);
270  cut->AddCut(VarManager::kEta, -0.8, 0.8);
271  return cut;
272  }
273 
274  if (!nameStr.compare("TightGlobalTrack")) {
275  cut->AddCut(VarManager::kIsSPDfirst, 0.5, 1.5);
276  cut->AddCut(VarManager::kIsITSrefit, 0.5, 1.5);
277  cut->AddCut(VarManager::kIsTPCrefit, 0.5, 1.5);
278  cut->AddCut(VarManager::kTPCchi2, 0.0, 4.0);
279  cut->AddCut(VarManager::kITSchi2, 0.0, 5.0);
280  cut->AddCut(VarManager::kTPCnclsCR, 80.0, 161.);
281  cut->AddCut(VarManager::kITSncls, 3.5, 7.5);
282  return cut;
283  }
284 
285  if (!nameStr.compare("electronStandardQuality")) {
286  cut->AddCut(VarManager::kIsSPDany, 0.5, 1.5);
287  cut->AddCut(VarManager::kIsITSrefit, 0.5, 1.5);
288  cut->AddCut(VarManager::kIsTPCrefit, 0.5, 1.5);
289  cut->AddCut(VarManager::kTPCchi2, 0.0, 4.0);
290  cut->AddCut(VarManager::kITSchi2, 0.1, 36.0);
291  cut->AddCut(VarManager::kTPCncls, 100.0, 161.);
292  return cut;
293  }
294 
295  if (!nameStr.compare("standardPrimaryTrack")) {
296  cut->AddCut(VarManager::kTrackDCAxy, -1.0, 1.0);
297  cut->AddCut(VarManager::kTrackDCAz, -3.0, 3.0);
298  return cut;
299  }
300 
301  TF1* cutLow1 = new TF1("cutLow1", "pol1", 0., 10.);
302  if (!nameStr.compare("electronPID1")) {
303  cutLow1->SetParameters(130., -40.0);
304  cut->AddCut(VarManager::kTPCsignal, 70., 100.);
305  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
306  return cut;
307  }
308 
309  if (!nameStr.compare("electronPID1shiftUp")) {
310  cut->AddCut(VarManager::kTPCsignal, 70. - 0.85, 100. - 0.85);
311  cutLow1->SetParameters(130. - 0.85, -40.0);
312  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0 - 0.85, false, VarManager::kPin, 0.5, 3.0);
313  return cut;
314  }
315 
316  if (!nameStr.compare("electronPID1shiftDown")) {
317  cut->AddCut(VarManager::kTPCsignal, 70.0 + 0.85, 100.0 + 0.85);
318  cutLow1->SetParameters(130. + 0.85, -40.0);
319  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0 + 0.85, false, VarManager::kPin, 0.5, 3.0);
320  return cut;
321  }
322 
323  if (!nameStr.compare("electronPID1randomized")) {
324  cutLow1->SetParameters(130., -40.0);
325  cut->AddCut(VarManager::kTPCsignalRandomized, 70., 100.);
326  cut->AddCut(VarManager::kTPCsignalRandomized, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
327  return cut;
328  }
329 
330  if (!nameStr.compare("electronPID2")) {
331  cutLow1->SetParameters(130., -40.0);
332  cut->AddCut(VarManager::kTPCsignal, 73., 100.);
333  cut->AddCut(VarManager::kTPCsignal, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
334  return cut;
335  }
336 
337  if (!nameStr.compare("electronPID2randomized")) {
338  cutLow1->SetParameters(130., -40.0);
339  cut->AddCut(VarManager::kTPCsignalRandomized, 73., 100.);
340  cut->AddCut(VarManager::kTPCsignalRandomized, cutLow1, 100.0, false, VarManager::kPin, 0.5, 3.0);
341  return cut;
342  }
343 
344  if (!nameStr.compare("electronPIDnsigma")) {
345  cut->AddCut(VarManager::kTPCnSigmaEl, -3.0, 3.0);
346  cut->AddCut(VarManager::kTPCnSigmaPr, 3.0, 3000.0);
347  cut->AddCut(VarManager::kTPCnSigmaPi, 3.0, 3000.0);
348  return cut;
349  }
350 
351  if (!nameStr.compare("electronPIDnsigmaRandomized")) {
352  cut->AddCut(VarManager::kTPCnSigmaElRandomized, -3.0, 3.0);
353  cut->AddCut(VarManager::kTPCnSigmaPrRandomized, 3.0, 3000.0);
354  cut->AddCut(VarManager::kTPCnSigmaPiRandomized, 3.0, 3000.0);
355  return cut;
356  }
357 
358  if (!nameStr.compare("electronPIDworseRes")) {
359  cut->AddCut(VarManager::kTPCnSigmaEl, -3.0, 3.0);
360  cut->AddCut(VarManager::kTPCnSigmaPr, 3.0 * 0.8, 3000.0); // emulates a 20% degradation in PID resolution
361  cut->AddCut(VarManager::kTPCnSigmaPi, 3.0 * 0.8, 3000.0); // proton and pion rejections are effectively relaxed by 20%
362  return cut;
363  }
364 
365  if (!nameStr.compare("electronPIDshift")) {
366  cut->AddCut(VarManager::kTPCnSigmaEl, -3.0, 3.0);
367  cut->AddCut(VarManager::kTPCnSigmaPr, 3.0 - 0.2, 3000.0);
368  cut->AddCut(VarManager::kTPCnSigmaPi, 3.0 - 0.2, 3000.0);
369  return cut;
370  }
371 
372  if (!nameStr.compare("tpc_pion_rejection")) {
373  TF1* f1maxPi = new TF1("f1maxPi", "[0]+[1]*x", 0, 10);
374  f1maxPi->SetParameters(85, -50);
375  cut->AddCut(VarManager::kTPCsignal, 70, f1maxPi, true, VarManager::kPin, 0.0, 0.4, false);
376  return cut;
377  }
378 
379  if (!nameStr.compare("tpc_kaon_rejection")) {
380  TF1* f1minKa = new TF1("f1minKa", "[0]+[1]*x", 0, 10);
381  f1minKa->SetParameters(220, -300);
382  TF1* f1maxKa = new TF1("f1maxKa", "[0]+[1]*x", 0, 10);
383  f1maxKa->SetParameters(182.5, -150);
384  cut->AddCut(VarManager::kTPCsignal, f1minKa, f1maxKa, true, VarManager::kPin, 0.4, 0.8, false);
385  return cut;
386  }
387 
388  if (!nameStr.compare("tpc_proton_rejection")) {
389  TF1* f1minPr = new TF1("f1minPr", "[0]+[1]*x", 0, 10);
390  f1minPr->SetParameters(170, -100);
391  TF1* f1maxPr = new TF1("f1maxPr", "[0]+[1]*x", 0, 10);
392  f1maxPr->SetParameters(175, -75);
393  cut->AddCut(VarManager::kTPCsignal, f1minPr, f1maxPr, true, VarManager::kPin, 0.8, 1.4, false);
394  return cut;
395  }
396 
397  if (!nameStr.compare("tpc_electron")) {
398  cut->AddCut(VarManager::kTPCsignal, 70, 90, false, VarManager::kPin, 0.0, 1e+10, false);
399  return cut;
400  }
401 
402  if (!nameStr.compare("tof_electron")) {
403  cut->AddCut(VarManager::kTOFbeta, 0.99, 1.01, false, VarManager::kPin, 0.0, 1e+10, false);
404  return cut;
405  }
406 
407  if (!nameStr.compare("muonQualityCuts")) {
408  cut->AddCut(VarManager::kEta, -4.0, -2.5);
409  cut->AddCut(VarManager::kMuonRAtAbsorberEnd, 25.0, 90.0);
410  cut->AddCut(VarManager::kMuonPDca, 0.0, 300.0);
411  return cut;
412  }
413 
414  if (!nameStr.compare("pairNoCut")) {
415  cut->AddCut(VarManager::kMass, 0.0, 1000.0);
416  return cut;
417  }
418 
419  if (!nameStr.compare("pairMassLow")) {
420  cut->AddCut(VarManager::kMass, 2.5, 1000.0);
421  return cut;
422  }
423 
424  if (!nameStr.compare("pairJpsi")) {
425  cut->AddCut(VarManager::kMass, 2.8, 3.3);
426  return cut;
427  }
428 
429  if (!nameStr.compare("pairPsi2S")) {
430  cut->AddCut(VarManager::kMass, 3.4, 3.9);
431  return cut;
432  }
433 
434  if (!nameStr.compare("pairUpsilon")) {
435  cut->AddCut(VarManager::kMass, 8.0, 11.0);
436  return cut;
437  }
438 
439  if (!nameStr.compare("pairPtLow1")) {
440  cut->AddCut(VarManager::kPt, 2.0, 1000.0);
441  return cut;
442  }
443 
444  if (!nameStr.compare("pairPtLow2")) {
445  cut->AddCut(VarManager::kPt, 5.0, 1000.0);
446  return cut;
447  }
448 
449  delete cut;
450  return nullptr;
