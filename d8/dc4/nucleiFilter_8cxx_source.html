71  void init(o2::framework::InitContext&)
72  {
73  std::vector<double> ptBinning = {0.5, 0.6, 0.7, 0.8, 0.9, 1.0, 1.1, 1.2, 1.3, 1.4, 1.5, 1.6, 1.8, 2.0, 2.2, 2.4, 2.8, 3.2, 3.6, 4., 5.};

76  AxisSpec ptAxis = {ptBinning, "#it{p}_{T} (GeV/#it{c})"};
77  AxisSpec centAxis = {centBinning, "V0M (%)"};
78 
79  spectra.add("fCollZpos", "collision z position", HistType::kTH1F, {{600, -20., +20., "z position (cm)"}});
80  spectra.add("fTPCsignal", "Specific energy loss", HistType::kTH2F, {{600, 0., 3, "#it{p} (GeV/#it{c})"}, {1400, 0, 1400, "d#it{E} / d#it{X} (a. u.)"}});
81  spectra.add("fTPCcounts", "n-sigma TPC", HistType::kTH2F, {ptAxis, {200, -100., +100., "n#sigma_{He} (a. u.)"}});
82  spectra.add("fProcessedEvents", "Nuclei - event filtered", HistType::kTH1F, {{4, -0.5, 3.5, "Event counter"}});
83  }

85  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
86  Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == static_cast<uint8_t>(1u));

88  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra, aod::pidTPCDe, aod::pidTPCTr, aod::pidTPCHe, aod::pidTPCAl, aod::pidTOFDe, aod::pidTOFTr, aod::pidTOFHe, aod::pidTOFAl, aod::TrackSelection>>;
89  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>::iterator const& collision, aod::BCsWithTimestamps const&, TrackCandidates const& tracks)
90  {
91  // collision process loop
92  bool keepEvent[nNuclei]{false};
93  //
94  spectra.fill(HIST("fCollZpos"), collision.posZ());
95  //
96 
97  for (auto track : tracks) { // start loop over tracks
98 
99  const float nSigmaTPC[nNuclei]{
100  track.tpcNSigmaDe(), track.tpcNSigmaTr(), track.tpcNSigmaHe(), track.tpcNSigmaAl()};
101  const float nSigmaTOF[nNuclei]{
102  track.tofNSigmaDe(), track.tofNSigmaTr(), track.tofNSigmaHe(), track.tofNSigmaAl()};
103 
104  for (int iN{0}; iN < nNuclei; ++iN) {
105  float y{rapidity(track.pt() * charges[iN], track.eta(), masses[iN])};
106  if (y < yMin + yBeam || y > yMax + yBeam) {
107  continue;
108  }
109  if (nSigmaTPC[iN] < cfgCutsPID->get(iN, 0u) || nSigmaTPC[iN] > cfgCutsPID->get(iN, 1u)) {
110  continue;
111  }
112  if (track.pt() > cfgCutsPID->get(iN, 4u) && (nSigmaTOF[iN] < cfgCutsPID->get(iN, 2u) || nSigmaTOF[iN] > cfgCutsPID->get(iN, 3u))) {
113  continue;
114  }
115  keepEvent[iN] = true;
116  }
117 
118  //
119  // fill QA histograms
120  //
121  spectra.fill(HIST("fTPCsignal"), track.tpcInnerParam(), track.tpcSignal());
122  spectra.fill(HIST("fTPCcounts"), track.tpcInnerParam(), track.tpcNSigmaHe());
123 
124  } // end loop over tracks
125  //
126  for (int iDecision{0}; iDecision < 4; ++iDecision) {
127  if (keepEvent[iDecision]) {
128  spectra.fill(HIST("fProcessedEvents"), iDecision);
129  }
130  }
131  tags(keepEvent[0], keepEvent[1], keepEvent[2], keepEvent[3]);
132  }
133 };
