58 void AlignableDetectorITS::defineVolumes()
59 {
60  // define ITS volumes
61  //
62  const int kNSPDSect = 10;
63  AlignableVolume *volITS = 0, *hstave = 0, *ladd = 0;
64  AlignableSensorITS* sens = 0;
65  //
66  int labDet = getDetLabel();
67  addVolume(volITS = new AlignableVolume("ITS", labDet));
68  //
69  // SPD
70  AlignableVolume* sect[kNSPDSect] = {0};
71  for (int isc = 0; isc < kNSPDSect; isc++) { // sectors
72  int iid = labDet + (10 + isc) * 10000;
73  addVolume(sect[isc] = new AlignableVolume(Form("ITS/SPD0/Sector%d", isc), iid));
74  sect[isc]->setParent(volITS);
75  }
76  for (int ilr = 0; ilr <= 1; ilr++) { // SPD layers
77  //
78  int cntVolID = 0, staveCnt = 0;
79  int nst = AliITSgeomTGeo::GetNLadders(ilr + 1) / kNSPDSect; // 2 or 4 staves per sector
80  for (int isc = 0; isc < kNSPDSect; isc++) { // sectors
81  for (int ist = 0; ist < nst; ist++) { // staves of SPDi
82  for (int ihst = 0; ihst < 2; ihst++) { // halfstave
83  int iid = labDet + (1 + ilr) * 10000 + (1 + staveCnt) * 100;
84  staveCnt++;
85  addVolume(hstave = new AlignableVolume(Form("ITS/SPD%d/Sector%d/Stave%d/HalfStave%d",
86  ilr, isc, ist, ihst),
87  iid));
88  hstave->setParent(sect[isc]);
89  hstave->setInternalID(iid);
90  for (int isn = 0; isn < 2; isn++) { // "ladder" (sensor)
91  int iids = iid + (1 + isn);
92  addVolume(sens =
93  new AlignableSensorITS(Form("ITS/SPD%d/Sector%d/Stave%d/HalfStave%d/Ladder%d",
94  ilr, isc, ist, ihst, isn + ihst * 2),
95  AliGeomManager::LayerToVolUID(ilr + 1, cntVolID++), iids));
96  sens->setParent(hstave);
97  }
98  }
99  } // staves of SPDi
100  } // sectors
101  } // SPD layers
102  //
103  // SDD
104  for (int ilr = 2; ilr <= 3; ilr++) { // layer
105  int cntVolID = 0, staveCnt = 0;
106  for (int ist = 0; ist < AliITSgeomTGeo::GetNLadders(ilr + 1); ist++) { // ladder
107  int iid = labDet + (1 + ilr) * 10000 + (1 + staveCnt) * 100;
108  staveCnt++;
109  addVolume(ladd = new AlignableVolume(Form("ITS/SDD%d/Ladder%d", ilr, ist), iid));
110  ladd->setParent(volITS);
111  for (int isn = 0; isn < AliITSgeomTGeo::GetNDetectors(ilr + 1); isn++) { // sensor
112  int iids = iid + (1 + isn);
113  addVolume(sens = new AlignableSensorITS(Form("ITS/SDD%d/Ladder%d/Sensor%d", ilr, ist, isn),
114  AliGeomManager::LayerToVolUID(ilr + 1, cntVolID++), iids));
115  sens->setParent(ladd);
116  }
117  } // ladder
118  } // layer
119  //
120  // SSD
121  for (int ilr = 4; ilr <= 5; ilr++) { // layer
122  int cntVolID = 0, staveCnt = 0;
123  for (int ist = 0; ist < AliITSgeomTGeo::GetNLadders(ilr + 1); ist++) { // ladder
124  int iid = labDet + (1 + ilr) * 10000 + (1 + staveCnt) * 100;
125  staveCnt++;
126  addVolume(ladd = new AlignableVolume(Form("ITS/SSD%d/Ladder%d", ilr, ist), iid));
127  ladd->setParent(volITS);
128  for (int isn = 0; isn < AliITSgeomTGeo::GetNDetectors(ilr + 1); isn++) { // sensor
129  int iids = iid + (1 + isn);
130  addVolume(sens = new AlignableSensorITS(Form("ITS/SSD%d/Ladder%d/Sensor%d", ilr, ist, isn),
131  AliGeomManager::LayerToVolUID(ilr + 1, cntVolID++), iids));
132  sens->setParent(ladd);
133  }
134  } // ladder
135  } // layer
136  //
137  //
138 }

163 void AlignableDetectorITS::SetAddErrorLr(int ilr, double sigY, double sigZ)
164 {

166  for (int isn = getNSensors(); isn--;) {

196 bool AlignableDetectorITS::CheckHitPattern(const AliESDtrack* trc, int sel)
197 {
198  // check if track hit pattern is ok
199  switch (sel) {
200  case kSPDBoth:
201  if (!trc->HasPointOnITSLayer(0) || !trc->HasPointOnITSLayer(1))
202  return false;
203  break;
204  case kSPDAny:
205  if (!trc->HasPointOnITSLayer(0) && !trc->HasPointOnITSLayer(1))
206  return false;
207  break;
208  case kSPD0:
209  if (!trc->HasPointOnITSLayer(0))
210  return false;
211  break;
212  case kSPD1:
213  if (!trc->HasPointOnITSLayer(1))
214  return false;
215  break;
216  default:
217  break;
218  }
219  return true;
220 }

223 void AlignableDetectorITS::UpdatePointByTrackInfo(AlignmentPoint* pnt, const AliExternalTrackParam* t) const
224 {
225  // update point using specific error parameterization
226  // the track must be in the detector tracking frame
227  const AlignableSensor* sens = pnt->getSensor();
228  int vid = sens->getVolID();
229  int lr = AliGeomManager::VolUIDToLayer(vid) - 1;
230  double angPol = ATan(t->GetTgl());
231  double angAz = ASin(t->GetSnp());
232  double errY, errZ;
233  GetErrorParamAngle(lr, angPol, angAz, errY, errZ);
234  const double* sysE = sens->getAddError(); // additional syst error
235  //
236  pnt->setYZErrTracking(errY * errY + sysE[0] * sysE[0], 0, errZ * errZ + sysE[1] * sysE[1]);
237  pnt->init();
238  //
239 }

241 void AlignableDetectorITS::GetErrorParamAngle(int layer, double anglePol, double angleAzi, double& erry, double& errz) const
242 {
243  // Modified version of AliITSClusterParam::GetErrorParamAngle
244  // Calculate cluster position error (parametrization extracted from rp-hit
245  // residuals, as a function of angle between track and det module plane.
246  // Origin: M.Lunardon, S.Moretto)
247  //
248  const int kNcfSPDResX = 21;
249  const float kCfSPDResX[kNcfSPDResX] = {+1.1201e+01, +2.0903e+00, -2.2909e-01, -2.6413e-01, +4.2135e-01, -3.7190e-01,
250  +4.2339e-01, +1.8679e-01, -5.1249e-01, +1.8421e-01, +4.8849e-02, -4.3127e-01,
251  -1.1148e-01, +3.1984e-03, -2.5743e-01, -6.6408e-02, +3.0756e-01, +2.6809e-01,
252  -5.0339e-03, -1.4964e-01, -1.1001e-01};
253  const float kSPDazMax = 56.000000;
254  //
255  /*
256  const int kNcfSPDMeanX = 16;
257  const float kCfSPDMeanX[kNcfSPDMeanX] = {-1.2532e+00,-3.8185e-01,-8.9039e-01,+2.6648e+00,+7.0361e-01,+1.2298e+00,
258  +3.2871e-01,+7.8487e-02,-1.6792e-01,-1.3966e-01,-3.1670e-01,-2.1795e-01,
259  -1.9451e-01,-4.9347e-02,-1.9186e-01,-1.9195e-01};
260  */
261  //
262  const int kNcfSPDResZ = 5;
263  const float kCfSPDResZ[kNcfSPDResZ] = {+9.2384e+01, +3.4352e-01, -2.7317e+01, -1.4642e-01, +2.0868e+00};
264  const float kSPDpolMin = 34.358002, kSPDpolMax = 145.000000;
265  //
266  const double kMaxSigmaSDDx = 100.;
267  const double kMaxSigmaSDDz = 400.;
268  const double kMaxSigmaSSDx = 100.;
269  const double kMaxSigmaSSDz = 1000.;
270  //
271  const double kParamSDDx[2] = {30.93, 0.059};
272  const double kParamSDDz[2] = {33.09, 0.011};
273  const double kParamSSDx[2] = {18.64, -0.0046};
274  const double kParamSSDz[2] = {784.4, -0.828};
275  double sigmax = 1000.0, sigmaz = 1000.0;
276  //double biasx = 0.0;
277 
278  angleAzi = Abs(angleAzi);
279  anglePol = Abs(anglePol);
280  //
281  if (angleAzi > 0.5 * Pi())
282  angleAzi = Pi() - angleAzi;
283  if (anglePol > 0.5 * Pi())
284  anglePol = Pi() - anglePol;
285  double angleAziDeg = angleAzi * RadToDeg();
286  double anglePolDeg = anglePol * RadToDeg();
287  //
288  if (layer == 0 || layer == 1) { // SPD
289  //
290  float phiInt = angleAziDeg / kSPDazMax; // mapped to -1:1
291  if (phiInt > 1)
292  phiInt = 1;
293  else if (phiInt < -1)
294  phiInt = -1;
295  float phiAbsInt = (TMath::Abs(angleAziDeg + angleAziDeg) - kSPDazMax) / kSPDazMax; // mapped to -1:1
296  if (phiAbsInt > 1)
297  phiAbsInt = 1;
298  else if (phiAbsInt < -1)
299  phiAbsInt = -1;
300  anglePolDeg += 90; // the parameterization was provided in polar angle (90 deg - normal to sensor)
301  float polInt = (anglePolDeg + anglePolDeg - (kSPDpolMax + kSPDpolMin)) / (kSPDpolMax - kSPDpolMin); // mapped to -1:1
302  if (polInt > 1)
303  polInt = 1;
304  else if (polInt < -1)
305  polInt = -1;
306  //
307  sigmax = AliCheb3DCalc::ChebEval1D(phiAbsInt, kCfSPDResX, kNcfSPDResX);
308  //biasx = AliCheb3DCalc::ChebEval1D(phiInt , kCfSPDMeanX, kNcfSPDMeanX);
309  sigmaz = AliCheb3DCalc::ChebEval1D(polInt, kCfSPDResZ, kNcfSPDResZ);
310  //
311  // for the moment for the SPD only, need to decide where to put it
312  //biasx *= 1e-4;
313 
314  } else if (layer == 2 || layer == 3) { // SDD
315 
316  sigmax = angleAziDeg * kParamSDDx[1] + kParamSDDx[0];
317  sigmaz = kParamSDDz[0] + kParamSDDz[1] * anglePolDeg;
318  if (sigmax > kMaxSigmaSDDx)
319  sigmax = kMaxSigmaSDDx;
320  if (sigmaz > kMaxSigmaSDDz)
321  sigmax = kMaxSigmaSDDz;
322 
323  } else if (layer == 4 || layer == 5) { // SSD
324 
325  sigmax = angleAziDeg * kParamSSDx[1] + kParamSSDx[0];
326  sigmaz = kParamSSDz[0] + kParamSSDz[1] * anglePolDeg;
327  if (sigmax > kMaxSigmaSSDx)
328  sigmax = kMaxSigmaSSDx;
329  if (sigmaz > kMaxSigmaSSDz)
330  sigmax = kMaxSigmaSSDz;
331  }
332  // convert from micron to cm
333  erry = 1.e-4 * sigmax;
334  errz = 1.e-4 * sigmaz;
335 }
