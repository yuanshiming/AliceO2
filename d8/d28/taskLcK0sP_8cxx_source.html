56  void process(soa::Filtered<soa::Join<aod::HfCandCascExt, aod::HFSelLcK0sPCandidate>> const& candidates)
57  {
58  //Printf("Candidates: %d", candidates.size());
59  for (auto& candidate : candidates) {
60  /*
61  // no such selection for LcK0sp for now - it is the only cascade
62  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
63  continue;
64  }
65  */
66  if (cutEtaCandMax >= 0. && std::abs(candidate.eta()) > cutEtaCandMax) {
67  //Printf("Candidate: eta rejection: %g", candidate.eta());
68  continue;
69  }
70 
71  registry.fill(HIST("hMass"), InvMassLcToK0sP(candidate));
72  registry.fill(HIST("hPtCand"), candidate.pt());
73  registry.fill(HIST("hPtBach"), candidate.ptProng0());
74  registry.fill(HIST("hPtV0"), candidate.ptProng1());
75  registry.fill(HIST("hd0Bach"), candidate.impactParameter0());
76  registry.fill(HIST("hd0V0pos"), candidate.dcapostopv());

105  void process(soa::Filtered<soa::Join<aod::HfCandCascExt, aod::HFSelLcK0sPCandidate, aod::HfCandCascadeMCRec>> const& candidates,
106  soa::Join<aod::McParticles, aod::HfCandCascadeMCGen> const& particlesMC, aod::BigTracksMC const& tracks)
107  {
108  // MC rec.
109  //Printf("MC Candidates: %d", candidates.size());
110  for (auto& candidate : candidates) {
111  if (cutEtaCandMax >= 0. && std::abs(candidate.eta()) > cutEtaCandMax) {
112  //Printf("MC Rec.: eta rejection: %g", candidate.eta());
113  continue;
114  }
115  if (std::abs(candidate.flagMCMatchRec()) == 1) {
116  // Get the corresponding MC particle.
117  auto indexMother = RecoDecay::getMother(particlesMC, candidate.index0_as<aod::BigTracksMC>().mcParticle_as<soa::Join<aod::McParticles, aod::HfCandCascadeMCGen>>(), pdg::Code::kLambdaCPlus, true);
118  auto particleMother = particlesMC.iteratorAt(indexMother);
119  registry.fill(HIST("hPtGenSig"), particleMother.pt()); // gen. level pT
120  registry.fill(HIST("hPtRecSig"), candidate.pt()); // rec. level pT
121  registry.fill(HIST("hCPARecSig"), candidate.cpa());
122  registry.fill(HIST("hEtaRecSig"), candidate.eta());
123  } else {
124  registry.fill(HIST("hPtRecBg"), candidate.pt());
125  registry.fill(HIST("hCPARecBg"), candidate.cpa());
126  registry.fill(HIST("hEtaRecBg"), candidate.eta());
127  }
128  }
129  // MC gen.
130  //Printf("MC Particles: %d", particlesMC.size());
131  for (auto& particle : particlesMC) {
132  if (cutEtaCandMax >= 0. && std::abs(particle.eta()) > cutEtaCandMax) {
133  //Printf("MC Gen.: eta rejection: %g", particle.eta());
134  continue;
135  }
136  if (std::abs(particle.flagMCMatchGen()) == 1) {
137  registry.fill(HIST("hPtGen"), particle.pt());
138  registry.fill(HIST("hEtaGen"), particle.eta());
139  }
140  }
141  }
142 };
