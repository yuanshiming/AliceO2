73  template <typename T>
74  void makelogaxis(T h)
75  {
76  const int nbins = h->GetNbinsX();
77  double binp[nbins + 1];
78  double max = h->GetXaxis()->GetBinUpEdge(nbins);
79  double min = h->GetXaxis()->GetBinLowEdge(1);
80  if (min <= 0) {
81  min = 0.00001;
82  }
83  double lmin = TMath::Log10(min);
84  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
85  for (int i = 0; i < nbins; i++) {
86  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
87  }
88  binp[nbins] = max + 1;
89  h->GetXaxis()->Set(nbins, binp);
90  }

130  template <uint8_t pidIndex, typename T, typename TT>
131  void fillNsigma(const T& track, const TT& mcParticles, const float& nsigma)
132  {
133  if (abs(track.mcParticle().pdgCode()) == PDGs[pidIndex]) {
134  histos.fill(HIST(hnsigmaMC[pidIndex]), track.pt(), nsigma);
135 
136  if (MC::isPhysicalPrimary(mcParticles, track.mcParticle())) { // Selecting primaries
137  histos.fill(HIST(hnsigmaMCprm[pidIndex]), track.pt(), nsigma);
138  } else {
139  histos.fill(HIST(hnsigmaMCsec[pidIndex]), track.pt(), nsigma);
140  }
141  }
142  }

144  void process(aod::Collision const& collision,
145  soa::Join<aod::Tracks, aod::TracksExtra,
146  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
147  aod::pidTOFFullKa, aod::pidTOFFullPr, aod::pidTOFFullDe,
148  aod::pidTOFFullTr, aod::pidTOFFullHe, aod::pidTOFFullAl,
149  aod::McTrackLabels, aod::pidTOFbeta> const& tracks,
150  aod::McParticles& mcParticles)
151  {
152  if (collision.numContrib() < nMinNumberOfContributors) {
153  return;
154  }
155  const float collisionTime_ps = collision.collisionTime() * 1000.f;
156  unsigned int nTracksWithTOF = 0;
157  for (auto t : tracks) {
158  //
159  if (t.tofSignal() < 0) { // Skipping tracks without TOF
160  continue;
161  }
162  if (t.eta() < MinEta || t.eta() > MaxEta) {
163  continue;
164  }
165  nTracksWithTOF++;
166  float nsigma = -999.f;
167  if constexpr (pid_type == 0) {
168  nsigma = t.tofNSigmaEl();
169  } else if constexpr (pid_type == 1) {
170  nsigma = t.tofNSigmaMu();
171  } else if constexpr (pid_type == 2) {
172  nsigma = t.tofNSigmaPi();
173  } else if constexpr (pid_type == 3) {
174  nsigma = t.tofNSigmaKa();
175  } else if constexpr (pid_type == 4) {
176  nsigma = t.tofNSigmaPr();
177  } else if constexpr (pid_type == 5) {
178  nsigma = t.tofNSigmaDe();
179  } else if constexpr (pid_type == 6) {
180  nsigma = t.tofNSigmaTr();
181  } else if constexpr (pid_type == 7) {
182  nsigma = t.tofNSigmaHe();
183  } else if constexpr (pid_type == 8) {
184  nsigma = t.tofNSigmaAl();
185  }
186 
187  // Fill for all
188  histos.fill(HIST(hnsigma[pid_type]), t.pt(), nsigma);
189  histos.fill(HIST("event/tofbeta"), t.p(), t.beta());
190  if (MC::isPhysicalPrimary(mcParticles, t.mcParticle())) { // Selecting primaries
191  histos.fill(HIST(hnsigmaprm[pid_type]), t.pt(), nsigma);
192  histos.fill(HIST("event/tofbetaPrm"), t.p(), t.beta());
193  } else {
194  histos.fill(HIST(hnsigmasec[pid_type]), t.pt(), nsigma);
195  histos.fill(HIST("event/tofbetaSec"), t.p(), t.beta());
196  }
197  // Fill with PDG codes
198  fillNsigma<0>(t, mcParticles, nsigma);
199  fillNsigma<1>(t, mcParticles, nsigma);
200  fillNsigma<2>(t, mcParticles, nsigma);
201  fillNsigma<3>(t, mcParticles, nsigma);
202  fillNsigma<4>(t, mcParticles, nsigma);
203  fillNsigma<5>(t, mcParticles, nsigma);
204  fillNsigma<6>(t, mcParticles, nsigma);
205  fillNsigma<7>(t, mcParticles, nsigma);
206  fillNsigma<8>(t, mcParticles, nsigma);
207  }
208  histos.fill(HIST("event/T0"), nTracksWithTOF, collisionTime_ps);
209  }
