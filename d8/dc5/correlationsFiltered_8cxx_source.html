144  void process(soa::Filtered<aod::CFCollisions>::iterator const& collision, soa::Filtered<aod::CFTracks> const& tracks)
145  {
146  if (cfgEfficiencyTrigger.value.empty() == false) {
147  cfg.mEfficiencyTrigger = ccdb->getForTimeStamp<THnT<float>>(cfgEfficiencyTrigger, collision.timestamp());
148  LOGF(info, "Loaded efficiency histogram for trigger particles from %s (%p)", cfgEfficiencyTrigger.value.c_str(), (void*)cfg.mEfficiencyTrigger);
149  }
150  if (cfgEfficiencyAssociated.value.empty() == false) {
151  cfg.mEfficiencyAssociated = ccdb->getForTimeStamp<THnT<float>>(cfgEfficiencyAssociated, collision.timestamp());
152  LOGF(info, "Loaded efficiency histogram for associated particles from %s (%p)", cfgEfficiencyAssociated.value.c_str(), (void*)cfg.mEfficiencyAssociated);
153  }
154 
155  LOGF(info, "Tracks for collision: %d | Vertex: %.1f | V0M: %.1f", tracks.size(), collision.posZ(), collision.centV0M());
156 
157  const auto centrality = collision.centV0M();
158 
159  same->fillEvent(centrality, CorrelationContainer::kCFStepReconstructed);
160 
161  int bSign = 1; // TODO magnetic field from CCDB
162 
163  // Cache efficiency for particles (too many FindBin lookups)
164  float* efficiencyAssociated = nullptr;
165  if (cfg.mEfficiencyAssociated) {
166  efficiencyAssociated = new float[tracks.size()];
167  int i = 0;
168  for (auto& track1 : tracks) {
169  efficiencyAssociated[i++] = getEfficiency(cfg.mEfficiencyAssociated, track1.eta(), track1.pt(), centrality, collision.posZ());
170  }
171  }
172 
173  for (auto& track1 : tracks) {
174  // LOGF(info, "Track %f | %f | %f %d %d", track1.eta(), track1.phi(), track1.pt(), track1.isGlobalTrack(), track1.isGlobalTrackSDD());
175 
176  registry.fill(HIST("yields"), centrality, track1.pt(), track1.eta());
177  registry.fill(HIST("etaphi"), centrality, track1.eta(), track1.phi());
178 
179  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.sign() < 0) {
180  continue;
181  }
182 
183  float triggerWeight = 1.0;
184  if (cfg.mEfficiencyTrigger) {
185  triggerWeight = getEfficiency(cfg.mEfficiencyTrigger, track1.eta(), track1.pt(), centrality, collision.posZ());
186  }
187 
188  same->getTriggerHist()->Fill(CorrelationContainer::kCFStepReconstructed, track1.pt(), centrality, collision.posZ(), triggerWeight);
189  //mixed->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
190 
191  int i = -1;
192  for (auto& track2 : tracks) {
193  i++; // HACK
194  if (track1 == track2) {
195  continue;
196  }
197 
198  if (cfgPtOrder != 0 && track2.pt() >= track1.pt()) {
199  continue;
200  }
201 
202  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.sign() < 0) {
203  continue;
204  }
205  if (cfgPairCharge != 0 && cfgPairCharge * track1.sign() * track2.sign() < 0) {
206  continue;
207  }
208 
209  if (cfg.mPairCuts && mPairCuts.conversionCuts(track1, track2)) {
210  continue;
211  }
212 
213  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign)) {
214  continue;
215  }
216 
217  float associatedWeight = 1.0;
218  if (cfg.mEfficiencyAssociated) {
219  associatedWeight = efficiencyAssociated[i];
220  }
221 
222  float deltaPhi = track1.phi() - track2.phi();
223  if (deltaPhi > 1.5 * M_PI) {
224  deltaPhi -= M_PI * 2;
225  }
226  if (deltaPhi < -0.5 * M_PI) {
227  deltaPhi += M_PI * 2;
228  }
229 
230  same->getPairHist()->Fill(CorrelationContainer::kCFStepReconstructed,
231  track1.eta() - track2.eta(), track2.pt(), track1.pt(), centrality, deltaPhi, collision.posZ(),
232  triggerWeight * associatedWeight);
233  //mixed->getPairHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
234  }
235  }
236 
237  delete[] efficiencyAssociated;
238  }

240  double getEfficiency(THn* eff, float eta, float pt, float centrality, float posZ)
241  {
242  int effVars[4];
243  effVars[0] = eff->GetAxis(0)->FindBin(eta);
244  effVars[1] = eff->GetAxis(1)->FindBin(pt);
245  effVars[2] = eff->GetAxis(2)->FindBin(centrality);
246  effVars[3] = eff->GetAxis(3)->FindBin(posZ);
247  return eff->GetBinContent(effVars);
248  }
