69  template <typename T1, typename T2>
70  bool selection(const T1& candidate, const T2& trackPion1, const T2& trackKaon, const T2& trackPion2)
71  {
72  auto candpT = candidate.pt();
73  int pTBin = findBin(pTBins, candpT);
74  if (pTBin == -1) {
75  return false;
76  }
77  // check that the candidate pT is within the analysis range
78  if (candpT < d_pTCandMin || candpT > d_pTCandMax) {
79  return false;
80  }
81  // cut on daughter pT
82  if (trackPion1.pt() < cuts->get(pTBin, "pT Pi") || trackKaon.pt() < cuts->get(pTBin, "pT K") || trackPion2.pt() < cuts->get(pTBin, "pT Pi")) {
83  return false;
84  }
85  // invariant-mass cut
86  if (std::abs(InvMassDPlus(candidate) - RecoDecay::getMassPDG(pdg::Code::kDPlus)) > cuts->get(pTBin, "deltaM")) {
87  return false;
88  }
89  if (candidate.decayLength() < cuts->get(pTBin, "decay length")) {
90  return false;
91  }
92  if (candidate.decayLengthXYNormalised() < cuts->get(pTBin, "normalized decay length XY")) {
93  return false;
94  }
95  if (candidate.cpa() < cuts->get(pTBin, "cos pointing angle")) {
96  return false;
97  }
98  if (candidate.cpaXY() < cuts->get(pTBin, "cos pointing angle XY")) {
99  return false;
100  }
101  if (std::abs(candidate.maxNormalisedDeltaIP()) > cuts->get(pTBin, "max normalized deltaIP")) {
102  return false;
103  }
104  return true;
105  }

107  void process(aod::HfCandProng3 const& candidates, aod::BigTracksPID const&)
108  {
109  TrackSelectorPID selectorPion(kPiPlus);
110  selectorPion.setRangePtTPC(d_pidTPCMinpT, d_pidTPCMaxpT);
111  selectorPion.setRangeNSigmaTPC(-d_nSigmaTPC, d_nSigmaTPC);
112  selectorPion.setRangePtTOF(d_pidTOFMinpT, d_pidTOFMaxpT);
113  selectorPion.setRangeNSigmaTOF(-d_nSigmaTOF, d_nSigmaTOF);
114 
115  TrackSelectorPID selectorKaon(selectorPion);
116  selectorKaon.setPDG(kKPlus);
117 
118  // looping over 3-prong candidates
119  for (auto& candidate : candidates) {
120 
121  // final selection flag: 0 - rejected, 1 - accepted
122  auto statusDplusToPiKPi = 0;
123 
124  if (!(candidate.hfflag() & 1 << DecayType::DPlusToPiKPi)) {
125  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
126  continue;
127  }
128 
129  auto trackPos1 = candidate.index0_as<aod::BigTracksPID>(); // positive daughter (negative for the antiparticles)
130  auto trackNeg = candidate.index1_as<aod::BigTracksPID>(); // negative daughter (positive for the antiparticles)
131  auto trackPos2 = candidate.index2_as<aod::BigTracksPID>(); // positive daughter (negative for the antiparticles)
132 
133  /*
134  // daughter track validity selection
135  if (!daughterSelection(trackPos1) ||
136  !daughterSelection(trackNeg) ||
137  !daughterSelection(trackPos2)) {
138  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
139  continue;
140  }
141  */
142 
143  // topological selection
144  if (!selection(candidate, trackPos1, trackNeg, trackPos2)) {
145  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
146  continue;
147  }
148 
149  // track-level PID selection
150  int pidTrackPos1Pion = selectorPion.getStatusTrackPIDAll(trackPos1);
151  int pidTrackNegKaon = selectorKaon.getStatusTrackPIDAll(trackNeg);
152  int pidTrackPos2Pion = selectorPion.getStatusTrackPIDAll(trackPos2);
153 
154  if (pidTrackPos1Pion == TrackSelectorPID::Status::PIDRejected ||
155  pidTrackNegKaon == TrackSelectorPID::Status::PIDRejected ||
156  pidTrackPos2Pion == TrackSelectorPID::Status::PIDRejected) { // exclude D±
157  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
158  continue;
159  }
160 
161  statusDplusToPiKPi = 1;
162  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
163  }
164  }
165 };
