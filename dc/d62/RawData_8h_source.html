122 struct TrackletHCHeader {
123  union {
124  // 10987654321098765432109876543210
125  // uint32_t: 33222222222211111111110000000000
126  // cccccccccccccccX LLL SSSSS
127  // ffff| |y| sss|
128  // | | ||| | |----- 0-4 supermodule
129  // | | ||| |-------- 5-7 stack
130  // | | ||------------ 8-10 layer
131  // | | |------------- 11 always 0x1
132  // | | |------------- 12 side of chamber
133  // | ----------------------------- 13-27 MCM Clock counter
134  // --------------------------------- 28-31 tracklet data format number
135  uint32_t word;
136  struct {
137  uint32_t supermodule : 5;
138  uint32_t stack : 3;
139  uint32_t layer : 3;
140  uint32_t one : 1; //always 1
141  uint32_t side : 1; // side of chamber
142  uint32_t MCLK : 15; // MCM clock counter 120MHz ... for simulation -- incrementing, and uniform across an event
143  uint32_t format : 4;
144  // 0 baseline PID 3 time slices, 7 bit each
145  // 1 DO NOT USE ! reserved for tracklet end marker disambiguation
146  // 14 Tracklet test-pattern mode
147  // 15 Reserved for testing
148  } __attribute__((__packed__));
149  };
