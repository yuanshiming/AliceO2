26 // Here the "sparse" matcher is used which means, there can be collisions without FV0A information
27 // To find out, collisionMatched.has_fv0a() has to be called. Otherwise collisionMatched.fv0a() will fail.
28 // NOTE: subscribing to Collisions separately will lead to a circular dependency due to forwarding
29 struct IterateV0 {
30  void process(aod::CollisionMatchedRun2Sparse const& collisionMatched, aod::FV0As const&)
31  {
32  LOGF(INFO, "Vertex = %f", collisionMatched.posZ());
33  if (collisionMatched.has_fv0a()) {
34  auto v0a = collisionMatched.fv0a();
35  LOGF(info, "V0A: %f %f", v0a.amplitude()[0], v0a.amplitude()[1]);
36  } else {
37  LOGF(INFO, "No V0A info");
38  }
39  }
40 };

42 // This example is identical to IterateV0, but uses the exclusive match. This means that collisions where any
43 // of the tables asked for in Run2MatchedExclusive (see AnalysisDataModel.h) are missing are not there.
44 // Therefore, the syntax is more complicated because we cannot join against Collision
45 // (the tables have different number of entries)
46 // Only to be used if one is sure that all your events have the desired information
47 struct IterateV0Exclusive {
48  void process(aod::Run2MatchedExclusive::iterator const& matcher, aod::Collisions const&, aod::FV0As const&)
49  {
50  LOGF(INFO, "Vertex = %f", matcher.collision().posZ());
51  auto fv0a = matcher.fv0a();
52  LOGF(info, "V0: %f %f", fv0a.amplitude()[0], fv0a.amplitude()[1]);
53  }
54 };
