59 //Armenteros-Podolanski variables
60 DECLARE_SOA_DYNAMIC_COLUMN(Alpha, alpha,
61  [](float pxpos, float pypos, float pzpos, float pxneg, float pyneg, float pzneg) {
62  float momTot = RecoDecay::P(pxpos + pxneg, pypos + pyneg, pzpos + pzneg);
63  float lQlNeg = RecoDecay::dotProd(array{pxneg, pyneg, pzneg}, array{pxpos + pxneg, pypos + pyneg, pzpos + pzneg}) / momTot;
64  float lQlPos = RecoDecay::dotProd(array{pxpos, pypos, pzpos}, array{pxpos + pxneg, pypos + pyneg, pzpos + pzneg}) / momTot;
65  return (lQlPos - lQlNeg) / (lQlPos + lQlNeg); //alphav0
66  });
67 
68 DECLARE_SOA_DYNAMIC_COLUMN(QtArm, qtarm,
69  [](float pxpos, float pypos, float pzpos, float pxneg, float pyneg, float pzneg) {
70  float momTot = RecoDecay::P2(pxpos + pxneg, pypos + pyneg, pzpos + pzneg);
71  float dp = RecoDecay::dotProd(array{pxneg, pyneg, pzneg}, array{pxpos + pxneg, pypos + pyneg, pzpos + pzneg});
72  return std::sqrt(RecoDecay::P2(pxneg, pyneg, pzneg) - dp * dp / momTot); //qtarm
73  });
