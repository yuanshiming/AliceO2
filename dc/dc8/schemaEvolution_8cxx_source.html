36 // Producer of EtaPhiV1
37 struct ProduceEtaPhiV1 {
38  Produces<aod::EtaPhiV1> etaphi;
39 
40  void process(aod::Tracks const& tracks)
41  {
42  for (auto& track : tracks) {
43  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
44  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
45 
46  etaphi(eta, phi);
47  }
48  }
49 };
