114  template <typename T>
115  bool selectionTopol(const T& hfCandCascade)
116  {
117  auto candPt = hfCandCascade.pt();
118  int ptBin = getPtBin(candPt);
119  if (ptBin == -1) {
120  return false;
121  }
122 
123  if (candPt < pTCandMin || candPt >= pTCandMax) {
124  LOG(DEBUG) << "cand pt (first check) cut failed: from cascade --> " << candPt << ", cut --> " << pTCandMax;
125  return false; //check that the candidate pT is within the analysis range
126  }
127 
128  if (std::abs(hfCandCascade.mK0Short() - RecoDecay::getMassPDG(kK0Short)) > cuts[ptBin][0]) {
129  LOG(DEBUG) << "massK0s cut failed: from v0 in cascade, K0s --> " << hfCandCascade.mK0Short() << ", in PDG K0s --> " << RecoDecay::getMassPDG(kK0Short) << ", cut --> " << cuts[ptBin][0];
130  return false; // mass of the K0s
131  }
132 
133  if ((std::abs(hfCandCascade.mLambda() - RecoDecay::getMassPDG(kLambda0)) < cuts[ptBin][1]) || (std::abs(hfCandCascade.mAntiLambda() - RecoDecay::getMassPDG(kLambda0)) < cuts[ptBin][1])) {
134  LOG(DEBUG) << "mass L cut failed: from v0 in cascade, Lambda --> " << hfCandCascade.mLambda() << ", AntiLambda --> " << hfCandCascade.mAntiLambda() << ", in PDG, Lambda --> " << RecoDecay::getMassPDG(kLambda0) << ", cut --> " << cuts[ptBin][1];
135  return false; // mass of the Lambda
136  }
137 
138  if (std::abs(InvMassGamma(hfCandCascade) - RecoDecay::getMassPDG(kGamma)) < cuts[ptBin][2]) {
139  LOG(DEBUG) << "mass gamma cut failed: from v0 in cascade, gamma --> " << InvMassGamma(hfCandCascade) << ", cut --> " << cuts[ptBin][2];
140  return false; // mass of the Gamma
141  }
142 
143  if (hfCandCascade.ptProng0() < cuts[ptBin][3]) {
144  LOG(DEBUG) << "bach pt cut failed, from cascade --> " << hfCandCascade.ptProng0() << " , cut --> " << cuts[ptBin][3];
145  return false; // pt of the p
146  }
147 
148  if (hfCandCascade.ptV0Pos() < cuts[ptBin][4]) {
149  LOG(DEBUG) << "v0 pos daugh pt cut failed, from cascade --> " << hfCandCascade.ptV0Pos() << ", cut --> " << cuts[ptBin][4];
150  return false; // pt of the K0
151  }
152 
153  if (hfCandCascade.ptV0Neg() < cuts[ptBin][4]) {
154  LOG(DEBUG) << "v0 neg daugh pt cut failed, from cascade --> " << hfCandCascade.ptV0Neg() << ", cut --> " << cuts[ptBin][4];
155  return false; // pt of the K0
156  }
157 
158  if (hfCandCascade.pt() < cuts[ptBin][5]) {
159  LOG(DEBUG) << "cand pt cut failed, from cascade --> " << hfCandCascade.pt() << ", cut --> " << cuts[ptBin][5];
160  return false; // pt of the Lc
161  }
162 
163  if (std::abs(hfCandCascade.impactParameter0()) > cuts[ptBin][6]) {
164  LOG(DEBUG) << "d0 bach cut failed, in cascade --> " << hfCandCascade.impactParameter0() << ", cut --> " << cuts[ptBin][6];
165  return false; // d0 of the bachelor
166  }
167 
168  /*
169  if ((std::abs(hfCandCascade.dcapostopv()) > cuts[ptBin][7]) || (std::abs(hfCandCascade.dcanegtopv()) > cuts[ptBin][7])) {
170  LOG(DEBUG) << "v0 daugh cut failed, positive v0 daugh --> " << hfCandCascade.dcapostopv() << ", negative v0 daugh --> " << hfCandCascade.dcanegtopv() << " , cut --> " << cuts[ptBin][7];
171  return false; // d0 of the K0s daughters
172  }
173  */
174 
175  if (std::abs(hfCandCascade.impactParameter1()) > cuts[ptBin][7]) {
176  LOG(DEBUG) << "d0 v0 cut failed, in cascade --> " << hfCandCascade.impactParameter1() << ", cut --> " << cuts[ptBin][7];
177  return false; // d0 of the v0
178  }
179 
180  return true;
181  }

326  void process(aod::HfCandCascade const& candidates, MyBigTracks const& tracks)
327  {
328  int statusLc = 0; // final selection flag : 0-rejected 1-accepted
329  bool topolLc = 0;
330  int pidProton = -1;
331  int pidLc = -1;
332 
333  for (auto& candidate : candidates) { //looping over cascade candidates
334 
335  const auto& bach = candidate.index0_as<MyBigTracks>(); //bachelor track
336 #ifdef MY_DEBUG
337  auto indexV0DaughPos = candidate.posTrack_as<MyBigTracks>().mcParticleId();
338  auto indexV0DaughNeg = candidate.negTrack_as<MyBigTracks>().mcParticleId();
339  auto indexBach = bach.mcParticleId();
340  bool isLc = isLcK0SpFunc(indexBach, indexV0DaughPos, indexV0DaughNeg, indexProton, indexK0Spos, indexK0Sneg);
341  bool isK0SfromLc = isK0SfromLcFunc(indexV0DaughPos, indexV0DaughNeg, indexK0Spos, indexK0Sneg);
342 #endif
343  MY_DEBUG_MSG(isLc, printf("\n"); LOG(INFO) << "In selector: correct Lc found: proton --> " << indexBach << ", posTrack --> " << indexV0DaughPos << ", negTrack --> " << indexV0DaughNeg);
344  //MY_DEBUG_MSG(isLc != 1, printf("\n"); LOG(INFO) << "In selector: wrong Lc found: proton --> " << indexBach << ", posTrack --> " << indexV0DaughPos << ", negTrack --> " << indexV0DaughNeg);
345 
346  statusLc = 0;
347  /* // not needed for the Lc
348  if (!(candidate.hfflag() & 1 << D0ToPiK)) {
349  hfSelD0Candidate(statusLc);
350  continue;
351  }
352  */
353 
354  topolLc = true;
355  pidProton = -1;
356 
357  // daughter track validity selection
358  LOG(DEBUG) << "daughterSelection(bach) = " << daughterSelection(bach);
359  if (!daughterSelection(bach)) {
360  MY_DEBUG_MSG(isLc, LOG(INFO) << "In selector: Lc rejected due to selections on bachelor");
361  hfSelLcK0sPCandidate(statusLc);
362  continue;
363  }
364 
365  //implement filter bit 4 cut - should be done before this task at the track selection level
366  //need to add special cuts (additional cuts on decay length and d0 norm)
367  LOG(DEBUG) << "selectionTopol(candidate) = " << selectionTopol(candidate);
368  if (!selectionTopol(candidate)) {
369  MY_DEBUG_MSG(isLc, LOG(INFO) << "In selector: Lc rejected due to topological selections");
370  hfSelLcK0sPCandidate(statusLc);
371  continue;
372  }
373 
374  pidProton = selectionPID(bach);
375 
376  LOG(DEBUG) << "pidProton = " << pidProton;
377 
378  if (pidProton == 1) {
379  statusLc = 1;
380  }
381 
382  MY_DEBUG_MSG(isLc && pidProton != 1, LOG(INFO) << "In selector: Lc rejected due to PID selections on bachelor");
383  MY_DEBUG_MSG(isLc && pidProton == 1, LOG(INFO) << "In selector: Lc ACCEPTED");
384 
385  hfSelLcK0sPCandidate(statusLc);
386  }
387  }
