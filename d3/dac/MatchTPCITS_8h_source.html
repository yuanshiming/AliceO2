145 struct MatchRecord {
146  float chi2 = -1.f;
147  int partnerID = MinusOne;
148  int nextRecID = MinusOne;
149  int matchedIC = MinusOne;
150  MatchRecord(int partID, float chi2match, int nxt = MinusOne, int candIC = MinusOne) : partnerID(partID), chi2(chi2match), nextRecID(nxt), matchedIC(candIC) {}
151  MatchRecord() = default;
152 
153  bool isBetter(float otherChi2, int otherIC = MinusOne) const
154  { // prefer record with matched IC candidate, otherwise, better chi2
155  if (otherIC == MinusOne) {
156  return matchedIC == MinusOne ? chi2 < otherChi2 : true;
157  } else {
158  return matchedIC == MinusOne ? false : chi2 < otherChi2;
159  }
160  }
161 
162  bool isBetter(const MatchRecord& other) const
163  {
164  return isBetter(other.chi2, other.matchedIC);
165  }
166 };
