36 int InputHelper::addInputSpecs(const ConfigContext& configcontext, WorkflowSpec& specs, GID::mask_t maskClusters, GID::mask_t maskMatches, GID::mask_t maskTracks, bool useMC, GID::mask_t maskClustersMC, GID::mask_t maskTracksMC)
37 {
38  if (configcontext.options().get<bool>("disable-root-input")) {
39  return 0;
40  }
41  if (useMC && configcontext.options().get<bool>("disable-mc")) {
42  useMC = false;
43  }
44  if (!useMC) {
45  maskClustersMC = GID::getSourcesMask(GID::NONE);
46  maskTracksMC = GID::getSourcesMask(GID::NONE);
47  }
48 
49  if (maskTracks[GID::ITS]) {
50  specs.emplace_back(o2::its::getITSTrackReaderSpec(maskTracksMC[GID::ITS]));
51  }
52  if (maskClusters[GID::ITS]) {
53  specs.emplace_back(o2::itsmft::getITSClusterReaderSpec(maskClustersMC[GID::ITS], true));
54  }
55  if (maskTracks[GID::MFT]) {
56  specs.emplace_back(o2::mft::getMFTTrackReaderSpec(maskTracksMC[GID::MFT]));
57  }
58  if (maskTracks[GID::TPC]) {
59  specs.emplace_back(o2::tpc::getTPCTrackReaderSpec(maskTracksMC[GID::TPC]));
60  }
61  if (maskClusters[GID::TPC]) {
62  specs.emplace_back(o2::tpc::getClusterReaderSpec(maskClustersMC[GID::TPC]));
63  }
64  if (maskTracks[GID::TPC] && maskClusters[GID::TPC]) {
65  specs.emplace_back(o2::tpc::getClusterSharingMapSpec());
66  }
67  if (maskMatches[GID::ITSTPC] || maskMatches[GID::ITSTPCTOF] || maskTracks[GID::ITSTPC] || maskTracks[GID::ITSTPCTOF]) {
68  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(maskTracksMC[GID::ITSTPC] || maskTracksMC[GID::ITSTPCTOF]));
69  }
70  if (maskMatches[GID::ITSTPCTOF] || maskTracks[GID::ITSTPCTOF]) {
71  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::ITSTPCTOF], false, /*maskTracks[GID::ITSTPCTOF]*/ false)); // ITSTPCTOF does not provide tracks, only matchInfo
72  }
73  if (maskClusters[GID::TOF] || maskTracks[GID::ITSTPCTOF]) { // Note: maskTracks[GID::ITSTPCTOF] is only here to match the behavior of RecoContainer::requestTracks
74  specs.emplace_back(o2::tof::getClusterReaderSpec(maskClustersMC[GID::TOF]));
75  }
76  if (maskMatches[GID::TPCTOF]) {
77  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::TPCTOF], true, maskTracks[GID::TPCTOF]));
78  }
79  if (maskTracks[GID::FT0] || maskClusters[GID::FT0]) {
80  specs.emplace_back(o2::ft0::getRecPointReaderSpec(maskTracksMC[GID::FT0] || maskClustersMC[GID::FT0]));
81  }
82  if (maskClusters[GID::TRD]) {
83  specs.emplace_back(o2::trd::getTRDTrackletReaderSpec(maskClustersMC[GID::TRD], true));
84  }
85  if (maskTracks[GID::ITSTPCTRD]) {
86  specs.emplace_back(o2::trd::getTRDGlobalTrackReaderSpec(false));
87  }
88  if (maskTracks[GID::TPCTRD]) {
89  specs.emplace_back(o2::trd::getTRDTPCTrackReaderSpec(false));
90  }
91 
92  return 0;
93 }
