17 labelmap_t LabelMap::populate(uint32_t size, std::vector<std::string> labels)
18 {
19  labelmap_t map;
20  if (labels.empty() == false) {
21  if (labels.size() != size) {
22  throw runtime_error("labels array size != array dimension");
23  }
24  for (auto i = 0u; i < size; ++i) {
25  map.emplace(labels[i], (uint32_t)i);
26  }
27  }
28  return map;
29 }
