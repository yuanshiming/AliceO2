28 class TOFReso : public Parametrization
29 {
30  public:
31  TOFReso() : Parametrization("TOFReso", 5){};
32  ~TOFReso() override = default;
33  float operator()(const float* x) const override
34  {
35  const float mom = abs(x[0]);
36  if (mom <= 0) {
37  return -999;
38  }
39  const float time = x[1];
40  const float evtimereso = x[2];
41  const float mass = x[3];
42  const float dpp = mParameters[0] + mParameters[1] * mom + mParameters[2] * mass / mom; // mean relative pt resolution;
43  const float sigma = dpp * time / (1. + mom * mom / (mass * mass));
44  return sqrt(sigma * sigma + mParameters[3] * mParameters[3] / mom / mom + mParameters[4] * mParameters[4] + evtimereso * evtimereso);
45  }
46  ClassDef(TOFReso, 1);
47 };

49 float TOFResoParam(const float& momentum, const float& time, const float& evtimereso, const float& mass, const Parameters& parameters)
50 {
51  if (momentum <= 0) {
52  return -999;
53  }
54  const float dpp = parameters[0] + parameters[1] * momentum + parameters[2] * mass / momentum; // mean relative pt resolution;
55  const float sigma = dpp * time / (1. + momentum * momentum / (mass * mass));
56  return sqrt(sigma * sigma + parameters[3] * parameters[3] / momentum / momentum + parameters[4] * parameters[4] + evtimereso * evtimereso);
57 }
