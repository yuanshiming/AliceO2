31 struct tofSpectra {
32  static constexpr int Np = 9;
33  static constexpr const char* pT[Np] = {"e", "#mu", "#pi", "K", "p", "d", "t", "^{3}He", "#alpha"};
34  static constexpr std::string_view hp[Np] = {"p/El", "p/Mu", "p/Pi", "p/Ka", "p/Pr", "p/De", "p/Tr", "p/He", "p/Al"};
35  static constexpr std::string_view hpt[Np] = {"pt/El", "pt/Mu", "pt/Pi", "pt/Ka", "pt/Pr", "pt/De", "pt/Tr", "pt/He", "pt/Al"};
36  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};
37 
38  void init(o2::framework::InitContext&)
39  {
40  histos.add("p/Unselected", "Unselected;#it{p} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
41  histos.add("pt/Unselected", "Unselected;#it{p}_{T} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
42  for (int i = 0; i < Np; i++) {
43  histos.add(hp[i].data(), Form("%s;#it{p} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
44  histos.add(hpt[i].data(), Form("%s;#it{p}_{T} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
45  }
46  histos.add("electronbeta/hp_El", ";#it{p} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
47  histos.add("electronbeta/hpt_El", ";#it{p}_{T} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
48  histos.add("electronbeta/hlength_El", ";Track Length (cm);Tracks", kTH1D, {{100, 0, 1000}});
49  histos.add("electronbeta/htime_El", ";TOF Time (ns);Tracks", kTH1D, {{1000, 0, 600}});
50  histos.add("electronbeta/hp_beta_El", ";#it{p} (GeV/#it{c});#beta - #beta_{e};Tracks", kTH2D, {{100, 0, 20}, {100, -0.01, 0.01}});
51  histos.add("electronbeta/hp_betasigma_El", ";#it{p} (GeV/#it{c});(#beta - #beta_{e})/#sigma;Tracks", kTH2D, {{100, 0, 20}, {100, -5, 5}});
52  }

54  template <std::size_t i, typename T>
55  void fillParticleHistos(const T& track, const float& nsigma)
56  {
57  if (abs(nsigma) > cfgNSigmaCut) {
58  return;
59  }
60  histos.fill(HIST(hp[i]), track.p());
61  histos.fill(HIST(hpt[i]), track.pt());
62  }
63 
64  //Defining filters and input
65  Configurable<float> cfgNSigmaCut{"cfgNSigmaCut", 3, "Value of the Nsigma cut"};
66  Configurable<float> cfgCutVertex{"cfgCutVertex", 10.0f, "Accepted z-vertex range"};
67  Configurable<float> cfgCutEta{"cfgCutEta", 0.8f, "Eta range for tracks"};
68  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
69  Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == (uint8_t) true);
70  Filter trackFilterTOF = (aod::track::tofSignal > 0.f); // Skip tracks without TOF
71  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra,
72  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
73  aod::pidTOFFullKa, aod::pidTOFFullPr, aod::pidTOFFullDe,
74  aod::pidTOFFullTr, aod::pidTOFFullHe, aod::pidTOFFullAl,
75  aod::pidTOFbeta, aod::TrackSelection>>;

77  void process(TrackCandidates::iterator const& track)
78  {
79  histos.fill(HIST("p/Unselected"), track.p());
80  histos.fill(HIST("pt/Unselected"), track.pt());
81 
82  fillParticleHistos<0>(track, track.tofNSigmaEl());
83  fillParticleHistos<1>(track, track.tofNSigmaMu());
84  fillParticleHistos<2>(track, track.tofNSigmaPi());
85  fillParticleHistos<3>(track, track.tofNSigmaKa());
86  fillParticleHistos<4>(track, track.tofNSigmaPr());
87  fillParticleHistos<5>(track, track.tofNSigmaDe());
88  fillParticleHistos<6>(track, track.tofNSigmaTr());
89  fillParticleHistos<7>(track, track.tofNSigmaHe());
90  fillParticleHistos<8>(track, track.tofNSigmaAl());
91 
92  //
93  if (TMath::Abs(track.separationbetael() < 1.f)) {
94  histos.fill(HIST("electronbeta/hp_El"), track.p());
95  histos.fill(HIST("electronbeta/hpt_El"), track.pt());
96  histos.fill(HIST("electronbeta/hlength_El"), track.length());
97  histos.fill(HIST("electronbeta/htime_El"), track.tofSignal() / 1000);
98  histos.fill(HIST("electronbeta/hp_beta_El"), track.p(), track.diffbetael());
99  histos.fill(HIST("electronbeta/hp_betasigma_El"), track.p(), track.separationbetael());
100  }
101  } // end of the process function
102 };
