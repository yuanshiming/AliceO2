49 struct HFCandidateCreatorCascade {
50 
51  Produces<aod::HfCandCascBase> rowCandidateBase;
52 
53  Configurable<double> bZ{"bZ", 5., "magnetic field"};
54  Configurable<bool> propDCA{"propDCA", true, "create tracks version propagated to PCA"};
55  Configurable<double> maxR{"maxR", 200., "reject PCA's above this radius"};
56  Configurable<double> maxDZIni{"maxDZIni", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
57  Configurable<double> minParamChange{"minParamChange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
58  Configurable<double> minRelChi2Change{"minRelChi2Change", 0.9, "stop iterations is chi2/chi2old > this"};
59  Configurable<bool> doValPlots{"doValPlots", true, "do validation plots"};
60 
61  // for debugging
62 #ifdef MY_DEBUG
63  Configurable<std::vector<int>> indexK0Spos{"indexK0Spos", {729, 2866, 4754, 5457, 6891, 7824, 9243, 9810}, "indices of K0S positive daughters, for debug"};
64  Configurable<std::vector<int>> indexK0Sneg{"indexK0Sneg", {730, 2867, 4755, 5458, 6892, 7825, 9244, 9811}, "indices of K0S negative daughters, for debug"};
65  Configurable<std::vector<int>> indexProton{"indexProton", {717, 2810, 4393, 5442, 6769, 7793, 9002, 9789}, "indices of protons, for debug"};
66 #endif
67 
68  OutputObj<TH1F> hmass2{TH1F("hmass2", "2-prong candidates;inv. mass (#pi K) (GeV/#it{c}^{2});entries", 500, 0., 5.)};
69  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "2-prong candidates;XX element of cov. matrix of prim. vtx. position (cm^{2});entries", 100, 0., 1.e-4)};
70  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "2-prong candidates;XX element of cov. matrix of sec. vtx. position (cm^{2});entries", 100, 0., 0.2)};
71 
72  double massP = RecoDecay::getMassPDG(kProton);
73  double massK0s = RecoDecay::getMassPDG(kK0Short);
74  double massPi = RecoDecay::getMassPDG(kPiPlus);
75  double massLc = RecoDecay::getMassPDG(pdg::Code::kLambdaCPlus);
76  double mass2K0sP{0.};

78  void process(aod::Collisions const& collisions,
79  aod::HfTrackIndexCasc const& rowsTrackIndexCasc,
80  MyBigTracks const& tracks,
81  aod::V0Datas const& V0s
82 #ifdef MY_DEBUG
83  ,
84  aod::McParticles& mcParticles
85 #endif
86  )
87  {
88  // 2-prong vertex fitter
89  o2::vertexing::DCAFitterN<2> df;
90  df.setBz(bZ);
91  df.setPropagateToPCA(propDCA);
92  df.setMaxR(maxR);
93  df.setMaxDZIni(maxDZIni);
94  df.setMinParamChange(minParamChange);
95  df.setMinRelChi2Change(minRelChi2Change);
96  df.setUseAbsDCA(true);
97 
98  // loop over pairs of track indeces
99  for (const auto& casc : rowsTrackIndexCasc) {
100 
101  const auto& bach = casc.index0_as<MyBigTracks>();
102  const auto& v0 = casc.indexV0_as<o2::aod::V0Datas>();
103  const auto& trackV0DaughPos = v0.posTrack_as<MyBigTracks>();
104  const auto& trackV0DaughNeg = v0.negTrack_as<MyBigTracks>();
105 
106 #ifdef MY_DEBUG
107  auto indexBach = bach.mcParticleId();
108  auto indexV0DaughPos = trackV0DaughPos.mcParticleId();
109  auto indexV0DaughNeg = trackV0DaughNeg.mcParticleId();
110  bool isLc = isLcK0SpFunc(indexBach, indexV0DaughPos, indexV0DaughNeg, indexProton, indexK0Spos, indexK0Sneg);
111 #endif
112 
113  MY_DEBUG_MSG(isLc, LOG(INFO) << "Processing the Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg);
114 
115  auto trackParCovBach = getTrackParCov(bach);
116  auto trackParCovV0DaughPos = getTrackParCov(trackV0DaughPos); // check that MyBigTracks does not need TracksExtended!
117  auto trackParCovV0DaughNeg = getTrackParCov(trackV0DaughNeg); // check that MyBigTracks does not need TracksExtended!
118  trackParCovV0DaughPos.propagateTo(v0.posX(), bZ); // propagate the track to the X closest to the V0 vertex
119  trackParCovV0DaughNeg.propagateTo(v0.negX(), bZ); // propagate the track to the X closest to the V0 vertex
120  const std::array<float, 3> vertexV0 = {v0.x(), v0.y(), v0.z()};
121  const std::array<float, 3> momentumV0 = {v0.px(), v0.py(), v0.pz()};
122  // we build the neutral track to then build the cascade
123  auto trackV0 = o2::dataformats::V0(vertexV0, momentumV0, {0, 0, 0, 0, 0, 0}, trackParCovV0DaughPos, trackParCovV0DaughNeg, {0, 0}, {0, 0}); // build the V0 track (indices for v0 daughters set to 0 for now)
124 
125  auto collision = bach.collision();
126 
127  // reconstruct the cascade secondary vertex
128  if (df.process(trackV0, trackParCovBach) == 0) {
129  MY_DEBUG_MSG(isLc, LOG(INFO) << "Vertexing failed for Lc candidate");
130  // if (isLc) {
131  // LOG(INFO) << "Vertexing failed for Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg;
132  //}
133  continue;
134  } else {
135  //LOG(INFO) << "Vertexing succeeded for Lc candidate";
136  }
137 
138  const auto& secondaryVertex = df.getPCACandidate();
139  auto chi2PCA = df.getChi2AtPCACandidate();
140  auto covMatrixPCA = df.calcPCACovMatrix().Array();
141  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
142  // do I have to call "df.propagateTracksToVertex();"?
143  auto trackParVarV0 = df.getTrack(0);
144  auto trackParVarBach = df.getTrack(1);
145 
146  // get track momenta
147  array<float, 3> pVecV0;
148  array<float, 3> pVecBach;
149  trackParVarV0.getPxPyPzGlo(pVecV0);
150  trackParVarBach.getPxPyPzGlo(pVecBach);
151 
152  // get track impact parameters
153  // This modifies track momenta!
154  auto primaryVertex = getPrimaryVertex(collision);
155  auto covMatrixPV = primaryVertex.getCov();
156  hCovPVXX->Fill(covMatrixPV[0]);
157  o2::dataformats::DCA impactParameterV0;
158  o2::dataformats::DCA impactParameterBach;
159  trackParVarV0.propagateToDCA(primaryVertex, bZ, &impactParameterV0); // we do this wrt the primary vtx
160  trackParVarBach.propagateToDCA(primaryVertex, bZ, &impactParameterBach);
161 
162  // get uncertainty of the decay length
163  double phi, theta;
164  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
165  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
166  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
167 
168  // fill candidate table rows
169  MY_DEBUG_MSG(isLc, LOG(INFO) << "IT IS A Lc! Filling for Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg);
170  rowCandidateBase(collision.globalIndex(),
171  collision.posX(), collision.posY(), collision.posZ(),
172  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
173  errorDecayLength, errorDecayLengthXY,
174  chi2PCA,
175  pVecBach[0], pVecBach[1], pVecBach[2],
176  pVecV0[0], pVecV0[1], pVecV0[2],
177  impactParameterBach.getY(), impactParameterV0.getY(),
178  std::sqrt(impactParameterBach.getSigmaY2()), std::sqrt(impactParameterV0.getSigmaY2()),
179  casc.index0Id(), casc.indexV0Id(),
180  casc.hfflag(),
181  v0.x(), v0.y(), v0.z(),
182  //v0.posTrack(), v0.negTrack(), // why this was not fine?
183  trackV0DaughPos.globalIndex(), trackV0DaughNeg.globalIndex(),
184  v0.pxpos(), v0.pypos(), v0.pzpos(),
185  v0.pxneg(), v0.pyneg(), v0.pzneg(),
186  v0.dcaV0daughters(),
187  v0.dcapostopv(),
188  v0.dcanegtopv());
189 
190  // fill histograms
191  if (doValPlots) {
192  // calculate invariant masses
193  mass2K0sP = RecoDecay::M(array{pVecBach, pVecV0}, array{massP, massK0s});
194  hmass2->Fill(mass2K0sP);
195  }
196  }
197  }
198 };

219  void process(aod::HfCandCascade const& candidates,
220  aod::BigTracksMC const& tracks,
221  aod::McParticles const& particlesMC)
222  {
223  int8_t sign = 0;
224  std::vector<int> arrDaughLcIndex;
225  std::array<int, 3> arrDaughLcPDG;
226  std::array<int, 3> arrDaughLcPDGRef = {2212, 211, -211};
227 
228  // Match reconstructed candidates.
229  for (auto& candidate : candidates) {
230 
231  const auto& bach = candidate.index0_as<aod::BigTracksMC>();
232  const auto& trackV0DaughPos = candidate.posTrack_as<aod::BigTracksMC>();
233  const auto& trackV0DaughNeg = candidate.negTrack_as<aod::BigTracksMC>();
234 
235  auto arrayDaughtersV0 = array{trackV0DaughPos, trackV0DaughNeg};
236  auto arrayDaughtersLc = array{bach, trackV0DaughPos, trackV0DaughNeg};
237 
238  // First we check the K0s
239  LOG(DEBUG) << "\n";
240  LOG(DEBUG) << "Checking MC for candidate!";
241  LOG(DEBUG) << "Looking for K0s";
242  auto indexV0DaughPos = trackV0DaughPos.mcParticleId();
243  auto indexV0DaughNeg = trackV0DaughNeg.mcParticleId();
244  auto indexBach = bach.mcParticleId();
245 
246 #ifdef MY_DEBUG
247  bool isLc = isLcK0SpFunc(indexBach, indexV0DaughPos, indexV0DaughNeg, indexProton, indexK0Spos, indexK0Sneg);
248  bool isK0SfromLc = isK0SfromLcFunc(indexV0DaughPos, indexV0DaughNeg, indexK0Spos, indexK0Sneg);
249 #endif
250  MY_DEBUG_MSG(isK0SfromLc, LOG(INFO) << "correct K0S in the Lc daughters: posTrack --> " << indexV0DaughPos << ", negTrack --> " << indexV0DaughNeg);
251 
252  //if (isLc) {
253  RecoDecay::getMatchedMCRec(particlesMC, arrayDaughtersV0, kK0Short, array{+kPiPlus, -kPiPlus}, true, &sign, 1); // does it matter the "acceptAntiParticle" in the K0s case? In principle, there is no anti-K0s
254 
255  if (sign != 0) { // we have already positively checked the K0s
256  // then we check the Lc
257  MY_DEBUG_MSG(sign, LOG(INFO) << "K0S was correct! now we check the Lc");
258  MY_DEBUG_MSG(sign, LOG(INFO) << "index proton = " << indexBach);
259  RecoDecay::getMatchedMCRec(particlesMC, arrayDaughtersLc, pdg::Code::kLambdaCPlus, array{+kProton, +kPiPlus, -kPiPlus}, true, &sign, 3); // 3-levels Lc --> p + K0 --> p + K0s --> p + pi+ pi-
260  MY_DEBUG_MSG(sign, LOG(INFO) << "Lc found with sign " << sign; printf("\n"));
261  }
262 
263  rowMCMatchRec(sign);
264  }
265  //}
266 
267  // Match generated particles.
268  for (auto& particle : particlesMC) {
269  // checking if I have a Lc --> K0S + p
270  RecoDecay::isMatchedMCGen(particlesMC, particle, pdg::Code::kLambdaCPlus, array{+kProton, +kK0Short}, true, &sign, 2);
271  if (sign != 0) {
272  MY_DEBUG_MSG(sign, LOG(INFO) << "Lc in K0S p");
273  arrDaughLcIndex.clear();
274  // checking that the final daughters (decay depth = 3) are p, pi+, pi-
275  RecoDecay::getDaughters(particlesMC, particle, &arrDaughLcIndex, arrDaughLcPDGRef, 3); // best would be to check the K0S daughters
276  if (arrDaughLcIndex.size() == 3) {
277  for (auto iProng = 0; iProng < arrDaughLcIndex.size(); ++iProng) {
278  auto daughI = particlesMC.iteratorAt(arrDaughLcIndex[iProng]);
279  arrDaughLcPDG[iProng] = daughI.pdgCode();
280  }
281  if (!(arrDaughLcPDG[0] == arrDaughLcPDGRef[0] && arrDaughLcPDG[1] == arrDaughLcPDGRef[1] && arrDaughLcPDG[2] == arrDaughLcPDGRef[2])) { // this should be the condition, first bach, then v0
282  sign = 0;
283  } else {
284  LOG(INFO) << "Lc --> K0S+p found in MC table";
285  }
286  MY_DEBUG_MSG(sign == 0, LOG(INFO) << "Pity, the three final daughters are not p, pi+, pi-, but " << arrDaughLcPDG[0] << ", " << arrDaughLcPDG[1] << ", " << arrDaughLcPDG[2]);
287  }
288  }
289  rowMCMatchGen(sign);
290  }
291  }
292 };
