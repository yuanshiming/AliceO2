83  template <typename T>
84  void makelogaxis(T h)
85  {
86  if (logAxis == 0) {
87  return;
88  }
89  const int nbins = h->GetNbinsX();
90  double binp[nbins + 1];
91  double max = h->GetXaxis()->GetBinUpEdge(nbins);
92  double min = h->GetXaxis()->GetBinLowEdge(1);
93  if (min <= 0) {
94  min = 0.00001;
95  }
96  double lmin = TMath::Log10(min);
97  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
98  for (int i = 0; i < nbins; i++) {
99  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
100  }
101  binp[nbins] = max + 1;
102  h->GetXaxis()->Set(nbins, binp);
103  }
