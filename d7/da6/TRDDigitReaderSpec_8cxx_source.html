48 void TRDDigitReaderSpec::run(ProcessingContext& pc)
49 {
50  if (mState != 1) {
51  LOG(info) << "mState is not 1";
52  return;
53  }
54  auto DPLTree = ((TTree*)mFile->Get(mDigitTreeName.c_str()));
55  if (DPLTree) {
56  std::vector<o2::trd::Digit>* digits = nullptr;
57  o2::dataformats::IOMCTruthContainerView* ioLabels = nullptr;
58  std::vector<o2::trd::TriggerRecord>* triggerRecords = nullptr;
59 
60  auto getFromBranch = [DPLTree](const char* name, void** ptr) {
61  auto br = DPLTree->GetBranch(name);
62  br->SetAddress(ptr);
63  br->GetEntry(0);
64  br->ResetAddress();
65  };
66  getFromBranch(mDigitBranchName.c_str(), (void**)&digits);
67  getFromBranch(mTriggerRecordBranchName.c_str(), (void**)&triggerRecords);
68  if (mUseMC) {
69  getFromBranch(mMCLabelsBranchName.c_str(), (void**)&ioLabels);
70  // publish labels in shared memory
71  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{"TRD", "LABELS", 0, Lifetime::Timeframe});
72  ioLabels->copyandflatten(sharedlabels);
73  LOG(info) << "TRDDigitReader labels size (in bytes) = " << sharedlabels.size();
74  }
75 
76  pc.outputs().snapshot(Output{"TRD", "DIGITS", 0, Lifetime::Timeframe}, *digits);
77  pc.outputs().snapshot(Output{"TRD", "TRGRDIG", 0, Lifetime::Timeframe}, *triggerRecords);
78  LOG(info) << "TRDDigitReader digits size=" << digits->size() << " triggerrecords size=" << triggerRecords->size();
79  }
80  //delete DPLTree; // next line will delete the pointer as well.
81  mFile->Close();
82 
83  mState = 2; // prevent coming in here again.
84  // send endOfData control event and mark the reader as ready to finish
85  pc.services().get<ControlService>().endOfStream();
86  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
87 }

90 {

97  return DataProcessorSpec{"TRDDIGITREADER",

100  AlgorithmSpec{adaptFromTask<TRDDigitReaderSpec>(channels, useMC)},

103  {"run2digitsfile", VariantType::String, "run2digits.root", {"Input data file containing run2 digitis going into Trap Simulator"}}}};
