34 struct ProduceEtaPhi {
35  // declare production of table etaphi
36  Produces<aod::EtaPhi> etaphi;
37 
38  void process(aod::Tracks const& tracks)
39  {
40  for (auto& track : tracks) {
41  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
42  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
43 
44  // update the table etaphi
45  etaphi(phi, eta);
46  }
47  }
48 };
