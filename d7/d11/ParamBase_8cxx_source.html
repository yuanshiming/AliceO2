42 void Parameters::LoadParamFromFile(const TString FileName, const TString ParamName)
43 {
44  TFile f(FileName, "READ");
45  if (!f.Get(ParamName)) {
46  LOG(fatal) << "Did not find parameters " << ParamName << " in file " << FileName;
47  }
48  LOG(info) << "Loading parameters " << ParamName << " from TFile " << FileName;
