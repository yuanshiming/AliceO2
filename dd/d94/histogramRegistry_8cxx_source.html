98  void process(aod::Tracks const& tracks)
99  {
100  using namespace aod::track;
101  // does not work with dynamic columns (e.g. Charge, NormalizedPhi)
102  registry.fill<Eta>(HIST("1d"), tracks, eta > -0.7f);
103  registry.fill<Pt, Eta, RawPhi>(HIST("3d"), tracks, eta > 0.f);
104  registry.fill<Pt, Eta, RawPhi, P, X>(HIST("5d"), tracks, pt > 0.15f);
105  registry.fill<Pt, Eta, RawPhi, P, X, Y, Z>(HIST("7d"), tracks, pt > 0.15f);
106  registry.fill<Pt, Eta, RawPhi>(HIST("2d-profile"), tracks, eta > -0.5f);
107 
108  // fill 4d histogram with weight (column X)
109  registry.fill<Pt, Eta, RawPhi, Z, X>(HIST("4d-weight"), tracks, eta > 0.f);
110 
111  registry.fill<Pt, Eta, RawPhi>(HIST("2d-weight"), tracks, eta > 0.f);
112 
113  registry.fill<Pt, Eta, RawPhi>(HIST("1d-profile-weight"), tracks, eta > 0.f);
114 
115  for (auto& track : tracks) {
116  registry.fill(HIST("2d"), track.eta(), track.pt());
117  registry.fill(HIST("4d"), track.pt(), track.eta(), track.phi(), track.signed1Pt());
118  registry.fill(HIST("6d"), track.pt(), track.eta(), track.phi(), track.snp(), track.tgl(), track.alpha());
119  registry.fill(HIST("1d-profile"), track.pt(), track.eta());
120  registry.fill(HIST("3d-profile"), track.pt(), track.eta(), track.phi(), track.snp());
121 
122  // fill 3d histogram with weight (2.)
123  registry.fill(HIST("3d-weight"), track.pt(), track.eta(), track.phi(), 2.);
124 
125  registry.fill(HIST("2d-profile-weight"), track.pt(), track.eta(), track.phi(), 5.);
126 
127  registry.fill(HIST("2d-step"), 1, track.pt(), track.eta());
128  registry.fill(HIST("2d-step-weight"), 2, track.pt(), track.eta(), track.phi());
129  }
130  }
131 };

154  spectra.get<TH2>(HIST("myControlHist"))->SetTitle("something meaningful");

156  spectra.add("charged/pions", "Pions", defaultParticleHist);
157  spectra.add("neutral/pions", "Pions", defaultParticleHist);
158  spectra.add("one/two/three/four/kaons", "Kaons", defaultParticleHist);
159  spectra.add("sigmas", "Sigmas", defaultParticleHist);
160  spectra.add("lambdas", "Lambd", defaultParticleHist);
161 
162  spectra.get<THn>(HIST("lambdas"))->SetTitle("Lambdas");
163 
164  etaStudy.add("positive", "A side spectra", kTH1I, {ptAxis});
165  etaStudy.add("negative", "C side spectra", kTH1I, {ptAxis});
166 
167  spectra.add("before_cuts/hist1", "asdf", defaultParticleHist);
168  spectra.add("before_cuts/hist2", "asdf", defaultParticleHist);
169  spectra.add("before_cuts/hist3", "asdf", defaultParticleHist);
170 
171  // clone whole category / group
172  spectra.addClone("before_cuts/", "after_cuts/");

179  template <bool mode, typename T>
180  void fillHistos(const T& track)
181  {
182  static constexpr std::string_view subDir[] = {"before_cuts/", "after_cuts/"};
183 
184  spectra.fill(HIST(subDir[mode]) + HIST("hist1"), track.pt(), track.eta(), 50., 0.);
185  spectra.fill(HIST(subDir[mode]) + HIST("hist2"), track.pt(), track.eta(), 50., 0.);
186  spectra.fill(HIST(subDir[mode]) + HIST("hist3"), track.pt(), track.eta(), 50., 0.);
187  }

189  void process(aod::Tracks const& tracks)
190  {
191  using namespace aod::track;
192 
193  etaStudy.fill<Pt>(HIST("positive"), tracks, eta > 0.f);
194  etaStudy.fill<Pt>(HIST("negative"), tracks, eta < 0.f);
195 
196  for (auto& track : tracks) {
197  spectra.fill(HIST("myControlHist"), track.pt(), track.eta());
198  spectra.fill(HIST("charged/pions"), track.pt(), track.eta(), 50., 0.);
199  spectra.fill(HIST("charged/pions"), track.pt(), track.eta(), 50., 0.);
200  spectra.fill(HIST("neutral/pions"), track.pt(), track.eta(), 50., 0.);
201  spectra.fill(HIST("one/two/three/four/kaons"), track.pt(), track.eta(), 50., 0.);
202  spectra.fill(HIST("sigmas"), track.pt(), track.eta(), 50., 0.);
203  spectra.fill(HIST("lambdas"), track.pt(), track.eta(), 50., 0.);
204 
205  // fill histograms before and after cuts
206  fillHistos<false>(track);
207  if (std::rand() > (RAND_MAX / 2)) {
208  fillHistos<true>(track);
209  }
210 
211  spectra.fill(HIST("cascades"), track.pt(), track.eta(), 50., 0.);
212  spectra.fill(HIST("strange/funny/particles"), track.pt(), track.eta(), 50., 0.);
213  }
214  }
