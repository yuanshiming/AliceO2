52  void process(soa::Join<aod::BCs, aod::Run2BCInfos, aod::BcSels>::iterator const& bc)
53  {
54  // Fill fired trigger classes
55  uint64_t triggerMask = bc.triggerMask();
56  uint64_t triggerMaskNext50 = bc.triggerMaskNext50();
57  for (int i = 0; i < 50; i++) {
58  if (triggerMask & 1ull << i) {
59  hFiredClasses->Fill(i);
60  }
61  if (triggerMaskNext50 & 1ull << i) {
62  hFiredClasses->Fill(i + 50);
63  }
64  }
65 
66  // Fill fired aliases
67  for (int i = 0; i < kNaliases; i++) {
68  if (bc.alias()[i]) {
69  hFiredAliases->Fill(i);
70  }
71  }
72  }
73 };

139  float multV0M = multV0A + multV0C;
