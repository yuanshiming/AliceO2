64  template <typename T1, typename T2>
65  bool selectionTopol(const T1& candidate, const T2& trackPositron, const T2& trackElectron)
66  {
67  auto candpT = candidate.pt();
68  auto pTBin = findBin(pTBins, candpT);
69  if (pTBin == -1) {
70  return false;
71  }
72 
73  // check that the candidate pT is within the analysis range
74  if (candpT < d_pTCandMin || candpT >= d_pTCandMax) {
75  return false;
76  }
77 
78  // cut on invariant mass
79  if (std::abs(InvMassJpsiToEE(candidate) - RecoDecay::getMassPDG(pdg::Code::kJpsi)) > cuts->get(pTBin, "m")) {
80  return false;
81  }
82 
83  // cut on daughter pT
84  if (trackElectron.pt() < cuts->get(pTBin, "pT El") || trackPositron.pt() < cuts->get(pTBin, "pT El")) {
85  return false;
86  }
87 
88  // cut on daughter DCA - need to add secondary vertex constraint here
89  if (std::abs(trackElectron.dcaPrim0()) > cuts->get(pTBin, "DCA_xy") || std::abs(trackPositron.dcaPrim0()) > cuts->get(pTBin, "DCA_xy")) {
90  return false;
91  }
92 
93  // cut on daughter DCA - need to add secondary vertex constraint here
94  if (std::abs(trackElectron.dcaPrim1()) > cuts->get(pTBin, "DCA_z") || std::abs(trackPositron.dcaPrim1()) > cuts->get(pTBin, "DCA_z")) {
95  return false;
96  }
97 
98  return true;
99  }
