25 struct TrackExtensionTask {
26  Configurable<float> magField{"magField", 5.f, "Magnetic field for the propagation to the primary vertex in kG"};
27  Produces<aod::TracksExtended> extendedTrackQuantities;
28 
29  void process(aod::Collision const& collision, aod::FullTracks const& tracks)
30  {
31  std::array<float, 2> dca{1e10f, 1e10f};
32  for (auto& track : tracks) {
33  auto trackPar = getTrackPar(track);
34  trackPar.propagateParamToDCA({collision.posX(), collision.posY(), collision.posZ()}, magField, &dca);
35  extendedTrackQuantities(dca[0], dca[1]);
36  }
37  }
38 };
