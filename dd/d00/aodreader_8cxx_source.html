119 struct aodReader {
120  void process(aod::Eins const& unos, aod::Zwei const& dues, aod::Drei const& tres)
121  {
122  int cnt = 0;
123  for (auto& uno : unos) {
124  auto eta = uno.eta();
125  auto mom = uno.mom();
126 
127  cnt++;
128  LOGF(INFO, "Eins (%i): (%f, %f)", cnt, eta, mom);
129  }
130  LOGF(INFO, "ATask Processed %i data points from Eins", cnt);
131 
132  cnt = 0;
133  for (auto& due : dues) {
134  auto ok = due.ok();
135  auto phi = due.phi();
136  auto pt = due.pt();
137 
138  cnt++;
139  LOGF(INFO, "Zwei (%i): (%i, %f, %f)", cnt, ok, phi, pt);
140  }
141  LOGF(INFO, "ATask Processed %i data points from Zwei", cnt);
142 
143  cnt = 0;
144  for (auto& tre : tres) {
145  auto eta = tre.eta();
146  auto phi = tre.phi();
147  auto mom = tre.mom();
148 
149  cnt++;
150  LOGF(INFO, "Drei (%i): (%f, %f, %f)", cnt, eta, phi, mom);
151  }
152  LOGF(INFO, "ATask Processed %i data points from Drei", cnt);
153  }
154 };
