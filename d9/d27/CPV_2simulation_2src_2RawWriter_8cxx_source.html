106 bool RawWriter::processOrbit(const gsl::span<o2::cpv::Digit> digitsbranch, const gsl::span<o2::cpv::TriggerRecord> trgs)
107 {
108  //Array used to sort properly digits
109  for (int i = kNcc; i--;) {
110  for (int j = kNDilogic; j--;) {
111  for (int k = kNGasiplex; k--;) {
112  mPadCharge[i][j][k].clear();
113  }
114  }
115  }
116  //clear payload
117  mPayload.clear();
118 
119  //we're going to prepare preformatted pages
120  int feeid = 0, cru = 0, link = 0, endpoint = 0;
121  bool preformatted = true;
122 
123  int gbtWordCounter = 0;
124  int gbtWordCounterBeforeCPVTrailer = 0;
125  int nMaxGbtWordsPerPage = o2::raw::RDHUtils::MAXCRUPage / o2::raw::RDHUtils::GBTWord; //512*16/16 = 512;
126  nMaxGbtWordsPerPage -= 4; //rdh size = 4 gbt words
127  for (auto& trg : trgs) {
128  gbtWordCounterBeforeCPVTrailer = 0; //every trigger we start this counter from 0
129 
130  //make payload for current trigger
131  int nDigsInTrg = 0;
132  for (auto& dig : gsl::span(digitsbranch.data() + trg.getFirstEntry(), trg.getNumberOfObjects())) {
133 
134  short absId = dig.getAbsId();
135  short ccId, dil, gas, pad;
136  o2::cpv::Geometry::absIdToHWaddress(absId, ccId, dil, gas, pad);
137 
138  //Convert Amp to ADC counts
139  short charge = dig.getAmplitude() / mCalibParams->getGain(absId) + mPedestals->getPedestal(absId);
140  if (charge > 4095) {
141  charge = 4095;
142  }
143  mPadCharge[ccId][dil][gas].emplace_back(charge, pad);
144  nDigsInTrg++;
145  }
146 
147  //we need to write header + trailer + at least 1 payload word
148  if (nMaxGbtWordsPerPage - gbtWordCounter < 3) { //otherwise flush already prepared data to file
149  LOG(DEBUG) << "RawWriter::processOrbit() : before header: adding preformatted dma page of size " << mPayload.size();
150  mRawWriter->addData(feeid, cru, link, endpoint, trg.getBCData(), gsl::span<char>(mPayload.data(), mPayload.size()), preformatted);
151  mPayload.clear();
152  gbtWordCounter = 0;
153  gbtWordCounterBeforeCPVTrailer = 0;
154  }
155 
156  //first, header goes
157  CpvHeader header(trg.getBCData(), false, false);
158  for (int i = 0; i < 16; i++) {
159  mPayload.push_back(header.mBytes[i]);
160  }
161  gbtWordCounter++;
162  gbtWordCounterBeforeCPVTrailer++;
163 
164  int ccWordCounter = 0;
165  for (char ccId = 0; ccId < kNcc; ccId++) {
166  for (char dil = 0; dil < kNDilogic; dil++) {
167  for (char gas = 0; gas < kNGasiplex; gas++) {
168  for (padCharge& pc : mPadCharge[ccId][dil][gas]) {
169  // Generate 3 CC words, add CC header and empty bits to complete 128 bits;
170  PadWord currentword = {0};
171  currentword.charge = pc.charge;
172  currentword.address = pc.pad;
173  currentword.gas = gas;
174  currentword.dil = dil;
175  mPayload.push_back(currentword.mBytes[0]);
176  mPayload.push_back(currentword.mBytes[1]);
177  mPayload.push_back(currentword.mBytes[2]);
178  ccWordCounter++;
179  if (ccWordCounter % 3 == 0) { // complete 3 channels (72 bit) + CC index (8 bits) + 6 empty bits = Generate 128 bits of data
180  mPayload.push_back(ccId);
181  for (int i = 6; i--;) {
182  mPayload.push_back(char(0));
183  }
184  gbtWordCounter++;
185  gbtWordCounterBeforeCPVTrailer++;
186  if (nMaxGbtWordsPerPage - gbtWordCounter < 2) { //the only space for trailer left on current page
187  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer, trg.getBCData().bc, true); //add trailer and flush page to file
188  for (int i = 0; i < 16; i++) {
189  mPayload.push_back(tr.mBytes[i]);
190  }
191  LOG(DEBUG) << "RawWriter::processOrbit() : middle of payload: adding preformatted dma page of size " << mPayload.size();
192  mRawWriter->addData(feeid, cru, link, endpoint, trg.getBCData(), gsl::span<char>(mPayload.data(), mPayload.size()), preformatted);
193  mPayload.clear();
194  gbtWordCounter = 0;
195  gbtWordCounterBeforeCPVTrailer = 0;
196  if (nDigsInTrg > gbtWordCounterBeforeCPVTrailer - 1) { //some digits left for writing
197  for (int i = 0; i < 16; i++) { //so put a new header and continue
198  mPayload.push_back(header.mBytes[i]);
199  }
200  gbtWordCounter++;
201  gbtWordCounterBeforeCPVTrailer++;
202  }
203  }
204  }
205  }
206  if (ccWordCounter % 3 != 0) {
207  while (ccWordCounter % 3 != 0) {
208  mPayload.push_back(char(255));
209  mPayload.push_back(char(255));
210  mPayload.push_back(char(255));
211  ccWordCounter++;
212  }
213  mPayload.push_back(ccId);
214  for (int i = 6; i--;) {
215  mPayload.push_back(char(0));
216  }
217  gbtWordCounter++;
218  gbtWordCounterBeforeCPVTrailer++;
219  if (nMaxGbtWordsPerPage - gbtWordCounter < 2) { //the only space for trailer left on current page
220  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer, trg.getBCData().bc, true); //add trailer and flush page to file
221  for (int i = 0; i < 16; i++) {
222  mPayload.push_back(tr.mBytes[i]);
223  }
224  LOG(DEBUG) << "RawWriter::processOrbit() : middle of payload (after filling empty words): adding preformatted dma page of size " << mPayload.size();
225  mRawWriter->addData(feeid, cru, link, endpoint, trg.getBCData(), gsl::span<char>(mPayload.data(), mPayload.size()), preformatted);
226  mPayload.clear();
227  gbtWordCounter = 0;
228  gbtWordCounterBeforeCPVTrailer = 0;
229  if (nDigsInTrg > gbtWordCounterBeforeCPVTrailer - 1) { //some digits left for writing
230  for (int i = 0; i < 16; i++) { //so put a new header and continue
231  mPayload.push_back(header.mBytes[i]);
232  }
233  gbtWordCounter++;
234  gbtWordCounterBeforeCPVTrailer++;
235  }
236  }
237  }
238  }
239  }
240  }
241  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer, trg.getBCData().bc, true); //cout GBT words
242  for (int i = 0; i < 16; i++) {
243  mPayload.push_back(tr.mBytes[i]);
244  }
245  gbtWordCounterBeforeCPVTrailer++;
246  gbtWordCounter++;
247  } //end of for (auto& trg : trgs)
248  if (mPayload.size()) { //flush payload to file (if any)
249  LOG(DEBUG) << "RawWriter::processOrbit() : final payload: adding preformatted dma page of size " << mPayload.size();
250  mRawWriter->addData(feeid, cru, link, endpoint, trgs.back().getBCData(), gsl::span<char>(mPayload.data(), mPayload.size()), preformatted);
251  mPayload.clear();
252  }
253  return true;
254 }

256 int RawWriter::carryOverMethod(const header::RDHAny* rdh, const gsl::span<char> data,
257  const char* ptr, int maxSize, int splitID,
258  std::vector<char>& trailer, std::vector<char>& header) const
259 {
260 
261  constexpr int cpvTrailerSize = 36;
262  int offs = ptr - &data[0]; // offset wrt the head of the payload
263  assert(offs >= 0 && size_t(offs + maxSize) <= data.size()); // make sure ptr and end of the suggested block are within the payload
264  int leftBefore = data.size() - offs; // payload left before this splitting
265  int leftAfter = leftBefore - maxSize; // what would be left after the suggested splitting
266  int actualSize = maxSize;
267  if (leftAfter && leftAfter <= cpvTrailerSize) { // avoid splitting the trailer or writing only it.
268  actualSize -= (cpvTrailerSize - leftAfter) + 4; // (as we work with int, not char in decoding)
269  }
270  return actualSize;
271 }
