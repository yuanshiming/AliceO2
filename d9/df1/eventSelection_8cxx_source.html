31 struct EvSelParameters {
32  // time-of-flight offset
33  float fV0ADist = 329.00 / TMath::Ccgs() * 1e9; // ns
34  float fV0CDist = 87.15 / TMath::Ccgs() * 1e9; // ns
35 
36  float fFDADist = (1695.30 + 1698.04) / TMath::Ccgs() * 1e9; // ns
37  float fFDCDist = (1952.90 + 1955.90) / TMath::Ccgs() * 1e9; // ns
38 
39  // beam-beam and beam-gas windows
40  float fV0ABBlower = +fV0ADist - 9.5; // ns
41  float fV0ABBupper = +fV0ADist + 22.5; // ns
42  float fV0ABGlower = -fV0ADist - 2.5; // ns
43  float fV0ABGupper = -fV0ADist + 5.0; // ns
44  float fV0CBBlower = +fV0CDist - 2.5; // ns
45  float fV0CBBupper = +fV0CDist + 22.5; // ns
46  float fV0CBGlower = -fV0CDist - 2.5; // ns
47  float fV0CBGupper = -fV0CDist + 2.5; // ns
48 
49  float fFDABBlower = +fFDADist - 2.5; // ns
50  float fFDABBupper = +fFDADist + 2.5; // ns
51  float fFDABGlower = -fFDADist - 4.0; // ns
52  float fFDABGupper = -fFDADist + 4.0; // ns
53 
54  float fFDCBBlower = +fFDCDist - 1.5; // ns
55  float fFDCBBupper = +fFDCDist + 1.5; // ns
56  float fFDCBGlower = -fFDCDist - 2.0; // ns
57  float fFDCBGupper = -fFDCDist + 2.0; // ns
58 
59  float fZNABBlower = -2.0; // ns
60  float fZNABBupper = 2.0; // ns
61  float fZNCBBlower = -2.0; // ns
62  float fZNCBBupper = 2.0; // ns
63  float fZNABGlower = 5.0; // ns
64  float fZNABGupper = 100.0; // ns
65  float fZNCBGlower = 5.0; // ns
66  float fZNCBGupper = 100.0; // ns
67 
68  // TODO rough cuts to be adjusted
69  float fT0ABBlower = -2.0; // ns
70  float fT0ABBupper = 2.0; // ns
71  float fT0CBBlower = -2.0; // ns
72  float fT0CBBupper = 2.0; // ns
73 
74  // Default values from AliOADBTriggerAnalysis constructor
75  // TODO store adjusted values period-by-period in CCDB
76  float fSPDClsVsTklA = 65.f;
77  float fSPDClsVsTklB = 4.f;
78  float fV0C012vsTklA = 150.f;
79  float fV0C012vsTklB = 20.f;
80  float fV0MOnVsOfA = -59.56f;
81  float fV0MOnVsOfB = 5.22f;
82  float fSPDOnVsOfA = -5.62f;
83  float fSPDOnVsOfB = 0.85f;
84  float fV0CasymA = -25.f;
85  float fV0CasymB = 0.15f;
86 
87  bool applySelection[kNsel] = {0};
88 };

90 struct BcSelectionTask {
91  Produces<aod::BcSels> bcsel;
92  Service<o2::ccdb::BasicCCDBManager> ccdb;
93  EvSelParameters par;
94  void init(InitContext&)
95  {
96  ccdb->setURL("http://ccdb-test.cern.ch:8080");
97  ccdb->setCaching(true);
98  ccdb->setLocalObjectValidityChecking();
99  }
100 
101  using BCsWithRun2InfosTimestampsAndMatches = soa::Join<aod::BCs, aod::Run2BCInfos, aod::Timestamps, aod::Run2MatchedToBCSparse>;
102 
103  void process(
104  BCsWithRun2InfosTimestampsAndMatches::iterator const& bc,
105  aod::Zdcs const& zdcs,
106  aod::FV0As const& fv0as,
107  aod::FV0Cs const& fv0cs,
108  aod::FT0s const& ft0s,
109  aod::FDDs const& fdds)
110  {
111  TriggerAliases* aliases = ccdb->getForTimeStamp<TriggerAliases>("Trigger/TriggerAliases", bc.timestamp());
112  if (!aliases) {
113  LOGF(fatal, "Trigger aliases are not available in CCDB for run=%d at timestamp=%llu", bc.runNumber(), bc.timestamp());
114  }
115 
116  // fill fired aliases
117  int32_t alias[kNaliases] = {0};
118  uint64_t triggerMask = bc.triggerMask();
119  for (auto& al : aliases->GetAliasToTriggerMaskMap()) {
120  alias[al.first] |= (triggerMask & al.second) > 0;
121  }
122  uint64_t triggerMaskNext50 = bc.triggerMaskNext50();
123  for (auto& al : aliases->GetAliasToTriggerMaskNext50Map()) {
124  alias[al.first] |= (triggerMaskNext50 & al.second) > 0;
125  }
126 
127  // get timing info from ZDC, FV0, FT0 and FDD
128  float timeZNA = bc.has_zdc() ? bc.zdc().timeZNA() : -999.f;
129  float timeZNC = bc.has_zdc() ? bc.zdc().timeZNC() : -999.f;
130  float timeV0A = bc.has_fv0a() ? bc.fv0a().time() : -999.f;
131  float timeV0C = bc.has_fv0c() ? bc.fv0c().time() : -999.f;
132  float timeT0A = bc.has_ft0() ? bc.ft0().timeA() : -999.f;
133  float timeT0C = bc.has_ft0() ? bc.ft0().timeC() : -999.f;
134  float timeFDA = bc.has_fdd() ? bc.fdd().timeA() : -999.f;
135  float timeFDC = bc.has_fdd() ? bc.fdd().timeC() : -999.f;
136 
137  LOGF(debug, "timeZNA=%f timeZNC=%f", timeZNA, timeZNC);
138  LOGF(debug, "timeV0A=%f timeV0C=%f", timeV0A, timeV0C);
139  LOGF(debug, "timeFDA=%f timeFDC=%f", timeFDA, timeFDC);
140  LOGF(debug, "timeT0A=%f timeT0C=%f", timeT0A, timeT0C);
141 
142  // applying timing selections
143  bool bbV0A = timeV0A > par.fV0ABBlower && timeV0A < par.fV0ABBupper;
144  bool bbV0C = timeV0C > par.fV0CBBlower && timeV0C < par.fV0CBBupper;
145  bool bbFDA = timeFDA > par.fFDABBlower && timeFDA < par.fFDABBupper;
146  bool bbFDC = timeFDC > par.fFDCBBlower && timeFDC < par.fFDCBBupper;
147  bool bgV0A = timeV0A > par.fV0ABGlower && timeV0A < par.fV0ABGupper;
148  bool bgV0C = timeV0C > par.fV0CBGlower && timeV0C < par.fV0CBGupper;
149  bool bgFDA = timeFDA > par.fFDABGlower && timeFDA < par.fFDABGupper;
150  bool bgFDC = timeFDC > par.fFDCBGlower && timeFDC < par.fFDCBGupper;
151 
152  // fill time-based selection criteria
153  int32_t selection[kNsel] = {0}; // TODO switch to bool array
154  selection[kIsBBV0A] = bbV0A;
155  selection[kIsBBV0C] = bbV0C;
156  selection[kIsBBFDA] = bgFDA;
157  selection[kIsBBFDC] = bgFDC;
158  selection[kNoBGFDA] = !bgFDA;
159  selection[kNoBGFDC] = !bgFDC;
160  selection[kNoBGFDA] = !bgFDA;
161  selection[kNoBGFDC] = !bgFDC;
162  selection[kIsBBT0A] = timeT0A > par.fT0ABBlower && timeT0A < par.fT0ABBupper;
163  selection[kIsBBT0C] = timeT0C > par.fT0CBBlower && timeT0C < par.fT0CBBupper;
164  selection[kIsBBZNA] = timeZNA > par.fZNABBlower && timeZNA < par.fZNABBupper;
165  selection[kIsBBZNC] = timeZNC > par.fZNCBBlower && timeZNC < par.fZNCBBupper;
166  selection[kNoBGZNA] = !(fabs(timeZNA) > par.fZNABGlower && fabs(timeZNA < par.fZNABGupper));
167  selection[kNoBGZNC] = !(fabs(timeZNC) > par.fZNCBGlower && fabs(timeZNC < par.fZNCBGupper));
168 
169  // Calculate V0 multiplicity per ring
170  float multRingV0A[5] = {0.};
171  float multRingV0C[4] = {0.};
172  float multV0A = 0;
173  float multV0C = 0;
174  if (bc.has_fv0a()) {
175  for (int ring = 0; ring < 4; ring++) { // TODO adjust for Run3 V0A
176  for (int sector = 0; sector < 8; sector++) {
177  multRingV0A[ring] += bc.fv0a().amplitude()[ring * 8 + sector];
178  }
179  multV0A += multRingV0A[ring];
180  }
181  }
182 
183  if (bc.has_fv0c()) {
184  for (int ring = 0; ring < 4; ring++) {
185  for (int sector = 0; sector < 8; sector++) {
186  multRingV0C[ring] += bc.fv0c().amplitude()[ring * 8 + sector];
187  }
188  multV0C += multRingV0C[ring];
189  }
190  }
191  uint32_t spdClusters = bc.spdClustersL0() + bc.spdClustersL1();
192 
193  // Calculate pileup and background related selection flags
194  float multV0C012 = multRingV0C[0] + multRingV0C[1] + multRingV0C[2];
195  float ofV0M = multV0A + multV0C;
196  float onV0M = bc.v0TriggerChargeA() + bc.v0TriggerChargeC();
197  float ofSPD = bc.spdFiredChipsL0() + bc.spdFiredChipsL1();
198  float onSPD = bc.spdFiredFastOrL0() + bc.spdFiredFastOrL1();
199  selection[kNoV0MOnVsOfPileup] = onV0M > par.fV0MOnVsOfA + par.fV0MOnVsOfB * ofV0M;
200  selection[kNoSPDOnVsOfPileup] = onSPD > par.fSPDOnVsOfA + par.fSPDOnVsOfB * ofSPD;
201  selection[kNoV0Casymmetry] = multRingV0C[3] > par.fV0CasymA + par.fV0CasymB * multV0C012;
202 
203  // copy remaining selection decisions from eventCuts
204  uint32_t eventCuts = bc.eventCuts();
205  selection[kIsGoodTimeRange] = (eventCuts & 1 << aod::kTimeRangeCut) > 0;
206  selection[kNoIncompleteDAQ] = (eventCuts & 1 << aod::kIncompleteDAQ) > 0;
207  selection[kNoTPCLaserWarmUp] = (eventCuts & 1 << aod::kIsTPCLaserWarmUp) == 0;
208  selection[kNoTPCHVdip] = (eventCuts & 1 << aod::kIsTPCHVdip) == 0;
209  selection[kNoPileupFromSPD] = (eventCuts & 1 << aod::kIsPileupFromSPD) == 0;
210  selection[kNoV0PFPileup] = (eventCuts & 1 << aod::kIsV0PFPileup) == 0;
211 
212  int64_t foundFT0 = bc.has_ft0() ? bc.ft0().globalIndex() : -1;
213 
214  // Fill bc selection columns
215  bcsel(alias, selection,
216  bbV0A, bbV0C, bgV0A, bgV0C,
217  bbFDA, bbFDC, bgFDA, bgFDC,
218  multRingV0A, multRingV0C, spdClusters, foundFT0);
219  }
220 };

369  void process(aod::Collision const& col, BCsWithBcSels const& bcs, aod::Tracks const& tracks)
370  {
371  auto bc = col.bc_as<BCsWithBcSels>();
372  int64_t foundFT0 = bc.foundFT0();
373 
374  // copy alias decisions from bcsel table
375  int32_t alias[kNaliases];
376  for (int i = 0; i < kNaliases; i++) {
377  alias[i] = bc.alias()[i];
378  }
379 
380  // copy selection decisions from bcsel table
381  int32_t selection[kNsel] = {0};
382  for (int i = 0; i < kNsel; i++) {
383  selection[i] = bc.selection()[i];
384  }
385 
386  // copy multiplicity per ring and calculate V0C012 multiplicity
387  float multRingV0A[5] = {0.};
388  float multRingV0C[4] = {0.};
389  for (int i = 0; i < 5; i++) {
390  multRingV0A[i] = bc.multRingV0A()[i];
391  }
392  for (int i = 0; i < 4; i++) {
393  multRingV0C[i] = bc.multRingV0C()[i];
394  }
395  float multV0C012 = bc.multRingV0C()[0] + bc.multRingV0C()[1] + bc.multRingV0C()[2];
396 
397  // applying selections depending on the number of tracklets
398  int nTkl = tracklets.size();
399  uint32_t spdClusters = bc.spdClusters();
400  selection[kNoSPDClsVsTklBG] = spdClusters < par.fSPDClsVsTklA + nTkl * par.fSPDClsVsTklB;
401  selection[kNoV0C012vsTklBG] = !(nTkl < 6 && multV0C012 > par.fV0C012vsTklA + nTkl * par.fV0C012vsTklB);
402 
403  // copy beam-beam and beam-gas flags from bcsel table
404  bool bbV0A = bc.bbV0A();
405  bool bbV0C = bc.bbV0C();
406  bool bgV0A = bc.bgV0A();
407  bool bgV0C = bc.bgV0C();
408  bool bbFDA = bc.bbFDA();
409  bool bbFDC = bc.bbFDC();
410  bool bgFDA = bc.bgFDA();
411  bool bgFDC = bc.bgFDC();
412 
413  // apply int7-like selections
414  bool sel7 = 1;
415  for (int i = 0; i < kNsel; i++) {
416  sel7 &= par.applySelection[i] ? selection[i] : 1;
417  }
418 
419  // TODO apply other cuts for sel8
420  // TODO introduce sel1 etc?
421  // TODO introduce array of sel[0]... sel[8] or similar?
422  bool sel8 = selection[kIsBBT0A] & selection[kIsBBT0C];
423 
424  evsel(alias, selection,
425  bbV0A, bbV0C, bgV0A, bgV0C,
426  bbFDA, bbFDC, bgFDA, bgFDC,
427  multRingV0A, multRingV0C, spdClusters, nTkl, sel7, sel8,
428  foundFT0);
429  }
430 };

435  void process(aod::Collision const& col, BCsWithBcSels const& bcs)
