50  GPUd() trackInterface<GPUTRDO2BaseTrack>(const o2::dataformats::TrackTPCITS& trkItsTpc, float vDrift) : GPUTRDO2BaseTrack(trkItsTpc.getParamOut())
51  {
52  mTime = trkItsTpc.getTimeMUS().getTimeStamp();
53  mTimeAddMax = trkItsTpc.getTimeMUS().getTimeStampError();
54  mTimeSubMax = trkItsTpc.getTimeMUS().getTimeStampError();
55  mRefITS = trkItsTpc.getRefITS();
56  mRefTPC = trkItsTpc.getRefTPC();
57  float tmp = trkItsTpc.getTimeMUS().getTimeStampError() * vDrift;
58  updateCov(tmp * tmp, o2::track::CovLabels::kSigZ2); // account for time uncertainty by increasing sigmaZ2
59  }
60  GPUd() trackInterface<GPUTRDO2BaseTrack>(const o2::tpc::TrackTPC& trkTpc, float tbWidth, float vDrift, unsigned int iTrk) : GPUTRDO2BaseTrack(trkTpc.getParamOut())
61  {
62  mRefTPC = {iTrk, o2::dataformats::GlobalTrackID::TPC};
63  mTime = trkTpc.getTime0() * tbWidth;
64  mTimeAddMax = trkTpc.getDeltaTFwd() * tbWidth;
65  mTimeSubMax = trkTpc.getDeltaTBwd() * tbWidth;
66  if (trkTpc.hasASideClustersOnly()) {
67  mSide = -1;
68  } else if (trkTpc.hasCSideClustersOnly()) {
69  mSide = 1;
70  } else {
71  // CE-crossing tracks are not shifted along z, but the time uncertainty is taken into account by increasing sigmaZ2
72  float timeWindow = (mTimeAddMax + mTimeSubMax) * .5f;
73  float tmp = timeWindow * vDrift;
74  updateCov(tmp * tmp, o2::track::CovLabels::kSigZ2);
75  }
76  }
77  GPUd() void set(float x, float alpha, const float* param, const float* cov)
78  {
79  setX(x);
80  setAlpha(alpha);
81  for (int i = 0; i < 5; i++) {
82  setParam(param[i], i);
83  }
84  for (int i = 0; i < 15; i++) {
85  setCov(cov[i], i);
86  }
87  }
