87  template <typename T>
88  bool isSelectedTrackPIDTPC(const T& track, bool& conditionalTOF)
89  {
90  // Accept if selection is disabled via large values.
91  if (mNSigmaTPCMin < -999. && mNSigmaTPCMax > 999.) {
92  return true;
93  }
94 
95  // Get nσ for a given particle hypothesis.
96  double nSigma = 100.;
97  switch (mPdg) {
98  case kElectron: {
99  nSigma = track.tpcNSigmaEl();
100  break;
101  }
102  case kMuonMinus: {
103  nSigma = track.tpcNSigmaMu();
104  break;
105  }
106  case kPiPlus: {
107  nSigma = track.tpcNSigmaPi();
108  break;
109  }
110  case kKPlus: {
111  nSigma = track.tpcNSigmaKa();
112  break;
113  }
114  case kProton: {
115  nSigma = track.tpcNSigmaPr();
116  break;
117  }
118  default: {
119  LOGF(error, "ERROR: TPC PID not implemented for PDG %d", mPdg);
120  assert(false);
121  }
122  }
123 
124  if (mNSigmaTPCMinCondTOF < -999. && mNSigmaTPCMaxCondTOF > 999.) {
125  conditionalTOF = true;
126  } else {
127  conditionalTOF = mNSigmaTPCMinCondTOF <= nSigma && nSigma <= mNSigmaTPCMaxCondTOF;
128  }
129  return mNSigmaTPCMin <= nSigma && nSigma <= mNSigmaTPCMax;
130  }

135  template <typename T>
136  int getStatusTrackPIDTPC(const T& track)
137  {
138  if (isValidTrackPIDTPC(track)) {
139  bool condTOF = false;
140  if (isSelectedTrackPIDTPC(track, condTOF)) {
141  return Status::PIDAccepted; // accepted

143  return Status::PIDConditional; // potential to be accepted if combined with TOF
144  } else {
145  return Status::PIDRejected; // rejected
146  }
147  } else {
148  return Status::PIDNotApplicable; // PID not applicable
149  }

257  template <typename T>
258  int getStatusTrackPIDAll(const T& track)
259  {
260  int statusTPC = getStatusTrackPIDTPC(track);
261  int statusTOF = getStatusTrackPIDTOF(track);
262 
263  if (statusTPC == Status::PIDAccepted || statusTOF == Status::PIDAccepted) {
264  return Status::PIDAccepted; // what if we have Accepted for one and Rejected for the other?
265  }
266  if (statusTPC == Status::PIDConditional && statusTOF == Status::PIDConditional) {
267  return Status::PIDAccepted;
268  }
269  if (statusTPC == Status::PIDRejected || statusTOF == Status::PIDRejected) {
270  return Status::PIDRejected;
271  }
272  return Status::PIDNotApplicable; // (NotApplicable for one detector) and (NotApplicable or Conditional for the other)
273  }
