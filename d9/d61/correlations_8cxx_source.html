160  bool fillCollision(TTarget target, TCollision collision, float centrality)
161  {
162  same->fillEvent(centrality, CorrelationContainer::kCFStepAll);
163 
164  if (!collision.alias()[kINT7] || !collision.sel7()) {
165  return false;
166  }
167 
168  same->fillEvent(centrality, CorrelationContainer::kCFStepTriggered);
169 
170  // vertex range already checked as filter, but bitwise operations not yet supported
171  // TODO (collision.flags() != 0) can be removed with next conversion (AliPhysics >= 20210305)
172  if ((collision.flags() != 0) && ((collision.flags() & aod::collision::CollisionFlagsRun2::Run2VertexerTracks) != aod::collision::CollisionFlagsRun2::Run2VertexerTracks)) {
173  return false;
174  }
175 
176  same->fillEvent(centrality, CorrelationContainer::kCFStepVertex);
177  same->fillEvent(centrality, CorrelationContainer::kCFStepReconstructed);

182  template <typename TTarget, typename TTracks>
183  void fillCorrelations(TTarget target, TTracks tracks1, TTracks tracks2, float centrality, float posZ, int bSign)
184  {
185  // Cache efficiency for particles (too many FindBin lookups)
186  float* efficiencyAssociated = nullptr;
187  if (cfg.mEfficiencyAssociated) {
188  efficiencyAssociated = new float[tracks2.size()];
189  int i = 0;
190  for (auto& track : tracks2) {
191  efficiencyAssociated[i++] = getEfficiency(cfg.mEfficiencyAssociated, track.eta(), track.pt(), centrality, posZ);
192  }
193  }
194 
195  for (auto& track1 : tracks1) {
196  // LOGF(info, "Track %f | %f | %f %d %d", track1.eta(), track1.phi(), track1.pt(), track1.isGlobalTrack(), track1.isGlobalTrackSDD());
197 
198  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.sign() < 0) {
199  continue;
200  }
201 
202  float triggerWeight = 1.0;
203  if (cfg.mEfficiencyTrigger) {
204  triggerWeight = getEfficiency(cfg.mEfficiencyTrigger, track1.eta(), track1.pt(), centrality, posZ);
205  }
206 
207  target->getTriggerHist()->Fill(CorrelationContainer::kCFStepReconstructed, track1.pt(), centrality, posZ, triggerWeight);
208 
209  int i = -1;
210  for (auto& track2 : tracks2) {
211  i++; // HACK
212  if (track1 == track2) {
213  continue;
214  }
215 
216  if (cfgPtOrder != 0 && track2.pt() >= track1.pt()) {
217  continue;
218  }
219 
220  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.sign() < 0) {
221  continue;
222  }
223  if (cfgPairCharge != 0 && cfgPairCharge * track1.sign() * track2.sign() < 0) {
224  continue;
225  }
226 
227  if (cfg.mPairCuts && mPairCuts.conversionCuts(track1, track2)) {
228  continue;
229  }
230 
231  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign)) {
232  continue;
233  }
234 
235  float associatedWeight = 1.0;
236  if (cfg.mEfficiencyAssociated) {
237  associatedWeight = efficiencyAssociated[i];
238  }
239 
240  float deltaPhi = track1.phi() - track2.phi();
241  if (deltaPhi > 1.5 * M_PI) {
242  deltaPhi -= M_PI * 2;
243  }
244  if (deltaPhi < -0.5 * M_PI) {
245  deltaPhi += M_PI * 2;
246  }
247 
248  target->getPairHist()->Fill(CorrelationContainer::kCFStepReconstructed,
249  track1.eta() - track2.eta(), track2.pt(), track1.pt(), centrality, deltaPhi, posZ,
250  triggerWeight * associatedWeight);
251  }
252  }
253 
254  delete[] efficiencyAssociated;
255  }

410  void fillArray(int length, double* source, std::vector<float>& target)

426  void init(o2::framework::InitContext& initContext)
427  {
428  // get own suffix. Is there a better way?
429  auto& deviceSpec = initContext.services().get<DeviceSpec const>();
430  std::string suffix(deviceSpec.name);
431  suffix.replace(0, strlen("correlation-hash-task"), "");
432 
433  // get axis config from CorrelationTask
434  auto& workflows = initContext.services().get<RunningWorkflowInfo const>();
435  for (DeviceSpec device : workflows.devices) {
436  if (device.name == "correlation-task" + suffix) {
437  for (auto option : device.options) {
438  if (option.name == "axisVertex") {
439  fillArray(option.defaultValue.size(), option.defaultValue.get<double*>(), vtxBins);
440  LOGF(info, "Initialized vertex binning for mixing from configurable %s", option.name);
441  }
442  if (option.name == "axisMultiplicity") {
443  fillArray(option.defaultValue.size(), option.defaultValue.get<double*>(), multBins);
444  LOGF(info, "Initialized multiplicity binning for mixing from configurable %s", option.name);
445  }
446  }
447  }
448  }
449 
450  if (vtxBins.size() == 0) {
451  LOGF(fatal, "vtxBins not configured. Check configuration.");
452  }
453  if (multBins.size() == 0) {
454  LOGF(fatal, "multBins not configured. Check configuration.");
455  }
456  }
457 
458  // Calculate hash for an element based on 2 properties and their bins.
459  int getHash(float vtx, float mult)
460  {
461  // underflow
462  if (vtx < vtxBins[0]) {
463  return -1;
464  }
465  if (mult < multBins[0]) {
466  return -1;
467  }
468 
469  for (int i = 1; i < vtxBins.size(); i++) {
470  if (vtx < vtxBins[i]) {
471  for (int j = 1; j < multBins.size(); j++) {
472  if (mult < multBins[j]) {
473  return i + j * (vtxBins.size() + 1);
474  }
475  }
476  }
477  }
478  // overflow
479  return -1;
480  }

482  void process(soa::Join<aod::Collisions, aod::Cents> const& collisions)
483  {
484  for (auto& collision : collisions) {
485  int hash = getHash(collision.posZ(), collision.centV0M());
486  LOGF(info, "Collision: %d (%f, %f) hash: %d", collision.index(), collision.posZ(), collision.centV0M(), hash);
487  hashes(hash);
488  }
