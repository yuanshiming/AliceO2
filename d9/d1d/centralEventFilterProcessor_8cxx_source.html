104 void CentralEventFilterProcessor::run(ProcessingContext& pc)
105 {
106  int64_t nEvents{-1};
107  for (auto& tableName : mDownscaling) {
108  auto tableConsumer = pc.inputs().get<TableConsumer>(tableName.first);
109 
110  auto tablePtr{tableConsumer->asArrowTable()};
111  int64_t nRows{tablePtr->num_rows()};
112  nEvents = nEvents < 0 ? nRows : nEvents;
113  if (nEvents != nRows) {
114  std::cerr << "Inconsistent number of rows across trigger tables, fatal" << std::endl;
115  }
116 
117  auto schema{tablePtr->schema()};
118  for (auto& colName : tableName.second) {
119  int bin{mScalers->GetXaxis()->FindBin(colName.first.data())};
120  double binCenter{mScalers->GetXaxis()->GetBinCenter(bin)};
121  auto column{tablePtr->GetColumnByName(colName.first)};
122  double downscaling{colName.second};
123  if (column) {
124  for (int64_t iC{0}; iC < column->num_chunks(); ++iC) {
125  auto chunk{column->chunk(iC)};
126  auto boolArray = std::static_pointer_cast<arrow::BooleanArray>(chunk);
127  for (int64_t iS{0}; iS < chunk->length(); ++iS) {
128  if (boolArray->Value(iS)) {
129  mScalers->Fill(binCenter);
130  if (mUniformGenerator(mGeneratorEngine) < downscaling) {
131  mFiltered->Fill(binCenter);
132  }
133  }
134  }
135  }
136  }
137  }
138  }
139  mScalers->SetBinContent(1, mScalers->GetBinContent(1) + nEvents);
140  mFiltered->SetBinContent(1, mFiltered->GetBinContent(1) + nEvents);
141 }

144 {

159 DataProcessorSpec getCentralEventFilterProcessorSpec(std::string& config)
160 {
161 
162  std::vector<InputSpec> inputs;
163  Document d = readJsonFile(config);
164 
165  for (auto& workflow : d["workflows"].GetArray()) {
166  for (unsigned int iFilter{0}; iFilter < AvailableFilters.size(); ++iFilter) {
167  if (std::string_view(workflow["subwagon_name"].GetString()) == std::string_view(AvailableFilters[iFilter])) {
168  inputs.emplace_back(std::string(AvailableFilters[iFilter]), "AOD", FilterDescriptions[iFilter], 0, Lifetime::Timeframe);
169  std::cout << "Adding input " << std::string_view(AvailableFilters[iFilter]) << std::endl;
170  break;
171  }
172  }
173  }
174 
175  std::vector<OutputSpec> outputs;
176  outputs.emplace_back("AOD", "Decision", 0, Lifetime::Timeframe);
177 
178  return DataProcessorSpec{
179  "o2-central-event-filter-processor",
180  inputs,
181  outputs,
182  AlgorithmSpec{adaptFromTask<CentralEventFilterProcessor>(config)},
183  Options{
184  {"filtering-config", VariantType::String, "", {"Path to the filtering json config file"}}}};
185 }
