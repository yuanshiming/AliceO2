40 bool ComputingQuotaEvaluator::selectOffer(int task, ComputingQuotaRequest const& selector)
41 {
42  auto selectOffer = [&loop = mLoop, &offers = this->mOffers, &infos = this->mInfos, task](int ref) {
43  auto& selected = offers[ref];
44  auto& info = infos[ref];
45  selected.user = task;
46  if (info.firstUsed == 0) {
47  info.firstUsed = uv_now(loop);
48  }
49  info.lastUsed = uv_now(loop);
50  };
51 
52  ComputingQuotaOffer accumulated;
53 
54  for (int i = 0; i != mOffers.size(); ++i) {
55  auto& offer = mOffers[i];
56  auto& info = mInfos[i];
57 
58  // Ignore:
59  // - Invalid offers
60  // - Offers which belong to another task
61  // - Expired offers
62  if (offer.valid == false) {
63  continue;
64  }
65  if (offer.user != -1 && offer.user != task) {
66  continue;
67  }
68  if (offer.runtime > 0 && offer.runtime + info.received < uv_now(mLoop)) {
69  continue;
70  }
72  assert(offer.sharedMemory >= 0);
73  auto tmp = accumulated;
74  tmp.cpu += offer.cpu;
75  tmp.memory += offer.memory;
76  tmp.sharedMemory += offer.sharedMemory;
77  offer.score = selector(offer, tmp);
78  switch (offer.score) {
79  case OfferScore::Unneeded:
80  case OfferScore::Unsuitable:
81  continue;
82  case OfferScore::More:
83  selectOffer(i);
84  break;
85  case OfferScore::Enough:
86  selectOffer(i);
87  return true;
88  };
89  }
90  // If we get here it means we never got enough offers, so we return false.
91  return false;
92 }

99 void ComputingQuotaEvaluator::dispose(int taskId)
100 {
101  for (int oi = 0; oi < mOffers.size(); ++oi) {
102  auto& offer = mOffers[oi];
103  if (offer.user != taskId) {
104  continue;
105  }
106  offer.user = -1;
107  // Disposing the offer so that the resource can be recyled.
110  if (oi == 0) {
111  return;
112  }
113  if (offer.valid == false) {
114  continue;
115  }
116  if (offer.sharedMemory <= 0) {
117  offer.valid = false;
118  offer.score = OfferScore::Unneeded;
119  }
120  }
121 }
