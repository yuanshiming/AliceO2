144  void process(MyEvent const& collision, soa::Filtered<MyMuons> const& tracksMuon, aod::BCs const& bcs, soa::Filtered<MyBarrelTracks> const& tracksBarrel)
145  {
146  uint32_t triggerAliases = 0;
147  for (int i = 0; i < kNaliases; i++) {
148  if (collision.alias()[i] > 0) {
149  triggerAliases |= (uint32_t(1) << i);
150  }
151  }
152  uint64_t tag = 0; // TODO: add here available computed event cuts (e.g. run2bcinfo().eventCuts()) or other event wise decisions
153 
154  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
155  VarManager::FillEvent<eventFillMap>(collision, fValues); // extract event information and place it in the fValues array
156  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event
157 
158  if (!fEventCut->IsSelected(fValues)) {
159  return;
160  }
161 
162  fHistMan->FillHistClass("Event_AfterCuts", fValues);
163 
164  event(tag, collision.bc().runNumber(), collision.posX(), collision.posY(), collision.posZ(), collision.numContrib());
165  eventExtended(collision.bc().globalBC(), collision.bc().triggerMask(), 0, triggerAliases, fValues[VarManager::kCentVZERO]);
166  eventVtxCov(collision.covXX(), collision.covXY(), collision.covXZ(), collision.covYY(), collision.covYZ(), collision.covZZ(), collision.chi2());
167 
168  uint64_t trackFilteringTag = 0;
169  trackBasic.reserve(tracksBarrel.size());
170  trackBarrel.reserve(tracksBarrel.size());
171  trackBarrelCov.reserve(tracksBarrel.size());
172  trackBarrelPID.reserve(tracksBarrel.size());
173 
174  for (auto& track : tracksBarrel) {
175  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
176  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
177  if (!fTrackCut->IsSelected(fValues)) {
178  continue;
179  }
180  fHistMan->FillHistClass("TrackBarrel_AfterCuts", fValues);
181 
182  etaH->Fill(track.eta());
183  if (track.isGlobalTrack()) {
184  trackFilteringTag |= (uint64_t(1) << 0);
185  }
186  if (track.isGlobalTrackSDD()) {
187  trackFilteringTag |= (uint64_t(1) << 1);
188  }
189  trackBasic(event.lastIndex(), track.globalIndex(), trackFilteringTag, track.pt(), track.eta(), track.phi(), track.sign());
190  trackBarrel(track.tpcInnerParam(), track.flags(), track.itsClusterMap(), track.itsChi2NCl(),
191  track.tpcNClsFindable(), track.tpcNClsFindableMinusFound(), track.tpcNClsFindableMinusCrossedRows(),
192  track.tpcNClsShared(), track.tpcChi2NCl(),
193  track.trdChi2(), track.trdPattern(), track.tofChi2(),
194  track.length(), track.dcaXY(), track.dcaZ());
195  trackBarrelCov(track.x(), track.alpha(), track.y(), track.z(), track.snp(), track.tgl(), track.signed1Pt(),
196  track.cYY(), track.cZY(), track.cZZ(), track.cSnpY(), track.cSnpZ(),
197  track.cSnpSnp(), track.cTglY(), track.cTglZ(), track.cTglSnp(), track.cTglTgl(),
198  track.c1PtY(), track.c1PtZ(), track.c1PtSnp(), track.c1PtTgl(), track.c1Pt21Pt2());
199  trackBarrelPID(track.tpcSignal(),
200  track.tpcNSigmaEl(), track.tpcNSigmaMu(),
201  track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(),
202  track.beta(),
203  track.tofNSigmaEl(), track.tofNSigmaMu(),
204  track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(),
205  track.trdSignal());
206  }
207 
208  // build the muon tables
209  muonBasic.reserve(tracksMuon.size());
210  muonExtra.reserve(tracksMuon.size());
211  muonCov.reserve(tracksMuon.size());
212  for (auto& muon : tracksMuon) {
213  VarManager::FillTrack<gkMuonFillMap>(muon, fValues);
214  fHistMan->FillHistClass("Muons_BeforeCuts", fValues);
215  if (!fMuonCut->IsSelected(fValues)) {
216  continue;
217  }
218  fHistMan->FillHistClass("Muons_AfterCuts", fValues);
219 
220  muonBasic(event.lastIndex(), trackFilteringTag, muon.pt(), muon.eta(), muon.phi(), muon.sign());
221  muonExtra(muon.nClusters(), muon.pDca(), muon.rAtAbsorberEnd(),
222  muon.chi2(), muon.chi2MatchMCHMID(), muon.chi2MatchMCHMFT(),
223  muon.matchScoreMCHMFT(), muon.matchMFTTrackID(), muon.matchMCHTrackID());
224  muonCov(muon.x(), muon.y(), muon.z(), muon.phi(), muon.tgl(), muon.signed1Pt(),
225  muon.cXX(), muon.cXY(), muon.cYY(), muon.cPhiX(), muon.cPhiY(), muon.cPhiPhi(),
226  muon.cTglX(), muon.cTglY(), muon.cTglPhi(), muon.cTglTgl(), muon.c1PtX(), muon.c1PtY(),
227  muon.c1PtPhi(), muon.c1PtTgl(), muon.c1Pt21Pt2());
228  }
229  }
