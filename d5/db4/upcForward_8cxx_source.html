56  void process(soa::Join<aod::BCs, aod::BcSels>::iterator const& bc, aod::Muons const& tracksMuon)
57  {
58  registry.fill(HIST("hSelectionCounter"), 0);
59 
60  int iMuonTracknumber = 0;
61  TLorentzVector p1, p2, p;
62  bool ispositive = kFALSE;
63  bool isnegative = kFALSE;
64 
65  // V0 and FD information
66  bool isBeamBeamV0A = bc.bbV0A();
67  bool isBeamGasV0A = bc.bgV0A();
68  bool isBeamBeamV0C = bc.bbV0C();
69  bool isBeamGasV0C = bc.bgV0C();
70 
71  bool isBeamBeamFDA = bc.bbFDA();
72  bool isBeamGasFDA = bc.bgFDA();
73  bool isBeamBeamFDC = bc.bbFDC();
74  bool isBeamGasFDC = bc.bgFDC();
75 
76  //offline V0 and FD selection
77  bool isV0Selection = isBeamBeamV0A || isBeamGasV0A || isBeamGasV0C;
78  bool isFDSelection = isBeamBeamFDA || isBeamGasFDA || isBeamBeamFDC || isBeamGasFDC;
79 
80  //CCUP10 and CCUP11 information
81  bool iskMUP11fired = bc.alias()[kMUP11];
82  bool iskMUP10fired = bc.alias()[kMUP10];
83 
84  // selecting kMUP10 and 11 triggers
85  if (!iskMUP11fired && !iskMUP10fired) {
86  return;
87  }
88  registry.fill(HIST("hSelectionCounter"), 1);
89 
90  if (isV0Selection) {
91  return;
92  }
93  registry.fill(HIST("hSelectionCounter"), 2);
94 
95  if (isFDSelection) {
96  return;
97  }
98  registry.fill(HIST("hSelectionCounter"), 3);
99 
100  for (auto& muon : tracksMuon) {
101  registry.fill(HIST("hCharge"), muon.sign());
102  iMuonTracknumber++;
103  if (muon.sign() > 0) {
104  p1.SetXYZM(muon.px(), muon.py(), muon.pz(), mmuon);
105  ispositive = kTRUE;
106  }
107  if (muon.sign() < 0) {
108  p2.SetXYZM(muon.px(), muon.py(), muon.pz(), mmuon);
109  isnegative = kTRUE;
110  }
111  }
112  if (iMuonTracknumber != 2) {
113  return;
114  }
115  registry.fill(HIST("hSelectionCounter"), 4);
116  if (!ispositive || !isnegative) {
117  return;
118  }
119  registry.fill(HIST("hSelectionCounter"), 5);
120 
121  if (-4 < p1.Eta() < -2.5 || -4 < p2.Eta() < -2.5) {
122 
123  return;
124  }
125  registry.fill(HIST("hSelectionCounter"), 6);
126  p = p1 + p2;
127  if (p.Pt() > 1) {
128  return;
129  }
130  registry.fill(HIST("hSelectionCounter"), 7);
131  registry.fill(HIST("hPt"), p.Pt());
132  registry.fill(HIST("hPx"), p.Px());
133  registry.fill(HIST("hPy"), p.Py());
134  registry.fill(HIST("hPz"), p.Pz());
135  registry.fill(HIST("hRap"), p.Rapidity());
136  registry.fill(HIST("hMass"), p.M());
137  registry.fill(HIST("hPhi"), p.Phi());
138  registry.fill(HIST("hEta"), p1.Eta());
139  registry.fill(HIST("hEta"), p2.Eta());
140  registry.fill(HIST("hPtsingle_muons"), p1.Pt());
141  registry.fill(HIST("hPtsingle_muons"), p2.Pt());
142 
143  } //end of process
144 };
