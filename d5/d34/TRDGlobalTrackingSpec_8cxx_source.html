52 void TRDGlobalTracking::init(InitContext& ic)
53 {
54 
55  //-------- init geometry and field --------//
56  o2::base::GeometryManager::loadGeometry();
57  o2::base::Propagator::initFieldFromGRP();
58  auto geo = Geometry::instance();
59  geo->createPadPlaneArray();
60  geo->createClusterMatrixArray();
61  mFlatGeo = std::make_unique<GeometryFlat>(*geo);
62 
63  //-------- init GPU reconstruction --------//
64  GPURecoStepConfiguration cfgRecoStep;
65  cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
66  cfgRecoStep.inputs.clear();
67  cfgRecoStep.outputs.clear();
68  mRec = GPUReconstruction::CreateInstance("CPU", true);
69  mRec->SetSettings(o2::base::Propagator::Instance()->getNominalBz(), &cfgRecoStep);
70 
71  mChainTracking = mRec->AddChain<GPUChainTracking>();
72 
73  mTracker = new GPUTRDTracker();
74  mTracker->SetNCandidates(mRec->GetProcessingSettings().trdNCandidates); // must be set before initialization
75  mTracker->SetProcessPerTimeFrame(true);
76  mTracker->SetGenerateSpacePoints(false); // set to true to force space point calculation by the TRD tracker itself
77  //mTracker->SetDoImpactAngleHistograms(true);
78 
79  mRec->RegisterGPUProcessor(mTracker, false);
80  mChainTracking->SetTRDGeometry(std::move(mFlatGeo));
81  if (mRec->Init()) {
82  LOG(FATAL) << "GPUReconstruction could not be initialized";
83  }
84 
85  mTracker->PrintSettings();
86 
87  mTimer.Stop();
88  mTimer.Reset();
89 }

102 void TRDGlobalTracking::fillTrackTriggerRecord(const std::vector<TrackTRD>& tracks, std::vector<TrackTriggerRecord>& trigRec, const gsl::span<const o2::trd::TriggerRecord>& trackletTrigRec) const
103 {
104  int currTrigRec = 0;
105  int nTracksCurr = 0;
106  int iTrackFirst = 0;
107  for (const auto& trk : tracks) {
108  if (trk.getCollisionId() != currTrigRec) {
109  // new collision ID, create new track trigger record
110  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
111  currTrigRec = trk.getCollisionId();
112  iTrackFirst += nTracksCurr;
113  nTracksCurr = 0;
114  }
115  ++nTracksCurr;
116  }
117  if (nTracksCurr > 0) {
118  // create track trigger record for remaining track range
119  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
120  }
121 }

123 void TRDGlobalTracking::run(ProcessingContext& pc)
124 {
125  mTimer.Start(false);
126  mChainTracking->ClearIOPointers();
127  o2::globaltracking::RecoContainer inputTracks;
128  inputTracks.collectData(pc, *mDataRequest);
129  auto tmpInputContainer = getRecoInputContainer(pc, &mChainTracking->mIOPtrs, &inputTracks);
130  auto tmpContainer = GPUWorkflowHelper::fillIOPtr(mChainTracking->mIOPtrs, inputTracks, mUseMC, nullptr, GTrackID::getSourcesMask("TRD"), mTrkMask, GTrackID::mask_t{GTrackID::MASK_NONE});
131 
132  LOGF(INFO, "There are %i tracklets in total from %i trigger records", mChainTracking->mIOPtrs.nTRDTracklets, mChainTracking->mIOPtrs.nTRDTriggerRecords);
133  LOGF(INFO, "As input seeds are available: %i ITS-TPC matched tracks and %i TPC tracks", mChainTracking->mIOPtrs.nTracksTPCITSO2, mChainTracking->mIOPtrs.nOutputTracksTPCO2);
134 
135  mTracker->Reset();
136  mTracker->ResetImpactAngleHistograms();
137  updateTimeDependentParams();
138  mRec->PrepareEvent();
139  mRec->SetupGPUProcessor(mTracker, true);
140 
141  // load input tracks
142  LOG(DEBUG) << "Start loading input seeds into TRD tracker";
143  int nTracksLoadedITSTPC = 0;
144  int nTracksLoadedTPC = 0;
145  // load ITS-TPC matched tracks
146  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nTracksTPCITSO2; ++iTrk) {
147  const auto& trkITSTPC = mChainTracking->mIOPtrs.tracksTPCITSO2[iTrk];
148  GPUTRDTrack trkLoad(trkITSTPC, mTPCVdrift);
149  auto trackGID = GTrackID(iTrk, GTrackID::ITSTPC);
150  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw())) {
151  continue;
152  }
153  ++nTracksLoadedITSTPC;
154  LOGF(DEBUG, "Loaded ITS-TPC track %i with time %f", nTracksLoadedITSTPC, trkLoad.getTime());
155  }
156  // load TPC-only tracks
157  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nOutputTracksTPCO2; ++iTrk) {
158  if (mChainTracking->mIOPtrs.tpcLinkITS && mChainTracking->mIOPtrs.tpcLinkITS[iTrk] != -1) {
159  // this TPC tracks has already been matched to ITS and the ITS-TPC track has already been loaded in the tracker
160  continue;
161  }
162  const auto& trkTpc = mChainTracking->mIOPtrs.outputTracksTPCO2[iTrk];
163  GPUTRDTrack trkLoad(trkTpc, mTPCTBinMUS, mTPCVdrift, iTrk);
164  auto trackGID = GTrackID(iTrk, GTrackID::TPC);
165  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw())) {
166  continue;
167  }
168  ++nTracksLoadedTPC;
169  LOGF(DEBUG, "Loaded TPC track %i with time %f", nTracksLoadedTPC, trkLoad.getTime());
170  }
171  LOGF(INFO, "%i tracks are loaded into the TRD tracker. Out of those %i ITS-TPC tracks and %i TPC tracks", nTracksLoadedITSTPC + nTracksLoadedTPC, nTracksLoadedITSTPC, nTracksLoadedTPC);
172 
173  // start the tracking
174  //mTracker->DumpTracks();
175  mTracker->DoTracking(mChainTracking);
176  //mTracker->DumpTracks();
177 
178  // finished tracking, now collect the output
179  std::vector<TrackTRD> tracksOutITSTPC;
180  std::vector<TrackTRD> tracksOutTPC;
181  std::vector<TrackTriggerRecord> trackTrigRecITSTPC;
182  std::vector<TrackTriggerRecord> trackTrigRecTPC;
183  GPUTRDTrack* tracksOutRaw = mTracker->Tracks();
184  std::vector<unsigned int> trackIdxArray(mTracker->NTracks()); // track indices sorted by trigger record index
185  std::iota(trackIdxArray.begin(), trackIdxArray.end(), 0);
186  std::sort(trackIdxArray.begin(), trackIdxArray.end(), [tracksOutRaw](int lhs, int rhs) { return tracksOutRaw[lhs].getCollisionId() < tracksOutRaw[rhs].getCollisionId(); });
187 
188  int nTrackletsAttached = 0; // only used for debug information
189  for (int iTrk = 0; iTrk < mTracker->NTracks(); ++iTrk) {
190  const auto& trdTrack = mTracker->Tracks()[trackIdxArray[iTrk]];
191  if (trdTrack.getCollisionId() < 0) {
192  // skip tracks without TRD tracklets (the collision ID for the TRD tracks is initialized to -1 and only changed if a tracklet is attached to the track)
193  continue;
194  }
195  nTrackletsAttached += trdTrack.getNtracklets();
196  auto trackGID = trdTrack.getRefGlobalTrackId();
197  if (trackGID.includesDet(GTrackID::Source::ITS)) {
198  // this track is from an ITS-TPC seed
199  tracksOutITSTPC.push_back(trdTrack);
200  } else {
201  // this track is from a TPC-only seed
202  tracksOutTPC.push_back(trdTrack);
203  }
204  }
205 
206  fillTrackTriggerRecord(tracksOutITSTPC, trackTrigRecITSTPC, tmpInputContainer->mTriggerRecords);
207  fillTrackTriggerRecord(tracksOutTPC, trackTrigRecTPC, tmpInputContainer->mTriggerRecords);
208 
209  LOGF(INFO, "The TRD tracker found %lu tracks from TPC seeds and %lu tracks from ITS-TPC seeds and attached in total %i tracklets out of %i",
210  tracksOutTPC.size(), tracksOutITSTPC.size(), nTrackletsAttached, mChainTracking->mIOPtrs.nTRDTracklets);
211 
212  // Temporary until it is transferred to its own DPL device for calibrations
213  mCalibVDrift.setAngleDiffSums(mTracker->AngleDiffSums());
214  mCalibVDrift.setAngleDiffCounters(mTracker->AngleDiffCounters());
215  mCalibVDrift.process();
216 
217  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::ITSTPC)) {
218  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCHTRD_GLO", 0, Lifetime::Timeframe}, tracksOutITSTPC);
219  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRKTRG_GLO", 0, Lifetime::Timeframe}, trackTrigRecITSTPC);
220  }
221  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::TPC)) {
222  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCHTRD_TPC", 0, Lifetime::Timeframe}, tracksOutTPC);
223  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRKTRG_TPC", 0, Lifetime::Timeframe}, trackTrigRecTPC);
224  }
225 
226  mTimer.Stop();
227 }

235 DataProcessorSpec getTRDGlobalTrackingSpec(bool useMC, GTrackID::mask_t src)
236 {
237  std::vector<OutputSpec> outputs;
238 
239  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
240  dataRequest->requestTracks(src, useMC);
241  dataRequest->requestClusters(GTrackID::getSourcesMask("TRD"), useMC);
242  auto& inputs = dataRequest->inputs;
243 
244 
245  if (useMC) {
246  LOG(FATAL) << "MC usage must be disabled for this workflow, since it is not yet implemented";
247  }
248 
249  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, src)) {
250  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCHTRD_GLO", 0, Lifetime::Timeframe);
251  outputs.emplace_back(o2::header::gDataOriginTRD, "TRKTRG_GLO", 0, Lifetime::Timeframe);
252  }
253  if (GTrackID::includesSource(GTrackID::Source::TPC, src)) {
254  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCHTRD_TPC", 0, Lifetime::Timeframe);
255  outputs.emplace_back(o2::header::gDataOriginTRD, "TRKTRG_TPC", 0, Lifetime::Timeframe);
256  }
257 
258  std::string processorName = o2::utils::Str::concat_string("trd-globaltracking", GTrackID::getSourcesNames(src));
259  std::replace(processorName.begin(), processorName.end(), ',', '_');
260 
261  return DataProcessorSpec{
262  processorName,
263  inputs,
264  outputs,
265  AlgorithmSpec{adaptFromTask<TRDGlobalTracking>(useMC, dataRequest, src)},
266  Options{}};
267 }
