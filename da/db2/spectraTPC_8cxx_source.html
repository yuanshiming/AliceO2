39 template <typename T>
40 void makelogaxis(T h)
41 {
42  const int nbins = h->GetNbinsX();
43  double binp[nbins + 1];
44  double max = h->GetXaxis()->GetBinUpEdge(nbins);
45  double min = h->GetXaxis()->GetBinLowEdge(1);
46  if (min <= 0) {
47  min = 0.00001;
48  }
49  double lmin = TMath::Log10(min);
50  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
51  for (int i = 0; i < nbins; i++) {
52  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
53  }
54  binp[nbins] = max + 1;
55  h->GetXaxis()->Set(nbins, binp);
56 }

59 struct tpcSpectra {

63  static constexpr std::string_view hpt[Np] = {"pt/El", "pt/Mu", "pt/Pi", "pt/Ka", "pt/Pr", "pt/De", "pt/Tr", "pt/He", "pt/Al"};
64  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};

66  void init(o2::framework::InitContext&)
67  {
68  histos.add("p/Unselected", "Unselected;#it{p} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
69  histos.add("pt/Unselected", "Unselected;#it{p}_{T} (GeV/#it{c})", kTH1F, {{100, 0, 20}});
70  for (int i = 0; i < Np; i++) {
71  histos.add(hp[i].data(), Form("%s;#it{p} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});
72  histos.add(hpt[i].data(), Form("%s;#it{p}_{T} (GeV/#it{c})", pT[i]), kTH1F, {{100, 0, 20}});

76  template <std::size_t i, typename T>
77  void fillParticleHistos(const T& track, const float& nsigma)
78  {
79  if (abs(nsigma) > cfgNSigmaCut) {
80  return;
81  }
82  histos.fill(HIST(hp[i]), track.p());
83  histos.fill(HIST(hpt[i]), track.pt());
84  }
85 
86  //Defining filters and input
87  Configurable<float> cfgNSigmaCut{"cfgNSigmaCut", 3, "Value of the Nsigma cut"};
88  Configurable<float> cfgCutVertex{"cfgCutVertex", 10.0f, "Accepted z-vertex range"};
89  Configurable<float> cfgCutEta{"cfgCutEta", 0.8f, "Eta range for tracks"};
90  Filter collisionFilter = nabs(aod::collision::posZ) < cfgCutVertex;
91  Filter trackFilter = (nabs(aod::track::eta) < cfgCutEta) && (aod::track::isGlobalTrack == (uint8_t) true);
92  using TrackCandidates = soa::Filtered<soa::Join<aod::Tracks, aod::TracksExtra,
93  aod::pidTPCFullEl, aod::pidTPCFullMu, aod::pidTPCFullPi,
94  aod::pidTPCFullKa, aod::pidTPCFullPr, aod::pidTPCFullDe,
95  aod::pidTPCFullTr, aod::pidTPCFullHe, aod::pidTPCFullAl,
96  aod::TrackSelection>>;
97 
98  void process(TrackCandidates::iterator const& track)
99  {
100  histos.fill(HIST("p/Unselected"), track.p());
101  histos.fill(HIST("pt/Unselected"), track.pt());
102 
103  fillParticleHistos<0>(track, track.tpcNSigmaEl());
104  fillParticleHistos<1>(track, track.tpcNSigmaMu());
105  fillParticleHistos<2>(track, track.tpcNSigmaPi());
106  fillParticleHistos<3>(track, track.tpcNSigmaKa());
107  fillParticleHistos<4>(track, track.tpcNSigmaPr());
108  fillParticleHistos<5>(track, track.tpcNSigmaDe());
109  fillParticleHistos<6>(track, track.tpcNSigmaTr());
110  fillParticleHistos<7>(track, track.tpcNSigmaHe());
111  fillParticleHistos<8>(track, track.tpcNSigmaAl());
112 
113  } // end of the process function
114 };
