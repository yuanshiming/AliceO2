50 void AODProducerWorkflowDPL::findMinMaxBc(gsl::span<const o2::ft0::RecPoints>& ft0RecPoints, gsl::span<const PVertex>& primVertices, const std::vector<o2::InteractionTimeRecord>& mcRecords)
51 {
52  for (auto& ft0RecPoint : ft0RecPoints) {
53  uint64_t bc = ft0RecPoint.getInteractionRecord().orbit * o2::constants::lhc::LHCMaxBunches + ft0RecPoint.getInteractionRecord().bc;
54  if (minGlBC > bc) {
55  minGlBC = bc;
56  }
57  if (maxGlBC < bc) {
58  maxGlBC = bc;
59  }
60  }
61 
62  for (auto& vertex : primVertices) {
63  const InteractionRecord& colIRMin = vertex.getIRMin();
64  const InteractionRecord& colIRMax = vertex.getIRMax();
65  uint64_t colBCMin = colIRMin.orbit * o2::constants::lhc::LHCMaxBunches + colIRMin.bc;
66  uint64_t colBCMax = colIRMax.orbit * o2::constants::lhc::LHCMaxBunches + colIRMax.bc;
67  if (minGlBC > colBCMin) {
68  minGlBC = colBCMin;
69  }
70  if (maxGlBC < colBCMax) {
71  maxGlBC = colBCMax;
72  }
73  }
74 
75  for (auto& rec : mcRecords) {
76  uint64_t bc = rec.bc + rec.orbit * o2::constants::lhc::LHCMaxBunches;
77  if (minGlBC > bc) {
78  minGlBC = bc;
79  }
80  if (maxGlBC < bc) {
81  maxGlBC = bc;
82  }
83  }
84 }

86 uint64_t AODProducerWorkflowDPL::getTFNumber(uint64_t firstVtxGlBC, int runNumber)
87 {
88  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
89  o2::ccdb::CcdbApi ccdb_api;
90  const std::string rct_path = "RCT/RunInformation/";
91  const std::string start_orbit_path = "Trigger/StartOrbit";
92  const std::string url = "http://ccdb-test.cern.ch:8080";
93 
94  mgr.setURL(url);
95  ccdb_api.init(url);
96 
97  std::map<int, int>* mapStartOrbit = mgr.get<std::map<int, int>>(start_orbit_path);
98  int64_t ts = 0;
99  std::map<std::string, std::string> metadata;
100  std::map<std::string, std::string> headers;
101  const std::string run_path = Form("%s/%i", rct_path.data(), runNumber);
102  headers = ccdb_api.retrieveHeaders(run_path, metadata, -1);
103  ts = atol(headers["SOR"].c_str());
104 
105  // ccdb returns timestamp in mus
106  // mus to ms
107  ts = ts / 1000;
108 
109  // firstRec --> calculated using `minimal` global BC in the simulation (see AODProducerWorkflowDPL::findMinMaxBc)
110  // firstVtxGlBC --> calculated using global BC corresponding to the first prim. vertex
111 
112  uint32_t initialOrbit = mapStartOrbit->at(runNumber);
113  uint16_t firstRecBC = minGlBC % o2::constants::lhc::LHCMaxBunches;
114  uint32_t firstRecOrbit = minGlBC / o2::constants::lhc::LHCMaxBunches;
115  uint16_t firstVtxBC = firstVtxGlBC % o2::constants::lhc::LHCMaxBunches;
116  uint32_t firstVtxOrbit = firstVtxGlBC / o2::constants::lhc::LHCMaxBunches;
117  const o2::InteractionRecord firstRec(firstRecBC, firstRecOrbit);
118  const o2::InteractionRecord firstVtx(firstVtxBC, firstVtxOrbit);
119  ts += (firstVtx - firstRec).bc2ns() / 1000000;
120 
121  return ts;
122 };

169  // TODO:

191  (Char_t)(128. * track.getSigmaTglZ() / track.getSigmaTgl2() / track.getSigmaZ2()),
192  (Char_t)(128. * track.getSigmaTglSnp() / track.getSigmaTgl2() / track.getSigmaSnp2()),
193  (Char_t)(128. * track.getSigma1PtY() / track.getSigma1Pt2() / track.getSigmaY2()),

196  (Char_t)(128. * track.getSigma1PtTgl() / track.getSigma1Pt2() / track.getSigmaTgl2()));
197  tracksExtraCursor(0,

199  flags,
200  itsClusterMap,
201  tpcNClsFindable,
202  tpcNClsFindableMinusFound,
203  tpcNClsFindableMinusCrossedRows,
204  tpcNClsShared,
205  trdPattern,
206  truncateFloatFraction(itsChi2NCl, mTrackCovOffDiag),
207  truncateFloatFraction(tpcChi2NCl, mTrackCovOffDiag),
208  truncateFloatFraction(trdChi2, mTrackCovOffDiag),
209  truncateFloatFraction(tofChi2, mTrackCovOffDiag),
210  truncateFloatFraction(tpcSignal, mTrackSignal),
211  truncateFloatFraction(trdSignal, mTrackSignal),
212  truncateFloatFraction(tofSignal, mTrackSignal),

240 template <typename MCParticlesCursorType>
241 void AODProducerWorkflowDPL::fillMCParticlesTable(o2::steer::MCKinematicsReader& mcReader, const MCParticlesCursorType& mcParticlesCursor,
242  gsl::span<const o2::MCCompLabel>& mcTruthITS, gsl::span<const o2::MCCompLabel>& mcTruthMFT,
243  gsl::span<const o2::MCCompLabel>& mcTruthTPC, TripletsMap_t& toStore)
244 {
245  // mark reconstructed MC particles to store them into the table
246  for (auto& mcTruth : mcTruthITS) {
247  if (!mcTruth.isValid()) {
248  continue;
249  }
250  int source = mcTruth.getSourceID();
251  int event = mcTruth.getEventID();
252  int particle = mcTruth.getTrackID();
253  toStore[Triplet_t(source, event, particle)] = 1;
254  }
255  for (auto& mcTruth : mcTruthMFT) {
256  if (!mcTruth.isValid()) {
257  continue;
258  }
259  int source = mcTruth.getSourceID();
260  int event = mcTruth.getEventID();
261  int particle = mcTruth.getTrackID();
262  toStore[Triplet_t(source, event, particle)] = 1;
263  }
264  for (auto& mcTruth : mcTruthTPC) {
265  if (!mcTruth.isValid()) {
266  continue;
267  }
268  int source = mcTruth.getSourceID();
269  int event = mcTruth.getEventID();
270  int particle = mcTruth.getTrackID();
271  toStore[Triplet_t(source, event, particle)] = 1;
272  }
273  int tableIndex = 1;
274  for (int source = 0; source < mcReader.getNSources(); source++) {
275  for (int event = 0; event < mcReader.getNEvents(source); event++) {
276  std::vector<MCTrack> const& mcParticles = mcReader.getTracks(source, event);
277  // mark tracks to be stored per event
278  // loop over stack of MC particles from end to beginning: daughters are stored after mothers
279  if (mRecoOnly) {
280  for (int particle = mcParticles.size() - 1; particle >= 0; particle--) {
281  int mother0 = mcParticles[particle].getMotherTrackId();
282  if (mother0 == -1) {
283  toStore[Triplet_t(source, event, particle)] = 1;
284  }
285  if (toStore.find(Triplet_t(source, event, particle)) == toStore.end()) {
286  continue;
287  }
288  if (mother0 != -1) {
289  toStore[Triplet_t(source, event, mother0)] = 1;
290  }
291  int mother1 = mcParticles[particle].getSecondMotherTrackId();
292  if (mother1 != -1) {
293  toStore[Triplet_t(source, particle, mother1)] = 1;
294  }
295  int daughter0 = mcParticles[particle].getFirstDaughterTrackId();
296  if (daughter0 != -1) {
297  toStore[Triplet_t(source, event, daughter0)] = 1;
298  }
299  int daughterL = mcParticles[particle].getLastDaughterTrackId();
300  if (daughterL != -1) {
301  toStore[Triplet_t(source, event, daughterL)] = 1;
302  }
303  }
304  // enumerate reconstructed mc particles and their relatives to get mother/daughter relations
305  for (int particle = 0; particle < mcParticles.size(); particle++) {
306  auto mapItem = toStore.find(Triplet_t(source, event, particle));
307  if (mapItem != toStore.end()) {
308  mapItem->second = tableIndex;
309  tableIndex++;
310  }
311  }
312  }
313  // if all mc particles are stored, all mc particles will be enumerated
314  if (!mRecoOnly) {
315  for (int particle = 0; particle < mcParticles.size(); particle++) {
316  toStore[Triplet_t(source, event, particle)] = tableIndex;
317  tableIndex++;
318  }
319  }
320  // fill survived mc tracks into the table
321  for (int particle = 0; particle < mcParticles.size(); particle++) {
322  if (toStore.find(Triplet_t(source, event, particle)) == toStore.end()) {
323  continue;
324  }
325  int statusCode = 0;
326  uint8_t flags = 0;
327  float weight = 0.f;
328  int mcMother0 = mcParticles[particle].getMotherTrackId();
329  auto item = toStore.find(Triplet_t(source, event, mcMother0));
330  int mother0 = -1;
331  if (item != toStore.end()) {
332  mother0 = item->second;
333  }
334  int mcMother1 = mcParticles[particle].getSecondMotherTrackId();
335  int mother1 = -1;
336  item = toStore.find(Triplet_t(source, event, mcMother1));
337  if (item != toStore.end()) {
338  mother1 = item->second;
339  }
340  int mcDaughter0 = mcParticles[particle].getFirstDaughterTrackId();
341  int daughter0 = -1;
342  item = toStore.find(Triplet_t(source, event, mcDaughter0));
343  if (item != toStore.end()) {
344  daughter0 = item->second;
345  }
346  int mcDaughterL = mcParticles[particle].getLastDaughterTrackId();
347  int daughterL = -1;
348  item = toStore.find(Triplet_t(source, event, mcDaughterL));
349  if (item != toStore.end()) {
350  daughterL = item->second;
351  }
352  mcParticlesCursor(0,
353  event,
354  mcParticles[particle].GetPdgCode(),
355  statusCode,
356  flags,
357  mother0,
358  mother1,
359  daughter0,
360  daughterL,
361  truncateFloatFraction(weight, mMcParticleW),
362  truncateFloatFraction((float)mcParticles[particle].Px(), mMcParticleMom),
363  truncateFloatFraction((float)mcParticles[particle].Py(), mMcParticleMom),
364  truncateFloatFraction((float)mcParticles[particle].Pz(), mMcParticleMom),
365  truncateFloatFraction((float)mcParticles[particle].GetEnergy(), mMcParticleMom),
366  truncateFloatFraction((float)mcParticles[particle].Vx(), mMcParticlePos),
367  truncateFloatFraction((float)mcParticles[particle].Vy(), mMcParticlePos),
368  truncateFloatFraction((float)mcParticles[particle].Vz(), mMcParticlePos),
369  truncateFloatFraction((float)mcParticles[particle].T(), mMcParticlePos));
370  }
371  mcReader.releaseTracksForSourceAndEvent(source, event);
372  }
373  }
374 }
