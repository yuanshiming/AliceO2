74  void init(o2::framework::InitContext& initContext)
75  {
76  // Checking the tables are requested in the workflow and enabling them
77  auto& workflows = initContext.services().get<RunningWorkflowInfo const>();
78  for (DeviceSpec device : workflows.devices) {
79  for (auto input : device.inputs) {
80  auto enableFlag = [&input](const std::string particle, Configurable<int>& flag) {
81  const std::string table = "pidTOFFull" + particle;
82  if (input.matcher.binding == table) {
83  if (flag < 0) {
84  flag.value = 1;
85  LOG(INFO) << "Auto-enabling table: " + table;
86  } else if (flag > 0) {
87  flag.value = 1;
88  LOG(INFO) << "Table enabled: " + table;
89  } else {
90  LOG(INFO) << "Table disabled: " + table;
91  }
92  }
93  };
94  enableFlag("El", pidEl);
95  enableFlag("Mu", pidMu);
96  enableFlag("Pi", pidPi);
97  enableFlag("Ka", pidKa);
98  enableFlag("Pr", pidPr);
99  enableFlag("De", pidDe);
100  enableFlag("Tr", pidTr);
101  enableFlag("He", pidHe);
102  enableFlag("Al", pidAl);
103  }
104  }
105  // Getting the parametrization parameters
106  ccdb->setURL(url.value);
107  ccdb->setTimestamp(timestamp.value);
108  ccdb->setCaching(true);
109  ccdb->setLocalObjectValidityChecking();
110  // Not later than now objects
111  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
112  //
113  const std::vector<float> p = {0.008, 0.008, 0.002, 40.0};
114  response.SetParameters(DetectorResponse::kSigma, p);
115  const std::string fname = paramfile.value;
116  if (!fname.empty()) { // Loading the parametrization from file
117  LOG(INFO) << "Loading exp. sigma parametrization from file" << fname << ", using param: " << sigmaname.value;
118  response.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
119  } else { // Loading it from CCDB
120  std::string path = ccdbPath.value + "/" + sigmaname.value;
121  LOG(INFO) << "Loading exp. sigma parametrization from CCDB, using path: " << path << " for timestamp " << timestamp.value;
122  response.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path, timestamp.value));
123  }
124  }

126  template <o2::track::PID::ID pid>
127  using ResponseImplementation = tof::ExpTimes<Coll::iterator, Trks::iterator, pid>;
128  void process(Coll const& collisions, Trks const& tracks)
129  {
130  constexpr auto responseEl = ResponseImplementation<PID::Electron>();
131  constexpr auto responseMu = ResponseImplementation<PID::Muon>();
132  constexpr auto responsePi = ResponseImplementation<PID::Pion>();
133  constexpr auto responseKa = ResponseImplementation<PID::Kaon>();
134  constexpr auto responsePr = ResponseImplementation<PID::Proton>();
135  constexpr auto responseDe = ResponseImplementation<PID::Deuteron>();
136  constexpr auto responseTr = ResponseImplementation<PID::Triton>();
137  constexpr auto responseHe = ResponseImplementation<PID::Helium3>();
138  constexpr auto responseAl = ResponseImplementation<PID::Alpha>();
139 
140  // Check and fill enabled tables
141  auto makeTable = [&tracks](const Configurable<int>& flag, auto& table, const DetectorResponse& response, const auto& responsePID) {
142  if (flag.value == 1) {
143  // Prepare memory for enabled tables
144  table.reserve(tracks.size());
145  for (auto const& trk : tracks) { // Loop on Tracks
146  table(responsePID.GetExpectedSigma(response, trk.collision(), trk),
147  responsePID.GetSeparation(response, trk.collision(), trk));
148  }
149  }
150  };
151  makeTable(pidEl, tablePIDEl, response, responseEl);
152  makeTable(pidMu, tablePIDMu, response, responseMu);
153  makeTable(pidPi, tablePIDPi, response, responsePi);
154  makeTable(pidKa, tablePIDKa, response, responseKa);
155  makeTable(pidPr, tablePIDPr, response, responsePr);
156  makeTable(pidDe, tablePIDDe, response, responseDe);
157  makeTable(pidTr, tablePIDTr, response, responseTr);
158  makeTable(pidHe, tablePIDHe, response, responseHe);
159  makeTable(pidAl, tablePIDAl, response, responseAl);
160  }

189  template <typename T>
190  void makelogaxis(T h)
191  {
192  if (logAxis == 0) {
193  return;
194  }
195  const int nbins = h->GetNbinsX();
196  double binp[nbins + 1];
197  double max = h->GetXaxis()->GetBinUpEdge(nbins);
198  double min = h->GetXaxis()->GetBinLowEdge(1);
199  if (min <= 0) {
200  min = 0.00001;
201  }
202  double lmin = TMath::Log10(min);
203  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
204  for (int i = 0; i < nbins; i++) {
205  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
206  }
207  binp[nbins] = max + 1;
208  h->GetXaxis()->Set(nbins, binp);
209  }

259  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra, aod::TracksCov,
260  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
261  aod::pidTOFFullKa, aod::pidTOFFullPr, aod::pidTOFFullDe,
262  aod::pidTOFFullTr, aod::pidTOFFullHe, aod::pidTOFFullAl,
263  aod::TrackSelection> const& tracks)
264  {
265  const float collisionTime_ps = collision.collisionTime() * 1000.f;
266  histos.fill(HIST("event/vertexz"), collision.posZ());
267  histos.fill(HIST("event/colltime"), collisionTime_ps);
268 
269  for (auto t : tracks) {
270  //
271  if (t.tofSignal() < 0) { // Skipping tracks without TOF
272  continue;
273  }
274  if (!t.isGlobalTrack()) {
275  continue;
276  }
277 
278  const float tof = t.tofSignal() - collisionTime_ps;
279 
280  //
281  histos.fill(HIST("event/tofsignal"), t.p(), t.tofSignal());
282  histos.fill(HIST("event/eta"), t.eta());
283  histos.fill(HIST("event/length"), t.length());
284  histos.fill(HIST("event/pt"), t.pt());
285  histos.fill(HIST("event/ptreso"), t.p(), t.sigma1Pt() * t.pt() * t.pt());
286  //
287  fillParticleHistos<0>(t, tof, t.tofExpSignalDiffEl(), t.tofNSigmaEl());
288  fillParticleHistos<1>(t, tof, t.tofExpSignalDiffMu(), t.tofNSigmaMu());
289  fillParticleHistos<2>(t, tof, t.tofExpSignalDiffPi(), t.tofNSigmaPi());
290  fillParticleHistos<3>(t, tof, t.tofExpSignalDiffKa(), t.tofNSigmaKa());
291  fillParticleHistos<4>(t, tof, t.tofExpSignalDiffPr(), t.tofNSigmaPr());
292  fillParticleHistos<5>(t, tof, t.tofExpSignalDiffDe(), t.tofNSigmaDe());
293  fillParticleHistos<6>(t, tof, t.tofExpSignalDiffTr(), t.tofNSigmaTr());
294  fillParticleHistos<7>(t, tof, t.tofExpSignalDiffHe(), t.tofNSigmaHe());
295  fillParticleHistos<8>(t, tof, t.tofExpSignalDiffAl(), t.tofNSigmaAl());
296  }
297  }
