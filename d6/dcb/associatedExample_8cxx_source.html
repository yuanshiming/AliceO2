50 struct ProduceEtaPhi {
51  Produces<aod::EtaPhi> etaphi;
52 
53  void process(aod::Tracks const& tracks)
54  {
55  for (auto& track : tracks) {
56  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
57  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
58  etaphi(eta, phi);
59  }
60  }
61 };

101 struct PartitionColExtra {
102  using myCol = soa::Join<aod::Collisions, aod::CollisionsExtra>;
103 
104  void process(myCol const& collisions, aod::Tracks const& tracks)
105  {
106  // create multiplicity partitions of the Collisions table
107  auto multbin0_10 = collisions.select(aod::collision::mult >= 0 && aod::collision::mult < 10);
108  auto multbin10_30 = collisions.select(aod::collision::mult >= 10 && aod::collision::mult < 30);
109  auto multbin30_100 = collisions.select(aod::collision::mult >= 30 && aod::collision::mult < 100);
110 
111  // loop over the collisions in each partition and ...
112  LOGF(INFO, "Bin 0-10");
113  for (auto& col : multbin0_10) {
114  // ... find all the associated tracks
115  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
116  LOGF(INFO, "Collision %d; Ntrk = %d vs %d", col.globalIndex(), col.mult(), groupedTracks.size());
117  if (groupedTracks.size() > 0) {
118  auto track = groupedTracks.begin();
119  LOGF(INFO, "Track 0 belongs to collision %d at Z = %f", track.collisionId(), track.collision_as<myCol>().posZ());
120  }
121  }
122 
123  LOGF(INFO, "Bin 10-30");
124  for (auto& col : multbin10_30) {
125  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
126  LOGF(INFO, "Collision %d; Ntrk = %d vs %d", col.globalIndex(), col.mult(), groupedTracks.size());
127  }
128 
129  LOGF(INFO, "Bin 30-100");
130  for (auto& col : multbin30_100) {
131  auto groupedTracks = tracks.sliceBy(aod::track::collisionId, col.globalIndex());
