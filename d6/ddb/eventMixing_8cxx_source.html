35 struct HashTask {
36  std::vector<float> xBins{-0.060f, -0.620f, -0.064f, 0.066f, 0.068f, 0.070f, 0.072f};
37  std::vector<float> yBins{-0.300f, -0.301f, -0.320f, 0.330f, 0.340f, 0.350f, 0.360f};
38  Produces<aod::Hashes> hashes;
39 
40  // Calculate hash for an element based on 2 properties and their bins.
41  int getHash(std::vector<float> const& xBins, std::vector<float> const& yBins, float colX, float colY)
42  {
43  if (colX < xBins[0] || colY < yBins[0]) {
44  return -1;
45  }
46  for (int i = 1; i < xBins.size(); i++) {
47  if (colX < xBins[i]) {
48  for (int j = 1; j < yBins.size(); j++) {
49  if (colY < yBins[j]) {
50  return i + j * (xBins.size() + 1);
51  }
52  }
53  return -1;
54  }
55  }
56  return -1;
57  }
58 
59  void process(aod::Collisions const& collisions)
60  {
61  for (auto& collision : collisions) {
62  int hash = getHash(xBins, yBins, collision.posX(), collision.posY());
63  LOGF(info, "Collision: %d (%f, %f, %f) hash: %d", collision.index(), collision.posX(), collision.posY(), collision.posZ(), hash);
64  hashes(hash);
65  }
66  }
67 };

77  // grouping of tracks according to collision
78  collisions.bindExternalIndices(&tracks);

82  // Strictly upper categorised collisions
83  for (auto& [c1, c2] : selfCombinations("fBin", 5, -1, join(hashes, collisions), join(hashes, collisions))) {
84  LOGF(info, "Collisions bin: %d pair: %d (%f, %f, %f), %d (%f, %f, %f)", c1.bin(), c1.index(), c1.posX(), c1.posY(), c1.posZ(), c2.index(), c2.posX(), c2.posY(), c2.posZ());
85 
86  auto it1 = slicer.begin();
87  auto it2 = slicer.begin();
88  for (auto& slice : slicer) {
89  if (slice.groupingElement().index() == c1.index()) {
90  it1 = slice;
91  break;
92  }
93  }
94  for (auto& slice : slicer) {
95  if (slice.groupingElement().index() == c2.index()) {
96  it2 = slice;
97  break;
98  }
99  }

101  tracks1.bindExternalIndices(&collisions);
102  auto tracks2 = std::get<soa::Filtered<aod::Tracks>>(it2.associatedTables());
103  tracks2.bindExternalIndices(&collisions);
104 
105  for (auto& [t1, t2] : combinations(CombinationsFullIndexPolicy(tracks1, tracks2))) {
106  LOGF(info, "Mixed event tracks pair: (%d, %d) from events (%d, %d)", t1.index(), t2.index(), c1.index(), c2.index());
107  }
108  }
109  }

124 
125  // Strictly upper categorised collisions

127  //LOGF(info, "Collisions bin: %d pair: %d (%f, %f, %f), %d (%f, %f, %f)", c1.bin(), c1.index(), c1.posX(), c1.posY(), c1.posZ(), c2.index(), c2.posX(), c2.posY(), c2.posZ());
128 
129  auto it1 = slicer.begin();
130  auto it2 = slicer.begin();
131  for (auto& slice : slicer) {
132  if (slice.groupingElement().index() == c1.index()) {
133  it1 = slice;
134  break;
135  }
136  }
137  for (auto& slice : slicer) {
138  if (slice.groupingElement().index() == c2.index()) {
139  it2 = slice;
140  break;
141  }
142  }
143  auto tracks1 = std::get<myTracks>(it1.associatedTables());
144  tracks1.bindExternalIndices(&collisions);
145  auto tracks2 = std::get<myTracks>(it2.associatedTables());
146  tracks2.bindExternalIndices(&collisions);
147 
148  Partition<myTracks> leftPhi1 = aod::track::phiraw < philow;
149  leftPhi1.bindTable(tracks1);
150  Partition<myTracks> leftPhi2 = aod::track::phiraw < philow;
151  leftPhi2.bindTable(tracks2);
152  Partition<myTracks> rightPhi1 = aod::track::phiraw >= phiup;
153  rightPhi1.bindTable(tracks1);
154  Partition<myTracks> rightPhi2 = aod::track::phiraw >= phiup;
155  rightPhi2.bindTable(tracks2);
156 
157  for (auto& [t1, t2] : combinations(CombinationsFullIndexPolicy(leftPhi1, leftPhi2))) {
158  if (t1.phiraw() >= (float)philow || t2.phiraw() >= (float)philow) {
159  LOGF(info, "WRONG Mixed event left tracks pair: (%d, %d) from events (%d, %d), phi: (%.3f. %.3f) < %.3f", t1.index(), t2.index(), c1.index(), c2.index(), t1.phiraw(), t2.phiraw(), philow);
160  }
161  }
162  for (auto& [t1, t2] : combinations(CombinationsFullIndexPolicy(rightPhi1, rightPhi2))) {
163  if (t1.phiraw() < (float)phiup || t2.phiraw() < (float)phiup) {
164  LOGF(info, "WRONG Mixed event right tracks pair: (%d, %d) from events (%d, %d), phi: (%.3f. %.3f) >= %.3f", t1.index(), t2.index(), c1.index(), c2.index(), t1.phiraw(), t2.phiraw(), phiup);
165  }
166  }
167  }
168  }
169 };
