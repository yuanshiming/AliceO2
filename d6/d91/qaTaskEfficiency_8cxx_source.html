43 template <typename T>
44 void makelogaxis(T h)
45 {
46  const int nbins = h->GetNbinsX();
47  double binp[nbins + 1];
48  double max = h->GetXaxis()->GetBinUpEdge(nbins);
49  double min = h->GetXaxis()->GetBinLowEdge(1);
50  if (min <= 0) {
51  min = 0.01;
52  }
53  double lmin = TMath::Log10(min);
54  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
55  for (int i = 0; i < nbins; i++) {
56  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
57  }
58  binp[nbins] = max + 1;
59  h->GetXaxis()->Set(nbins, binp);
60 }

91  void init(InitContext&)
92  {

95  etaMin.value, etaMax.value,

100 
101  const TString tagEta = Form("%s #it{p}_{T} [%.2f,%.2f] #it{#varphi} [%.2f,%.2f] Prim %i",
102  o2::track::pid_constants::sNames[particle],
103  ptMin.value, ptMax.value,
104  phiMin.value, phiMax.value,
105  selPrim.value);
106  const TString xEta = "#it{#eta}";
107  AxisSpec axisEta{etaBins, etaMin, etaMax};
108 
109  const TString tagPhi = Form("%s #it{#eta} [%.2f,%.2f] #it{p}_{T} [%.2f,%.2f] Prim %i",
110  o2::track::pid_constants::sNames[particle],
111  etaMin.value, etaMax.value,
112  ptMin.value, ptMax.value,
113  selPrim.value);
114  const TString xPhi = "#it{#varphi} (rad)";
115  AxisSpec axisPhi{phiBins, phiMin, phiMax};
116 
117  histos.add("pt/num", "Numerator " + tagPt + ";" + xPt, kTH1D, {axisPt});
118  histos.add("pt/den", "Denominator " + tagPt + ";" + xPt, kTH1D, {axisPt});
119  if (logPt) {
120  makelogaxis(histos.get<TH1>(HIST("pt/num")));
121  makelogaxis(histos.get<TH1>(HIST("pt/den")));

123 
124  histos.add("eta/num", "Numerator " + tagEta + ";" + xEta, kTH1D, {axisEta});
125  histos.add("eta/den", "Denominator " + tagEta + ";" + xEta, kTH1D, {axisEta});
126 
127  histos.add("phi/num", "Numerator " + tagPhi + ";" + xPhi, kTH1D, {axisPhi});
128  histos.add("phi/den", "Denominator " + tagPhi + ";" + xPhi, kTH1D, {axisPhi});
129 
130  list.setObject(new TList);
131  if (makeEff) {
132  auto makeEfficiency = [&](TString effname, TString efftitle, auto templateHisto) {
133  TAxis* axis = histos.get<TH1>(templateHisto)->GetXaxis();

135  list->Add(new TEfficiency(effname, efftitle, axis->GetNbins(), axis->GetXbins()->GetArray()));
136  } else {
137  list->Add(new TEfficiency(effname, efftitle, axis->GetNbins(), axis->GetXmin(), axis->GetXmax()));
138  }
139  };
140  makeEfficiency("efficiencyVsPt", "Efficiency " + tagPt + ";" + xPt + ";Efficiency", HIST("pt/num"));
141  makeEfficiency("efficiencyVsEta", "Efficiency " + tagEta + ";" + xEta + ";Efficiency", HIST("eta/num"));
142  makeEfficiency("efficiencyVsPhi", "Efficiency " + tagPhi + ";" + xPhi + ";Efficiency", HIST("phi/num"));
143  }

146  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>& collisions,
147  const o2::soa::Join<o2::aod::Tracks, o2::aod::McTrackLabels>& tracks,
148  const o2::aod::McCollisions& mcCollisions,
149  const o2::aod::McParticles& mcParticles)
150  {
151  std::vector<int64_t> recoEvt(collisions.size());
152  int nevts = 0;
153  for (const auto& collision : collisions) {
154  if (collision.numContrib() < nMinNumberOfContributors) {
155  continue;
156  }
157  const auto mcCollision = collision.mcCollision();
158  if ((mcCollision.posZ() < vertexZMin || mcCollision.posZ() > vertexZMax)) {
159  continue;
160  }
161  recoEvt[nevts++] = mcCollision.globalIndex();
162  }
163  recoEvt.resize(nevts);
164 
165  std::vector<int64_t> recoTracks(tracks.size());
166  int ntrks = 0;
167  for (const auto& track : tracks) {
168  const auto mcParticle = track.mcParticle();
169  if ((mcParticle.pt() < ptMin || mcParticle.pt() > ptMax)) { // Check pt
170  continue;
171  }
172  if ((mcParticle.eta() < etaMin || mcParticle.eta() > etaMax)) { // Check eta
173  continue;
174  }
175  if ((mcParticle.phi() < phiMin || mcParticle.phi() > phiMax)) { // Check phi
176  continue;
177  }
178  if ((selPrim == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
179  continue;
180  }
181  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
182  histos.fill(HIST("pt/num"), mcParticle.pt());
183  histos.fill(HIST("eta/num"), mcParticle.eta());
184  histos.fill(HIST("phi/num"), mcParticle.phi());
185  recoTracks[ntrks++] = mcParticle.globalIndex();
186  }
187  }
188 
189  for (const auto& mcParticle : mcParticles) {
190  const auto evtReconstructed = std::find(recoEvt.begin(), recoEvt.end(), mcParticle.mcCollision().globalIndex()) != recoEvt.end();
191  if (!evtReconstructed) {
192  continue;
193  }
194  if ((mcParticle.eta() < etaMin || mcParticle.eta() > etaMax)) { // Check eta
195  continue;
196  }
197  if ((mcParticle.phi() < phiMin || mcParticle.phi() > phiMax)) { // Check phi
198  continue;
199  }
200  if ((selPrim == 1) && (!MC::isPhysicalPrimary(mcParticles, mcParticle))) { // Requiring is physical primary
201  continue;
202  }
203  if (abs(mcParticle.pdgCode()) == particlePDG) { // Checking PDG code
204  if (makeEff) {
205  const auto particleReconstructed = std::find(recoTracks.begin(), recoTracks.end(), mcParticle.globalIndex()) != recoTracks.end();
206  static_cast<TEfficiency*>(list->At(0))->Fill(particleReconstructed, mcParticle.pt());
207  static_cast<TEfficiency*>(list->At(1))->Fill(particleReconstructed, mcParticle.eta());
208  static_cast<TEfficiency*>(list->At(2))->Fill(particleReconstructed, mcParticle.phi());
209  }
210  histos.fill(HIST("pt/den"), mcParticle.pt());
211  histos.fill(HIST("eta/den"), mcParticle.eta());
212  histos.fill(HIST("phi/den"), mcParticle.phi());
213  }
214  }
215  }
216 };
