40 template <typename T1, typename T2>
41 int findBin(o2::framework::Configurable<std::vector<T1>> const& bins, T2 value)
42 {
43  if (value < bins->front()) {
44  return -1;
45  }
46  if (value >= bins->back()) {
47  return -1;
48  }
49  return std::distance(bins->begin(), std::upper_bound(bins->begin(), bins->end(), value)) - 1;
50 }
