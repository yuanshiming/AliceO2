27  static int getMCMfromPad(int irow, int icol)
28  {
29  if (irow < 0 || icol < 0 || irow > constants::NROWC1 || icol > constants::NCOLUMN) {
30  return -1;
31  }
32  return (icol % (constants::NCOLUMN / 2)) / constants::NCOLMCM + constants::NMCMROBINCOL * (irow % constants::NMCMROBINROW);
33  }

35  static int getColSide(int icol)
36  {
37  if (icol < 0 || icol >= constants::NCOLUMN) {
38  return -1;
39  }
40 
41  return icol / (constants::NCOLUMN / 2);
42  }
43 
44  static int getPadRowFromMCM(int irob, int imcm)
45  {
46  return constants::NMCMROBINROW * (irob / 2) + imcm / constants::NMCMROBINCOL;

49  static int getPadColFromADC(int irob, int imcm, int iadc)
50  {
51  if (iadc < 0 || iadc > constants::NADCMCM) {
52  return -100;
53  }
54  int mcmcol = imcm % constants::NMCMROBINCOL + getROBSide(irob) * constants::NMCMROBINCOL; // MCM column number on ROC [0..7]
55  int padcol = mcmcol * constants::NCOLMCM + constants::NCOLMCM + 1 - iadc;
56  if (padcol < 0 || padcol >= constants::NCOLUMN) {
57  return -1; // this is commented because of reason above OK
58  }
59  return padcol;
60  }
61 
62  static int getROBSide(int irob)
63  {
64  if (irob < 0 || irob >= constants::NROBC1) {
65  return -1;
66  }
67  return irob % 2;
68  }
