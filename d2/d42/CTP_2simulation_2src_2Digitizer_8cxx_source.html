24 std::vector<CTPDigit> Digitizer::process(const gsl::span<o2::ctp::CTPInputDigit> digits)
25 {
26  std::map<o2::InteractionRecord, std::vector<const CTPInputDigit*>> prerawdata;
27  for (auto const& inp : digits) {
28  prerawdata[inp.intRecord].push_back(&inp);
29  }
30  std::vector<CTPDigit> vrawdata;
31  for (auto const& coll : prerawdata) {
32  CTPDigit data;
33  data.intRecord = coll.first;
34  std::bitset<CTP_NINPUTS> inpmaskcoll = 0;
35  for (auto const inp : coll.second) {
36  switch (inp->detector) {
37  case o2::detectors::DetID::FT0: {
38  std::bitset<CTP_NINPUTS> inpmask = std::bitset<CTP_NINPUTS>(
39  (inp->inputsMask & CTP_INPUTMASK_FT0.second).to_ullong());
40  inpmaskcoll |= inpmask << CTP_INPUTMASK_FT0.first;
41  break;
42  }
43  case o2::detectors::DetID::FV0: {
44  std::bitset<CTP_NINPUTS> inpmask = std::bitset<CTP_NINPUTS>(
45  (inp->inputsMask & CTP_INPUTMASK_FV0.second).to_ullong());
46  inpmaskcoll |= inpmask << CTP_INPUTMASK_FV0.first;
47  break;
48  }
49  default:
50  // Error
51  LOG(ERROR) << "CTP Digitizer: unknown detector:" << inp->detector;
52  break;
53  }
54  }
55  data.CTPInputMask = inpmaskcoll;
56  calculateClassMask(coll.second, data.CTPClassMask);
57  vrawdata.emplace_back(data);
58  }
59  return std::move(vrawdata);
60 }
