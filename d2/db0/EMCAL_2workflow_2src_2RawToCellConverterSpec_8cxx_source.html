43 void RawToCellConverterSpec::init(framework::InitContext& ctx)
44 {
45  LOG(DEBUG) << "[EMCALRawToCellConverter - init] Initialize converter ";
46  if (!mGeometry) {
47  mGeometry = Geometry::GetInstanceFromRunNumber(223409);
48  }
49  if (!mGeometry) {
50  LOG(ERROR) << "Failure accessing geometry";
51  }
52 
53  if (!mMapper) {
54  mMapper = std::unique_ptr<MappingHandler>(new o2::emcal::MappingHandler);
55  }
56  if (!mMapper) {
57  LOG(ERROR) << "Failed to initialize mapper";
58  }
59 
60  auto fitmethod = ctx.options().get<std::string>("fitmethod");
61  if (fitmethod == "standard") {
62  LOG(INFO) << "Using standard raw fitter";
63  mRawFitter = std::unique_ptr<CaloRawFitter>(new o2::emcal::CaloRawFitterStandard);
64  } else if (fitmethod == "gamma2") {
65  mRawFitter = std::unique_ptr<CaloRawFitter>(new o2::emcal::CaloRawFitterGamma2);
66  }
67 
68  mMaxErrorMessages = ctx.options().get<int>("maxmessage");
69  LOG(INFO) << "Suppressing error messages after " << mMaxErrorMessages << " messages";
70 
71  mRawFitter->setAmpCut(mNoiseThreshold);
72  mRawFitter->setL1Phase(0.);
73 }

75 void RawToCellConverterSpec::run(framework::ProcessingContext& ctx)
76 {
77  LOG(DEBUG) << "[EMCALRawToCellConverter - run] called";
78  const double CONVADCGEV = 0.016; // Conversion from ADC counts to energy: E = 16 MeV / ADC
79 
80  // Cache cells from for bunch crossings as the component reads timeframes from many links consecutively
81  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<Cell>>> cellBuffer; // Internal cell buffer
82  std::map<o2::InteractionRecord, uint32_t> triggerBuffer;
83 
84  mOutputDecoderErrors.clear();
85 
86  int firstEntry = 0;
87  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs())) {
88 
89  // Skip SOX headers
90  auto rdhblock = reinterpret_cast<const o2::header::RDHAny*>(rawData.payload);
91  if (o2::raw::RDHUtils::getHeaderSize(rdhblock) == static_cast<int>(o2::framework::DataRefUtils::getPayloadSize(rawData))) {
92  continue;
93  }
94 
95  //o2::emcal::RawReaderMemory<o2::header::RAWDataHeaderV4> rawreader(gsl::span(rawData.payload, o2::framework::DataRefUtils::getPayloadSize(rawData)));
96 
97  o2::emcal::RawReaderMemory rawreader(framework::DataRefUtils::as<const char>(rawData));
98 
99  // loop over all the DMA pages
100  while (rawreader.hasNext()) {
101 
102  rawreader.next();
103 
104  auto& header = rawreader.getRawHeader();
105  auto triggerBC = raw::RDHUtils::getTriggerBC(header);
106  auto triggerOrbit = raw::RDHUtils::getTriggerOrbit(header);
107  auto feeID = raw::RDHUtils::getFEEID(header);
108  auto triggerbits = raw::RDHUtils::getTriggerType(header);
109 
110  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
111  std::shared_ptr<std::vector<Cell>> currentCellContainer;
112  auto found = cellBuffer.find(currentIR);
113  if (found == cellBuffer.end()) {
114  currentCellContainer = std::make_shared<std::vector<Cell>>();
115  cellBuffer[currentIR] = currentCellContainer;
116  // also add trigger bits
117  triggerBuffer[currentIR] = triggerbits;
118  } else {
119  currentCellContainer = found->second;
120  }
121 
122  if (feeID > 40) {
123  continue; //skip STU ddl
124  }
125 
126  //std::cout<<rawreader.getRawHeader()<<std::endl;
127 
128  // use the altro decoder to decode the raw data, and extract the RCU trailer
129  AltroDecoder decoder(rawreader);
130  //check the words of the payload exception in altrodecoder
131  try {
132  decoder.decode();
133  } catch (AltroDecoderError& e) {
134  std::string errormessage;
135  using AltroErrType = AltroDecoderError::ErrorType_t;
137  ErrorTypeFEE errornum(feeID, AltroDecoderError::errorTypeToInt(e.getErrorType()), -1);
138  switch (e.getErrorType()) {
139  case AltroErrType::RCU_TRAILER_ERROR:
140  errormessage = " RCU Trailer Error ";
141  break;
142  case AltroErrType::RCU_VERSION_ERROR:
143  errormessage = " RCU Version Error ";
144  break;
145  case AltroErrType::RCU_TRAILER_SIZE_ERROR:
146  errormessage = " RCU Trailer Size Error ";
147  break;
148  case AltroErrType::ALTRO_BUNCH_HEADER_ERROR:
149  errormessage = " ALTRO Bunch Header Error ";
150  break;
151  case AltroErrType::ALTRO_BUNCH_LENGTH_ERROR:
152  errormessage = " ALTRO Bunch Length Error ";
153  break;
154  case AltroErrType::ALTRO_PAYLOAD_ERROR:
155  errormessage = " ALTRO Payload Error ";
156  break;
157  case AltroErrType::ALTRO_MAPPING_ERROR:
158  errormessage = " ALTRO Mapping Error ";
159  break;
160  case AltroErrType::CHANNEL_ERROR:
161  errormessage = " Channel Error ";
162  break;
163  default:
164  break;
165  }
166  if (mNumErrorMessages < mMaxErrorMessages) {
167  LOG(ERROR) << " EMCAL raw task: " << errormessage << " in Supermodule " << feeID << std::endl;
168  mNumErrorMessages++;
169  if (mNumErrorMessages == mMaxErrorMessages) {
170  LOG(ERROR) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
171  }
172  } else {
173  mErrorMessagesSuppressed++;
174  }
175  //fill histograms with error types
176  mOutputDecoderErrors.push_back(errornum);
177  continue;
178  }
179 
180  LOG(DEBUG) << decoder.getRCUTrailer();
181  // Apply zero suppression only in case it was enabled
182  mRawFitter->setIsZeroSuppressed(decoder.getRCUTrailer().hasZeroSuppression());
183 
184  const auto& map = mMapper->getMappingForDDL(feeID);
185  int iSM = feeID / 2;
186 
187  // Loop over all the channels
188  for (auto& chan : decoder.getChannels()) {
189 
190  int iRow, iCol;
191  ChannelType_t chantype;
192  try {
193  iRow = map.getRow(chan.getHardwareAddress());
194  iCol = map.getColumn(chan.getHardwareAddress());
195  chantype = map.getChannelType(chan.getHardwareAddress());
196  } catch (Mapper::AddressNotFoundException& ex) {
197  std::cerr << ex.what() << std::endl;
198  continue;
199  };
200 
201  int CellID = mGeometry->GetAbsCellIdFromCellIndexes(iSM, iRow, iCol);
202 
203  // define the conatiner for the fit results, and perform the raw fitting using the stadnard raw fitter
204  CaloFitResults fitResults;
205  try {
206  fitResults = mRawFitter->evaluate(chan.getBunches(), 0, 0);
207  // Prevent negative entries - we should no longer get here as the raw fit usually will end in an error state
208  if (fitResults.getAmp() < 0) {
209  fitResults.setAmp(0.);
210  }
211  if (fitResults.getTime() < 0) {
212  fitResults.setTime(0.);
213  }
214  } catch (CaloRawFitter::RawFitterError_t& fiterror) {
215  if (mNumErrorMessages < mMaxErrorMessages) {
216  LOG(ERROR) << "Failure in raw fitting: " << CaloRawFitter::createErrorMessage(fiterror);
217  mNumErrorMessages++;
218  if (mNumErrorMessages == mMaxErrorMessages) {
219  LOG(ERROR) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
220  }
221  } else {
222  mErrorMessagesSuppressed++;
223  }
224  mOutputDecoderErrors.emplace_back(feeID, -1, CaloRawFitter::getErrorNumber(fiterror));
225  }
226  currentCellContainer->emplace_back(CellID, fitResults.getAmp() * CONVADCGEV, fitResults.getTime(), chantype);
227  }
228  }
229  }
230 
231  // Loop over BCs, sort cells with increasing tower ID and write to output containers
232  mOutputCells.clear();
233  mOutputTriggerRecords.clear();
234  for (auto [bc, cells] : cellBuffer) {
235  mOutputTriggerRecords.emplace_back(bc, triggerBuffer[bc], mOutputCells.size(), cells->size());
236  if (cells->size()) {
237  // Sort cells according to cell ID
238  std::sort(cells->begin(), cells->end(), [](Cell& lhs, Cell& rhs) { return lhs.getTower() < rhs.getTower(); });
239  for (auto cell : *cells) {
240  mOutputCells.push_back(cell);
241  }
242  }
243  }
244 
245  LOG(DEBUG) << "[EMCALRawToCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
246  ctx.outputs().snapshot(framework::Output{"EMC", "CELLS", 0, framework::Lifetime::Timeframe}, mOutputCells);
247  ctx.outputs().snapshot(framework::Output{"EMC", "CELLSTRGR", 0, framework::Lifetime::Timeframe}, mOutputTriggerRecords);
248  ctx.outputs().snapshot(framework::Output{"EMC", "DECODERERR", 0, framework::Lifetime::Timeframe}, mOutputDecoderErrors);
249 }
