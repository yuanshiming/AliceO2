23 // No need to know the datamodel because the branch names follow a canonical standard (identified by fIndex)

30  int this_option_optind = optind ? optind : 1;

32  static struct option long_options[] = {
33  {"input", required_argument, nullptr, 0},
34  {"output", required_argument, nullptr, 1},
35  {"max-size", required_argument, nullptr, 2},
36  {"help", no_argument, nullptr, 3},
37  {nullptr, 0, nullptr, 0}};
38

40  int c = getopt_long(argc, argv, "", long_options, &option_index);

42  break;
43  } else if (c == 0) {
44  inputCollection = optarg;
45  } else if (c == 1) {
46  outputFileName = optarg;
47  } else if (c == 2) {
48  maxDirSize = atol(optarg);
49  } else if (c == 3) {
50  printf("AOD merging tool. Options: \n");
51  printf(" --input <inputfile.txt> Contains path to files to be merged. Default: %s\n", inputCollection.c_str());
52  printf(" --output <outputfile.root> Target output ROOT file. Default: %s\n", outputFileName.c_str());
53  printf(" --max-size <size in Bytes> Target directory size: %ld \n", maxDirSize);
54  return -1;
55  } else {
56  return -2;
57  }
58  }
59 
60  printf("AOD merger started with:\n");
61  printf(" Input file: %s\n", inputCollection.c_str());
62  printf(" Ouput file name: %s\n", outputFileName.c_str());
63  printf(" Maximal folder size (uncompressed): %ld\n", maxDirSize);

75  bool connectedToAliEn = false;
76  while (in.good()) {
77  in >> line;
78 
79  if (line.Length() == 0) {
80  continue;
81  }
82 
83  if (line.BeginsWith("alien:") && !connectedToAliEn) {
84  printf("Connecting to AliEn...");

98  }

111 
112  if (trees.count(treeName) == 0) {
113  // clone tree
114  // NOTE Basket size etc. are copied in CloneTree() ?
115  if (!outputDir) {
116  outputDir = outputFile->mkdir(dfName);
117  currentDirSize = 0;

133  TObjArray* branches = inputTree->GetListOfBranches();
134  for (int i = 0; i < branches->GetEntriesFast(); ++i) {
135  TBranch* br = (TBranch*)branches->UncheckedAt(i);
136  TString branchName(br->GetName());
137  if (branchName.BeginsWith("fIndex")) {
138  // Syntax: fIndex<Table>[_<Suffix>]
139  branchName.Remove(0, 6);
140  if (branchName.First("_") > 0) {
141  branchName.Remove(branchName.First("_"));
142  }
143  branchName.Remove(branchName.Length() - 1); // remove s
144  branchName.ToLower();

146

148 
149  inputTree->SetBranchAddress(br->GetName(), indexList.back().first);
150  outputTree->SetBranchAddress(br->GetName(), indexList.back().first);
151  }
152  }
153 
154  auto entries = inputTree->GetEntries();
155  for (int i = 0; i < entries; i++) {
156  inputTree->GetEntry(i);
157  // shift index columns by offset
158  for (const auto& idx : indexList) {
159  *(idx.first) += idx.second;
160  }
161  int nbytes = outputTree->Fill();
162  if (nbytes > 0) {
163  currentDirSize += nbytes;
164  }
165  }
166 
167  for (const auto& idx : indexList) {
168  delete idx.first;
169  }
170 
171  delete inputTree;
172  }
173  }
174 
175  // update offsets
176  for (auto const& tree : trees) {
177  offsets[tree.first] = tree.second->GetEntries();
178  }
179 
180  // check for not found tables
181  for (auto const& offset : offsets) {
182  if (trees.count(offset.first) == 0) {
183  printf("ERROR: Index on %s but no tree found\n", offset.first.c_str());
184  }
185  }
186 
187  if (currentDirSize > maxDirSize) {
188  printf("Maximum size reached: %ld. Closing folder.\n", currentDirSize);
189  for (auto const& tree : trees) {
190  //Printf("Writing %s", tree.first.c_str());
191  outputDir->cd();
192  tree.second->Write();
193  delete tree.second;
194  }
195  outputDir = nullptr;
196  trees.clear();
197  offsets.clear();
198  }
199  }
200  inputFile->Close();
201  }
202  outputFile->Write();
203  outputFile->Close();
204 
205  printf("AOD merger finished.\n");
206 
207  return 0;
208 }
