114 struct TripletHash : std::unary_function<Triplet_t, std::size_t> {
115  std::size_t operator()(Triplet_t const& e) const
116  {
117  std::size_t seed = 0;
118  boost::hash_combine(seed, e.get<0>());
119  boost::hash_combine(seed, e.get<1>());
120  boost::hash_combine(seed, e.get<2>());
121  return seed;
122  }
123 };
124 
125 struct TripletEqualTo : std::binary_function<Triplet_t, Triplet_t, bool> {
126  bool operator()(Triplet_t const& x, Triplet_t const& y) const
127  {
128  return (x.get<0>() == y.get<0>() &&
129  x.get<1>() == y.get<1>() &&
130  x.get<2>() == y.get<2>());
131  }
132 };
