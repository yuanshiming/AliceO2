128  return pt * (snp * cs + r * sn);
129  });

548 DECLARE_SOA_DYNAMIC_COLUMN(Eta, eta,
549  [](float inverseBendingMomentum, float thetaX, float thetaY) -> float {
550  float pz = -std::sqrt(1.0 + std::tan(thetaY) * std::tan(thetaY)) / std::abs(inverseBendingMomentum);
551  float pt = std::abs(pz) * std::sqrt(std::tan(thetaX) * std::tan(thetaX) + std::tan(thetaY) * std::tan(thetaY));
552  float eta = std::acos(pz / std::sqrt(pt * pt + pz * pz));
553  eta = std::tan(0.5 * eta);
554  if (eta > 0.0)
555  return -std::log(eta);
556  else
557  return 0.0;
558  });
559 DECLARE_SOA_DYNAMIC_COLUMN(Phi, phi,
560  [](float thetaX, float thetaY) -> float {
561  float phi = std::atan2(std::tan(thetaY), std::tan(thetaX));
562  constexpr float twopi = 2.0f * static_cast<float>(M_PI);
563  return (phi >= 0.0 ? phi : phi + twopi);
564  });
565 DECLARE_SOA_DYNAMIC_COLUMN(RAtAbsorberEnd, rAtAbsorberEnd,
566  [](float bendingCoor, float nonBendingCoor, float zMu, float thetaX, float thetaY) -> float {
567  float dZ = -505. - zMu;
568  float NonBendingSlope = std::tan(thetaX);
569  float BendingSlope = std::tan(thetaY);
570  float xAbs = nonBendingCoor + NonBendingSlope * dZ;
571  float yAbs = bendingCoor + BendingSlope * dZ;
572  float rAtAbsorberEnd = std::sqrt(xAbs * xAbs + yAbs * yAbs);
573  return rAtAbsorberEnd;
574  });
575 DECLARE_SOA_DYNAMIC_COLUMN(PDca, pDca,
576  [](float inverseBendingMomentum, float thetaX, float thetaY, float bendingCoor, float nonBendingCoor, float zMu) -> float {
577  float dca = std::sqrt(bendingCoor * bendingCoor + nonBendingCoor * nonBendingCoor + zMu * zMu);
578  float pz = -std::sqrt(1.0 + std::tan(thetaY) * std::tan(thetaY)) / std::abs(inverseBendingMomentum);
579  float pt = std::abs(pz) * std::sqrt(std::tan(thetaX) * std::tan(thetaX) + std::tan(thetaY) * std::tan(thetaY));
580  float pTot = std::sqrt(pt * pt + pz * pz);
581  float pDca = pTot * dca;
582  return pDca;
583  });
