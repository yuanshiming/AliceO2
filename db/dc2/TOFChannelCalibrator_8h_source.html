128  static double FuncDeltaOffset(double* x, double* params)
129  {
130  int i1 = int(x[0]) % 96;
131  int i2 = int(x[0]) / 96 ? (i1 + 48) : (i1 + 1);
132 
133  if (i1 < 0) {
134  return 0;
135  }
136  if (i2 >= Geo::NPADS) {
137  return 0;
138  }
139 
140  return (params[i1] - params[i2]);
141  }

173  void finalizeSlotWithCosmics(Slot& slot)
174  {
175  // Extract results for the single slot
176  o2::tof::TOFChannelData* c = slot.getContainer();
177  LOG(INFO) << "Finalize slot for calibration with cosmics " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
178 
179  if (!mFuncDeltaOffset) {
180  mFuncDeltaOffset = new TF1("fChanOffset", FuncDeltaOffset, 0, NCOMBINSTRIP, Geo::NPADS);
181  mFuncDeltaOffset->FixParameter(24, 0); // fix one pad to fixed offset (as reference)
182  for (int ichLocal = 0; ichLocal < Geo::NPADS; ichLocal++) {
183  if (ichLocal == 24) {
184  continue; //already fixed
185  mFuncDeltaOffset->SetParLimits(ichLocal, -mRange, mRange);
186  }
187  }
188  }
189 
190  // for the CCDB entry
191  std::map<std::string, std::string> md;
192  TimeSlewing& ts = mCalibTOFapi->getSlewParamObj(); // we take the current CCDB object, since we want to simply update the offset
193 
194  float xp[NCOMBINSTRIP], exp[NCOMBINSTRIP], deltat[NCOMBINSTRIP], edeltat[NCOMBINSTRIP], fracUnderPeak[Geo::NPADS];
195 
196  for (int sector = 0; sector < Geo::NSECTORS; sector++) {
197  int offsetsector = sector * Geo::NSTRIPXSECTOR * Geo::NPADS;
198  for (int istrip = 0; istrip < Geo::NSTRIPXSECTOR; istrip++) {
199  int offsetstrip = istrip * Geo::NPADS + offsetsector;
200  int goodpoints = 0;
201 
202  memset(&fracUnderPeak[0], 0, sizeof(fracUnderPeak));
203 
204  for (int ipair = 0; ipair < NCOMBINSTRIP; ipair++) {
205  int chinsector = ipair + istrip * NCOMBINSTRIP;
206  int ich = chinsector + sector * Geo::NSTRIPXSECTOR * NCOMBINSTRIP;
207  auto entriesInPair = c->integral(ich);
208  if (entriesInPair < mMinEntries && entriesInPair != 0) {
209  LOG(DEBUG) << "pair " << ich << " will not be calibrated since it has only " << entriesInPair << " entries (min = " << mMinEntries << ")";
210  continue;
211  }
212  // make the slice of the 2D histogram so that we have the 1D of the current channel
213  std::vector<float> fitValues;
214  std::vector<float> histoValues;
215  std::vector<int> entriesPerChannel = c->getEntriesPerChannel();
216  if (entriesPerChannel.at(ich) == 0) {
217  continue; // skip always since a channel with 0 entries is normal, it will be flagged as problematic
218  if (mTest) {
219  LOG(INFO) << "Skipping channel " << ich << " because it has zero entries, but it should not be"; // should become error!
220  continue;
221  } else {
222  throw std::runtime_error("We found one channel with no entries, we cannot calibrate!");
223  }
224  }
225 
226  // more efficient way
227  auto histo = c->getHisto(sector);
228  for (unsigned j = chinsector; j <= chinsector; ++j) {
229  for (unsigned i = 0; i < c->getNbins(); ++i) {
230  const auto& v = histo.at(i, j);
231  LOG(DEBUG) << "channel = " << ich << ", in sector = " << sector << " (where it is channel = " << chinsector << ") bin = " << i << " value = " << v;
232  histoValues.push_back(v);
233  }
234  }
235 
236  double fitres = fitGaus(c->getNbins(), histoValues.data(), -(c->getRange()), c->getRange(), fitValues);
237 
238  if (fitres >= 0) {
239  LOG(DEBUG) << "Pair " << ich << " :: Fit result " << fitres << " Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
240  } else {
241  //LOG(INFO) << "Pair " << ich << " :: Fit failed with result = " << fitres;
242  continue;
243  }
244 
245  if (fitValues[2] < 0) {
246  fitValues[2] = -fitValues[2];
247  }
248 
249  float intmin = fitValues[1] - 5 * fitValues[2]; // mean - 5*sigma
250  float intmax = fitValues[1] + 5 * fitValues[2]; // mean + 5*sigma
251 
252  if (intmin < -mRange) {
253  intmin = -mRange;
254  }
255  if (intmax < -mRange) {
256  intmax = -mRange;
257  }
258  if (intmin > mRange) {
259  intmin = mRange;
260  }
261  if (intmax > mRange) {
262  intmax = mRange;
263  }
264 
265  xp[goodpoints] = ipair + 0.5; // pair index
266  exp[goodpoints] = 0.0; // error on pair index (dummy since it is on the pair index)
267  deltat[goodpoints] = fitValues[1]; // delta between offsets from channels in pair (from the fit) - in ps
268  edeltat[goodpoints] = 20; // TODO: for now put by default to 20 ps since it was seen to be reasonable; but it should come from the fit: who gives us the error from the fit ??????
269  goodpoints++;
270  int ch1 = ipair % 96;
271  int ch2 = ipair / 96 ? ch1 + 48 : ch1 + 1;
272  float fractionUnderPeak = entriesInPair > 0 ? c->integral(ich, intmin, intmax) / entriesInPair : 0;
273  // we keep as fractionUnderPeak of the channel the largest one that is found in the 3 possible pairs with that channel (for both channels ch1 and ch2 in the pair)
274  if (fracUnderPeak[ch1] < fractionUnderPeak) {
275  fracUnderPeak[ch1] = fractionUnderPeak;
276  }
277  if (fracUnderPeak[ch2] < fractionUnderPeak) {
278  fracUnderPeak[ch2] = fractionUnderPeak;
279  }
280  } // end loop pairs
281 
282  // fit strip offset
283  if (goodpoints == 0) {
284  //LOG(INFO) << "We did not find any good point for strip " << istrip << " in sector " << sector;
285  continue;
286  }
287  LOG(DEBUG) << "We found " << goodpoints << " good points for strip " << istrip << " in sector " << sector << " --> we can fit the TGraph";
288  TGraphErrors g(goodpoints, xp, deltat, exp, edeltat);
289  g.Fit(mFuncDeltaOffset, "Q0");
290 
291  //update calibrations
292  for (int ichLocal = 0; ichLocal < Geo::NPADS; ichLocal++) {
293  int ich = ichLocal + offsetstrip;
294  ts.updateOffsetInfo(ich, mFuncDeltaOffset->GetParameter(ichLocal));
295  ts.setFractionUnderPeak(ich / Geo::NPADSXSECTOR, ich % Geo::NPADSXSECTOR, fracUnderPeak[ichLocal]);
296  ts.setSigmaPeak(ich / Geo::NPADSXSECTOR, ich % Geo::NPADSXSECTOR, abs(mFuncDeltaOffset->GetParError(ichLocal)));
297  }
298 
299  } // end loop strips
300  } // end loop sectors
301 
302  auto clName = o2::utils::MemFileHelper::getClassName(ts);
303  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
304  mInfoVector.emplace_back("TOF/ChannelCalib", clName, flName, md, slot.getTFStart(), 99999999999999);
305  mTimeSlewingVector.emplace_back(ts);
306  }

308  void finalizeSlotWithTracks(Slot& slot)
309  {
310  // Extract results for the single slot
311  o2::tof::TOFChannelData* c = slot.getContainer();
312  LOG(INFO) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
313 
314  // for the CCDB entry
315  std::map<std::string, std::string> md;
316  TimeSlewing& ts = mCalibTOFapi->getSlewParamObj(); // we take the current CCDB object, since we want to simply update the offset
317 
318  for (int ich = 0; ich < Geo::NCHANNELS; ich++) {
319  // make the slice of the 2D histogram so that we have the 1D of the current channel
320  int sector = ich / Geo::NPADSXSECTOR;
321  int chinsector = ich % Geo::NPADSXSECTOR;
322  auto entriesInChannel = c->integral(ich);
323  if (entriesInChannel < mMinEntries) {
324  LOG(DEBUG) << "channel " << ich << " will not be calibrated since it has only " << entriesInChannel << " entries (min = " << mMinEntries << ")";
325  continue;
326  }
327  std::vector<float> fitValues;
328  std::vector<float> histoValues;
329  std::vector<int> entriesPerChannel = c->getEntriesPerChannel();
330  if (entriesPerChannel.at(ich) == 0) {
331  continue; // skip always since a channel with 0 entries is normal, it will be flagged as problematic
332  if (mTest) {
333  LOG(DEBUG) << "Skipping channel " << ich << " because it has zero entries, but it should not be"; // should become error!
334  continue;
335  } else {
336  throw std::runtime_error("We found one channel with no entries, we cannot calibrate!");
337  }
338  }
339 
340  // more efficient way
341  auto histo = c->getHisto(sector);
342  for (unsigned j = chinsector; j <= chinsector; ++j) {
343  for (unsigned i = 0; i < c->getNbins(); ++i) {
344  const auto& v = histo.at(i, j);
345  LOG(DEBUG) << "channel = " << ich << ", in sector = " << sector << " (where it is channel = " << chinsector << ") bin = " << i << " value = " << v;
346  histoValues.push_back(v);
347  }
348  }
349 
350  double fitres = fitGaus(c->getNbins(), histoValues.data(), -(c->getRange()), c->getRange(), fitValues);
351 
352  if (fitres >= 0) {
353  LOG(DEBUG) << "Channel " << ich << " :: Fit result " << fitres << " Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
354  } else {
355  // LOG(INFO) << "Channel " << ich << " :: Fit failed with result = " << fitres;
356  continue;
357  }
358 
359  if (fitValues[2] < 0) {
360  fitValues[2] = -fitValues[2];
361  }
362 
363  float fractionUnderPeak;
364  float intmin = fitValues[1] - 5 * fitValues[2]; // mean - 5*sigma
365  float intmax = fitValues[1] + 5 * fitValues[2]; // mean + 5*sigma
366 
367  if (intmin < -mRange) {
368  intmin = -mRange;
369  }
370  if (intmax < -mRange) {
371  intmax = -mRange;
372  }
373  if (intmin > mRange) {
374  intmin = mRange;
375  }
376  if (intmax > mRange) {
377  intmax = mRange;
378  }
379 
380  fractionUnderPeak = entriesInChannel > 0 ? c->integral(ich, intmin, intmax) / entriesInChannel : 0;
381  // now we need to store the results in the TimeSlewingObject
382  ts.setFractionUnderPeak(ich / Geo::NPADSXSECTOR, ich % Geo::NPADSXSECTOR, fractionUnderPeak);
383  ts.setSigmaPeak(ich / Geo::NPADSXSECTOR, ich % Geo::NPADSXSECTOR, abs(fitValues[2]));
384  ts.updateOffsetInfo(ich, fitValues[1]);
385  }
386  auto clName = o2::utils::MemFileHelper::getClassName(ts);
387  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
388  mInfoVector.emplace_back("TOF/ChannelCalib", clName, flName, md, slot.getTFStart(), 99999999999999);
389  mTimeSlewingVector.emplace_back(ts);
390  }
