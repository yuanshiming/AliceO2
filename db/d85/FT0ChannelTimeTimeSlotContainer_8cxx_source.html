57 int16_t FT0ChannelTimeTimeSlotContainer::getMeanGaussianFitValue(std::size_t channelID) const
58 {
59 
60  static constexpr size_t MEAN_VALUE_INDEX_IN_OUTPUT_VECTOR = 1;
61 
62  if (0 == mEntriesPerChannel[channelID]) {
63  return 0;
64  }
65 
66  std::vector<double> channelHistogramData(NUMBER_OF_HISTOGRAM_BINS);
67  std::vector<double> outputGaussianFitValues;
68  for (int iBin = 0; iBin < NUMBER_OF_HISTOGRAM_BINS; ++iBin) {
69  channelHistogramData[iBin] = mHistogram.at(iBin, channelID);
70  }
71 
72  double returnCode = math_utils::fitGaus<double>(NUMBER_OF_HISTOGRAM_BINS, channelHistogramData.data(),
73  -HISTOGRAM_RANGE, HISTOGRAM_RANGE, outputGaussianFitValues);
74  if (returnCode < 0) {
75  LOG(ERROR) << "Gaussian fit error!";
76  return 0;
77  }
78 
79  return static_cast<int16_t>(std::round(outputGaussianFitValues[MEAN_VALUE_INDEX_IN_OUTPUT_VECTOR]));
80 }
