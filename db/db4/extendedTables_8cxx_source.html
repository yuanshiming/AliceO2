43 // Extend table Tracks with expression column
44 struct ExtendTable {
45  // group tracks according to collisions
46  void process(aod::Collision const&, aod::Tracks const& tracks)
47  {
48  // add expression column o2::aod::extension::P2exp to table
49  // o2::aod::Tracks
50  auto table_extension = soa::Extend<aod::Tracks, aod::extension::P2exp>(tracks);
51 
52  // loop over the rows of the new table
53  for (auto& row : table_extension) {
54  if (row.trackType() != 3) {
55  if (row.index() % 10000 == 0) {
56  LOGF(info, "EXPRESSION Pt^2 = %.3f", row.p2exp());
57  }
58  }
59  }
60  }
61 };

83 // extend and attach within process function
84 struct ExtendAndAttach {
85  void process(aod::Collision const&, aod::Tracks const& tracks)
86  {
87 
88  // combine Extend and Attach to create a new table
89  auto table_extension = soa::Extend<aod::Tracks, aod::extension::P2exp>(tracks);
90  auto table_attached = soa::Attach<decltype(table_extension),
91  aod::extension::P2dyn<aod::track::P>,
92  aod::extension::R2dyn<aod::track::X, aod::track::Y>>(table_extension);
93 
94  // loop over the rows of the new table
95  for (auto& row : table_attached) {
96  if (row.trackType() != 3) {
97  if (row.index() % 10000 == 0) {
98  LOGF(info, "C: EXPRESSION P^2 = %.3f, DYNAMIC P^2 = %.3f R^2 = %.3f", row.p2exp(), row.p2dyn(), row.r2dyn());
99  }
100  }
101  }
102  }
