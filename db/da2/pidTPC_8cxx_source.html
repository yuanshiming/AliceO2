75  void init(o2::framework::InitContext& initContext)
76  {
77  // Checking the tables are requested in the workflow and enabling them
78  auto& workflows = initContext.services().get<RunningWorkflowInfo const>();
79  for (DeviceSpec device : workflows.devices) {
80  for (auto input : device.inputs) {
81  auto enableFlag = [&input](const std::string particle, Configurable<int>& flag) {
82  const std::string table = "pidTPC" + particle;
83  if (input.matcher.binding == table) {
84  if (flag < 0) {
85  flag.value = 1;
86  LOG(INFO) << "Auto-enabling table: " + table;
87  } else if (flag > 0) {
88  flag.value = 1;
89  LOG(INFO) << "Table enabled: " + table;
90  } else {
91  LOG(INFO) << "Table disabled: " + table;
92  }
93  }
94  };
95  enableFlag("El", pidEl);
96  enableFlag("Mu", pidMu);
97  enableFlag("Pi", pidPi);
98  enableFlag("Ka", pidKa);
99  enableFlag("Pr", pidPr);
100  enableFlag("De", pidDe);
101  enableFlag("Tr", pidTr);
102  enableFlag("He", pidHe);
103  enableFlag("Al", pidAl);
104  }
105  }
106  // Getting the parametrization parameters
107  ccdb->setURL(url.value);
108  ccdb->setTimestamp(timestamp.value);
109  ccdb->setCaching(true);
110  ccdb->setLocalObjectValidityChecking();
111  // Not later than now objects
112  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
113  //
114  const std::string fname = paramfile.value;
115  if (!fname.empty()) { // Loading the parametrization from file
116  LOG(INFO) << "Loading exp. signal parametrization from file" << fname << ", using param: " << signalname.value;
117  response.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
118 
119  LOG(INFO) << "Loading exp. sigma parametrization from file" << fname << ", using param: " << sigmaname.value;
120  response.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
121  } else { // Loading it from CCDB
122  std::string path = ccdbPath.value + "/" + signalname.value;
123  LOG(INFO) << "Loading exp. signal parametrization from CCDB, using path: " << path << " for timestamp " << timestamp.value;
124  response.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path, timestamp.value));
125 
126  path = ccdbPath.value + "/" + sigmaname.value;
127  LOG(INFO) << "Loading exp. sigma parametrization from CCDB, using path: " << path << " for timestamp " << timestamp.value;
128  response.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path, timestamp.value));
129  }
130  }

133  using ResponseImplementation = o2::pid::tpc::ELoss<Coll::iterator, Trks::iterator, pid>;
134  void process(Coll const& collisions, Trks const& tracks)
135  {
136  constexpr auto responseEl = ResponseImplementation<PID::Electron>();
137  constexpr auto responseMu = ResponseImplementation<PID::Muon>();
138  constexpr auto responsePi = ResponseImplementation<PID::Pion>();
139  constexpr auto responseKa = ResponseImplementation<PID::Kaon>();
140  constexpr auto responsePr = ResponseImplementation<PID::Proton>();
141  constexpr auto responseDe = ResponseImplementation<PID::Deuteron>();
142  constexpr auto responseTr = ResponseImplementation<PID::Triton>();
143  constexpr auto responseHe = ResponseImplementation<PID::Helium3>();
144  constexpr auto responseAl = ResponseImplementation<PID::Alpha>();
145 
146  // Check and fill enabled tables
147  auto makeTable = [&tracks](const Configurable<int>& flag, auto& table, const DetectorResponse& response, const auto& responsePID) {
148  if (flag.value == 1) {
149  // Prepare memory for enabled tables
150  table.reserve(tracks.size());
151  for (auto const& trk : tracks) { // Loop on Tracks
152  const float separation = responsePID.GetSeparation(response, trk.collision(), trk);
153  if (separation <= o2::aod::pidtpc_tiny::binned_min) {
154  table(o2::aod::pidtpc_tiny::lower_bin);
155  } else if (separation >= o2::aod::pidtpc_tiny::binned_max) {
156  table(o2::aod::pidtpc_tiny::upper_bin);
157  } else if (separation >= 0) {
158  table(separation / o2::aod::pidtpc_tiny::bin_width + 0.5f);
159  } else {
160  table(separation / o2::aod::pidtpc_tiny::bin_width - 0.5f);
161  }
162  }
163  }
164  };
165  makeTable(pidEl, tablePIDEl, response, responseEl);
166  makeTable(pidMu, tablePIDMu, response, responseMu);
167  makeTable(pidPi, tablePIDPi, response, responsePi);
168  makeTable(pidKa, tablePIDKa, response, responseKa);
169  makeTable(pidPr, tablePIDPr, response, responsePr);
170  makeTable(pidDe, tablePIDDe, response, responseDe);
171  makeTable(pidTr, tablePIDTr, response, responseTr);
172  makeTable(pidHe, tablePIDHe, response, responseHe);
173  makeTable(pidAl, tablePIDAl, response, responseAl);
174  }
175 };

195  template <typename T>
196  void makelogaxis(T h)
197  {
198  const int nbins = h->GetNbinsX();
199  double binp[nbins + 1];
200  double max = h->GetXaxis()->GetBinUpEdge(nbins);
201  double min = h->GetXaxis()->GetBinLowEdge(1);
202  if (min <= 0) {
203  min = 0.00001;
204  }
205  double lmin = TMath::Log10(min);
206  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
207  for (int i = 0; i < nbins; i++) {
208  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
209  }
210  binp[nbins] = max + 1;
211  h->GetXaxis()->Set(nbins, binp);
212  }
