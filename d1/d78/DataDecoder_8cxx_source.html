70 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
71 {
72  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
73 
74  auto existingFeeId = o2::raw::RDHUtils::getFEEID(rdhAny);
75  if (existingFeeId == 0) {
76  // early versions of raw data did not set the feeId
77  // which we need to select the right decoder
78 
79  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
80  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);
81  auto flags = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF00;
82 
83  uint32_t feeId = cruId * 2 + endpoint + flags;
84  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);
85  }

135 void DataDecoder::decodeBuffer(gsl::span<const std::byte> buf)
136 {
137  if (mDebug) {
138  std::cout << "\n\n============================\nStart of new buffer\n";
139  }
140  size_t bufSize = buf.size();
141  size_t pageStart = 0;
142  while (bufSize > pageStart) {
143  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
144  if (mDebug) {
145  if (pageStart == 0) {
146  std::cout << "+++\n[decodeBuffer]" << std::endl;
147  } else {
148  std::cout << "---\n[decodeBuffer]" << std::endl;
149  }
150  o2::raw::RDHUtils::printRDH(rdh);
151  }
152  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
153  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
154  if (rdhHeaderSize != 64) {
155  break;
156  }
157  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
158 
159  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
160  decodePage(page);
161 
162  pageStart += pageSize;
163  }
164 
165  if (mDebug) {
166  std::cout << "[decodeBuffer] mOrbits size: " << mOrbits.size() << std::endl;
167  dumpOrbits(mOrbits);
168  std::cout << "[decodeBuffer] mDigits size: " << mDigits.size() << std::endl;
169  dumpDigits();
170  }

173 void DataDecoder::dumpDigits()
174 {
175  for (size_t di = 0; di < mDigits.size(); di++) {
176  auto& d = mDigits[di];
177  auto detID = d.getDetID();
178  auto padID = d.getPadID();
179  if (padID < 0) {
180  continue;
181  }
182  const Segmentation& segment = segmentation(detID);
183  bool bend = segment.isBendingPad(padID);
184  float X = segment.padPositionX(padID);
185  float Y = segment.padPositionY(padID);
186  uint32_t orbit = d.getOrbit();
187  uint32_t bunchCrossing = d.getBunchCrossing();
188  uint32_t sampaTime = d.getSampaTime();
189  std::cout << fmt::format(" DE {:4d} PAD {:5d} ADC {:6d} TIME ({} {} {:4d})",
190  detID, padID, d.getADC(), orbit, bunchCrossing, sampaTime);
191  std::cout << fmt::format("\tC {} PAD_XY {:+2.2f} , {:+2.2f}", (bend ? (int)0 : (int)1), X, Y);
192  std::cout << std::endl;

204 void DataDecoder::decodePage(gsl::span<const std::byte> page)
205 {
206  size_t ndigits{0};
207 
208  auto heartBeatHandler = [&](DsElecId dsElecId, uint8_t chip, uint32_t bunchCrossing) {
209  SampaInfo sampaId;
210  sampaId.chip = chip;
211  sampaId.ds = dsElecId.elinkId();
212  sampaId.solar = dsElecId.solarId();
213 
214  mSampaTimeFrameStarts[sampaId.id].emplace(mOrbit, bunchCrossing);
215  if (mDebug) {
216  auto s = asString(dsElecId);
217  std::cout << "HeartBeat: " << s << " ID " << sampaId.id << " solar " << sampaId.solar << " ds " << sampaId.ds << " chip " << sampaId.chip
218  << " mOrbit " << mOrbit << " bunchCrossing " << bunchCrossing << std::endl;
219  }
220  };
221 
222  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
223  if (mChannelHandler) {
224  mChannelHandler(dsElecId, channel, sc);
225  }
226 
227  if (mDs2manu) {
228  channel = ds2manu(int(channel));
229  }
230 
231  uint32_t digitadc = sc.sum();
232 
233  int deId{-1};
234  int dsIddet{-1};
235  if (auto opt = mElec2Det(dsElecId); opt.has_value()) {
236  DsDetId dsDetId = opt.value();
237  dsIddet = dsDetId.dsId();
238  deId = dsDetId.deId();
239  }
240  if (mDebug) {
241  auto s = asString(dsElecId);
242  auto ch = fmt::format("{}-CH{:02d}", s, channel);
243  std::cout << ch << " "
244  << "deId " << deId << " dsIddet " << dsIddet << std::endl;
245  }
246 
247  if (deId < 0 || dsIddet < 0 || !isValidDeID(deId)) {
248  return;
249  }
250 
251  int padId = -1;
252  const Segmentation& segment = segmentation(deId);
253 
254  padId = segment.findPadByFEE(dsIddet, int(channel));
255  if (mDebug) {
256  auto s = asString(dsElecId);
257  auto ch = fmt::format("{}-CH{:02d}", s, channel);
258  std::cout << ch << " "
259  << fmt::format("PAD ({:04d} {:04d} {:04d})\tADC {:06d} TIME ({} {} {:02d}) SIZE {} END {}",
260  deId, dsIddet, padId, digitadc, mOrbit, sc.bunchCrossing, sc.sampaTime, sc.nofSamples(), (sc.sampaTime + sc.nofSamples() - 1))
261  << (((sc.sampaTime + sc.nofSamples() - 1) >= 98) ? " *" : "") << std::endl;
262  }
263 
264  // skip channels not associated to any pad
265  if (padId < 0) {
266  return;
267  }
268 
269  SampaInfo sampaInfo;
270  sampaInfo.chip = channel / 32;
271  sampaInfo.ds = dsElecId.elinkId();
272  sampaInfo.solar = dsElecId.solarId();
273  sampaInfo.sampaTime = sc.sampaTime;
274  sampaInfo.bunchCrossing = sc.bunchCrossing;
275  sampaInfo.orbit = mOrbit;
276 
277  RawDigit digit;
278  digit.digit = o2::mch::Digit(deId, padId, digitadc, 0, sc.nofSamples());
279  digit.info.chip = channel / 32;
280  digit.info.ds = dsElecId.elinkId();
281  digit.info.solar = dsElecId.solarId();
282  digit.info.sampaTime = sc.sampaTime;
283  digit.info.bunchCrossing = sc.bunchCrossing;
284  digit.info.orbit = mOrbit;
285 
286  mDigits.emplace_back(digit);
287 
288  if (mDebug) {
289  RawDigit& lastDigit = mDigits.back();
290  std::cout << "DIGIT STORED:\n ORBIT " << mOrbitId << " ADC " << lastDigit.getADC()
291  << " DE# " << lastDigit.getDetID() << " PadId " << lastDigit.getPadID()
292  << " time " << lastDigit.getSampaTime() << std::endl;
293  }
294  ++ndigits;
295  };
296 
297  patchPage(page, mDebug);
298 
299  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(page[0])));
300  mOrbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdhAny);
301  mOrbitId = mOrbit - mFirstTForbit;
302 
303  if (mRdhHandler) {
304  mRdhHandler(&rdhAny);
305  }
306 
307  // add orbit to vector if not present yet
308  mOrbits.emplace(page);
309 
310  if (!mDecoder) {
311  DecodedDataHandlers handlers;
312  handlers.sampaChannelHandler = channelHandler;
313  handlers.sampaHeartBeatHandler = heartBeatHandler;
314  mDecoder = mFee2Solar ? o2::mch::raw::createPageDecoder(page, handlers, mFee2Solar)
315  : o2::mch::raw::createPageDecoder(page, handlers);
316  }
317  mDecoder(page);
318 };

320 int32_t DataDecoder::digitsTimeDiff(uint32_t orbit1, uint32_t bc1, uint32_t orbit2, uint32_t bc2)
321 {
322  // bunch crossings are stored with 20 bits
323  static const int32_t BCROLLOVER = (1 << 20);
324  // number of bunch crossings in one orbit
325  static const int32_t BCINORBIT = o2::constants::lhc::LHCMaxBunches;
326 
327  // We use the difference of orbits values to estimate the minimum and maximum allowed
328  // difference in bunch crossings
329  int64_t dOrbit = static_cast<int64_t>(orbit2) - static_cast<int64_t>(orbit1);
330 
331  // Digits might be sent out later than the orbit in which they were recorded.
332  // We account for this by allowing an extra +/- 3 orbits when converting the
333  // difference from orbit numbers to bunch crossings.
334  int64_t dBcMin = (dOrbit - 3) * BCINORBIT;
335  int64_t dBcMax = (dOrbit + 3) * BCINORBIT;
336 
337  // Difference in bunch crossing values
338  int64_t dBc = static_cast<int64_t>(bc2) - static_cast<int64_t>(bc1);
339 
340  if (dBc < dBcMin) {
341  // the difference is too small, so we assume that it needs to be
342  // incremented by one rollover factor
343  dBc += BCROLLOVER;
344  } else if (dBc > dBcMax) {
345  // the difference is too big, so we assume that it needs to be
346  // decremented by one rollover factor
347  dBc -= BCROLLOVER;
348  }
349 
350  return static_cast<int32_t>(dBc);
351 }

353 void DataDecoder::computeDigitsTime_(RawDigitVector& digits, SampaTimeFrameStarts& sampaTimeFrameStarts, bool debug)
354 {
355  constexpr int32_t timeInvalid = DataDecoder::tfTimeInvalid;
356  auto setDigitTime = [&](Digit& d, int32_t tfTime) {
357  d.setTime(tfTime);
358  if (debug) {
359  std::cout << "[computeDigitsTime_] hit time set to " << d.getTime() << std::endl;
360  }
361  };
362 
363  for (size_t di = 0; di < digits.size(); di++) {
364  Digit& d = digits[di].digit;
365  SampaInfo& info = digits[di].info;
366  std::optional<SampaTimeFrameStart>& tfStart = sampaTimeFrameStarts[info.id];
367 
368  if (!tfStart) {
369  LOG(ERROR) << "\n[computeDigitsTime_] missing TimeFrame start for ID " << info.id << " DS " << info.solar << ","
370  << info.ds << " (J" << (info.ds / 5) + 1 << " B" << (info.ds % 5) << ")," << info.chip
371  << " pad " << d.getDetID() << "," << d.getPadID() << " "
372  << info.orbit << " " << info.tfTime << " " << info.bunchCrossing << " "
373  << info.sampaTime << " " << info.getBXTime();
374  d.setTime(timeInvalid);
375  info.tfTime = timeInvalid;
376  } else {
377  int32_t tfTime = 0;
378  uint32_t bc = tfStart->mBunchCrossing;
379  uint32_t orbit = tfStart->mOrbit;
380  tfTime = DataDecoder::digitsTimeDiff(orbit, bc, info.orbit, info.getBXTime());
381  if (debug) {
382  std::cout << "\n[computeDigitsTime_] hit " << info.orbit << "," << info.getBXTime()
383  << " tfTime(1) " << orbit << "," << bc << " diff " << tfTime << std::endl;
384  }
385  setDigitTime(d, tfTime);
386  info.tfTime = tfTime;
387  }
388 
389  if (debug) {
390  std::cout << " solar " << info.solar << " ds " << info.ds << " chip " << info.chip << std::endl;
391  std::cout << " pad " << d.getDetID() << "," << d.getPadID() << " "
392  << info.orbit << " " << info.tfTime << " " << info.getBXTime() << std::endl;
393  }
394  }
395 }
