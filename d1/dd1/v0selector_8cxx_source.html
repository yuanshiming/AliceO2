99  float alphav0(const array<float, 3>& ppos, const array<float, 3>& pneg)
100  {
101  std::array<float, 3> pv0 = {ppos[0] + pneg[0], ppos[1] + pneg[1], ppos[2] + pneg[2]};
102  float momTot = RecoDecay::P(pv0);
103  float lQlNeg = RecoDecay::dotProd(pneg, pv0) / momTot;
104  float lQlPos = RecoDecay::dotProd(ppos, pv0) / momTot;
105  return (lQlPos - lQlNeg) / (lQlPos + lQlNeg); //longitudinal momentum asymmetry
106  }
107 
108  float qtarmv0(const array<float, 3>& ppos, const array<float, 3>& pneg)
109  {
110  std::array<float, 3> pv0 = {ppos[0] + pneg[0], ppos[1] + pneg[1], ppos[2] + pneg[2]};
111  float momTot2 = RecoDecay::P2(pv0);
112  float dp = RecoDecay::dotProd(pneg, pv0);
113  return std::sqrt(RecoDecay::P2(pneg) - dp * dp / momTot2); //qtarm
114  }

116  float phivv0(const array<float, 3>& ppos, const array<float, 3>& pneg, const int cpos, const int cneg, const float bz)
117  {
118  //momentum of e+ and e- in (ax,ay,az) axis. Note that az=0 by definition.
119  //vector product of pep X pem
120  float vpx = 0, vpy = 0, vpz = 0;
121  if (cpos * cneg > 0.) { // Like Sign
122  if (bz * cpos < 0) {
123  vpx = ppos[1] * pneg[2] - ppos[2] * pneg[1];
124  vpy = ppos[2] * pneg[0] - ppos[0] * pneg[2];
125  vpz = ppos[0] * pneg[1] - ppos[2] * pneg[0];
126  } else {
127  vpx = pneg[1] * ppos[2] - pneg[2] * ppos[1];
128  vpy = pneg[2] * ppos[0] - pneg[0] * ppos[2];
129  vpz = pneg[0] * ppos[1] - pneg[2] * ppos[0];
130  }
131  } else { // Unlike Sign
132  if (bz * cpos > 0) {
133  vpx = ppos[1] * pneg[2] - ppos[2] * pneg[1];
134  vpy = ppos[2] * pneg[0] - ppos[0] * pneg[2];
135  vpz = ppos[0] * pneg[1] - ppos[2] * pneg[0];
136  } else {
137  vpx = pneg[1] * ppos[2] - pneg[2] * ppos[1];
138  vpy = pneg[2] * ppos[0] - pneg[0] * ppos[2];
139  vpz = pneg[0] * ppos[1] - pneg[2] * ppos[0];
140  }
141  }
142 
143  float vp = RecoDecay::P(array{vpx, vpy, vpz});
144  //unit vector of pep X pem
145  float vx = vpx / vp;
146  float vy = vpy / vp;
147  float vz = vpz / vp;
148 
149  float px = ppos[0] + pneg[0];
150  float py = ppos[1] + pneg[1];
151  float pz = ppos[2] + pneg[2];
152 
153  //unit vector of (pep+pem)
154  float pl = RecoDecay::P(array{px, py, pz});
155  float ux = px / pl;
156  float uy = py / pl;
157  float uz = pz / pl;
158  float ax = uy / RecoDecay::sqrtSumOfSquares(ux, uy);
159  float ay = -ux / RecoDecay::sqrtSumOfSquares(ux, uy);
160 
161  //The third axis defined by vector product (ux,uy,uz)X(vx,vy,vz)
162  float wx = uy * vz - uz * vy;
163  float wy = uz * vx - ux * vz;
164  // by construction, (wx,wy,wz) must be a unit vector. Measure angle between (wx,wy,wz) and (ax,ay,0).
165  // The angle between them should be small if the pair is conversion. This function then returns values close to pi!
166  float cosPhiV = wx * ax + wy * ay;
167  return TMath::ACos(cosPhiV); //phiv in [0,pi]
168  }
169 
170  float psipairv0(const array<float, 3>& ppos, const array<float, 3>& pneg, const float bz)
171  {
172  //Following idea to use opening of colinear pairs in magnetic field from e.g. PHENIX to ID conversions.
173  float deltat = TMath::ATan(pneg[2] / (TMath::Sqrt(pneg[0] * pneg[0] + pneg[1] * pneg[1]))) - TMath::ATan(ppos[2] / (TMath::Sqrt(ppos[0] * ppos[0] + ppos[1] * ppos[1]))); //difference of angles of the two daughter tracks with z-axis
174  float pEle = RecoDecay::P(pneg); //absolute momentum val
175  float pPos = RecoDecay::P(ppos); //absolute momentum val
176  float chipair = TMath::ACos(RecoDecay::dotProd(ppos, pneg) / (pEle * pPos)); //Angle between daughter tracks
177  return TMath::Abs(TMath::ASin(deltat / chipair)); //psipair in [0,pi/2]
178  }

180  int processV0(const array<float, 3>& ppos, const array<float, 3>& pneg)
181  {
182  float alpha = alphav0(ppos, pneg);
183  float qt = qtarmv0(ppos, pneg);
184 
185  // Gamma cuts
186  const float cutAlphaG = 0.4;
187  const float cutQTG = 0.03;
188  const float cutAlphaG2[2] = {0.4, 0.8};
189  const float cutQTG2 = 0.02;
190 
191  // K0S cuts
192  const float cutQTK0S[2] = {0.1075, 0.215};
193  const float cutAPK0S[2] = {0.199, 0.8}; // parameters for curved QT cut
194 
195  // Lambda & A-Lambda cuts
196  const float cutQTL = 0.03;
197  const float cutAlphaL[2] = {0.35, 0.7};
198  const float cutAlphaAL[2] = {-0.7, -0.35};
199  const float cutAPL[3] = {0.107, -0.69, 0.5}; // parameters fir curved QT cut
200 
201  // Check for Gamma candidates
202  if (qt < cutQTG) {
203  if ((TMath::Abs(alpha) < cutAlphaG)) {
204  return kGamma;
205  }
206  }
207  if (qt < cutQTG2) {
208  // additional region - should help high pT gammas
209  if ((TMath::Abs(alpha) > cutAlphaG2[0]) && (TMath::Abs(alpha) < cutAlphaG2[1])) {
210  return kGamma;
211  }
212  }
213 
214  // Check for K0S candidates
215  float q = cutAPK0S[0] * TMath::Sqrt(TMath::Abs(1 - alpha * alpha / (cutAPK0S[1] * cutAPK0S[1])));
216  if ((qt > cutQTK0S[0]) && (qt < cutQTK0S[1]) && (qt > q)) {
217  return kK0S;
218  }
219 
220  // Check for Lambda candidates
221  q = cutAPL[0] * TMath::Sqrt(TMath::Abs(1 - ((alpha + cutAPL[1]) * (alpha + cutAPL[1])) / (cutAPL[2] * cutAPL[2])));
222  if ((alpha > cutAlphaL[0]) && (alpha < cutAlphaL[1]) && (qt > cutQTL) && (qt < q)) {
223  return kLambda;
224  }
225 
226  // Check for A-Lambda candidates
227  q = cutAPL[0] * TMath::Sqrt(TMath::Abs(1 - ((alpha - cutAPL[1]) * (alpha - cutAPL[1])) / (cutAPL[2] * cutAPL[2])));
228  if ((alpha > cutAlphaAL[0]) && (alpha < cutAlphaAL[1]) && (qt > cutQTL) && (qt < q)) {
229  return kAntiLambda;
230  }
231 
232  return kUndef;
233  }

340  void process(aod::Collision const& collision, aod::V0s const& V0s, FullTracksExt const& tracks)
341  {
342
