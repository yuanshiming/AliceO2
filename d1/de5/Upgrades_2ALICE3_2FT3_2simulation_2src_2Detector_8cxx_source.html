62 void Detector::buildBasicFT3(int nLayers, Float_t z_first, Float_t z_length, Float_t etaIn, Float_t etaOut, Float_t Layerx2X0)
63 {
64  // Build a basic parametrized FT3 detector with nLayers equally spaced between z_first and z_first+z_length
65  // Covering pseudo rapidity [etaIn,etaOut]. Passive silicon thinkness computed to match layer x/X0
66 
67  mNumberOfLayers = nLayers;
68  Float_t sensorThickness = 30.e-4;
69  mLayerName.resize(2);
70  mLayerName[0].resize(mNumberOfLayers);
71  mLayerName[1].resize(mNumberOfLayers);
72  mLayerID.resize(2);
73  mLayerID[0].resize(mNumberOfLayers);
74  mLayerID[1].resize(mNumberOfLayers);
75  mLayers.resize(2);
76 
77  for (Int_t direction : {0, 1}) {
78  for (Int_t layerNumber = 0; layerNumber < mNumberOfLayers; layerNumber++) {
79  std::string layerName = GeometryTGeo::getFT3LayerPattern() + std::to_string(layerNumber + mNumberOfLayers * direction);
80  mLayerName[direction][layerNumber] = layerName;
81 
82  // Adds evenly spaced layers
83  Float_t layerZ = z_first + (layerNumber * z_length / (mNumberOfLayers - 1)) * std::copysign(1, z_first);
84  Float_t rIn = std::abs(layerZ * std::tan(2.f * std::atan(std::exp(-etaIn))));
85  Float_t rOut = std::abs(layerZ * std::tan(2.f * std::atan(std::exp(-etaOut))));
86 
87  auto& thisLayer = mLayers[direction].emplace_back(direction, layerNumber, layerName, layerZ, rIn, rOut, sensorThickness, Layerx2X0);
88  }
89  }
90 }

93 void Detector::buildFT3V1()
94 {
95  //Build FT3 detector according to
96  //https://indico.cern.ch/event/992488/contributions/4174473/attachments/2168881/3661331/tracker_parameters_werner_jan_11_2021.pdf
97 
98  LOG(INFO) << "Building FT3 Detector V1";
99 
100  mNumberOfLayers = 10;
101  Float_t sensorThickness = 30.e-4;
102  Float_t layersx2X0 = 1.e-2;
103  std::vector<std::array<Float_t, 5>> layersConfig{
104  {16., .5, 3., sensorThickness, 0.1f * layersx2X0}, // {z_layer, r_in, r_out, sensor_thickness, Layerx2X0}
105  {20., .5, 3., sensorThickness, 0.1f * layersx2X0},
106  {24., .5, 3., sensorThickness, 0.1f * layersx2X0},
107  {77., 3.5, 35., sensorThickness, layersx2X0},
108  {100., 3.5, 35., sensorThickness, layersx2X0},
109  {122., 3.5, 35., sensorThickness, layersx2X0},
110  {150., 3.5, 100., sensorThickness, layersx2X0},
111  {180., 3.5, 100., sensorThickness, layersx2X0},
112  {220., 3.5, 100., sensorThickness, layersx2X0},
113  {279., 3.5, 100., sensorThickness, layersx2X0}};
114 
115  mLayerName.resize(2);
116  mLayerName[0].resize(mNumberOfLayers);
117  mLayerName[1].resize(mNumberOfLayers);
118  mLayerID.resize(2);
119  mLayerID[0].resize(mNumberOfLayers);
120  mLayerID[1].resize(mNumberOfLayers);
121  mLayers.resize(2);
122 
123  for (auto direction : {0, 1}) {
124  for (int layerNumber = 0; layerNumber < mNumberOfLayers; layerNumber++) {
125  std::string directionName = std::to_string(direction);
126  std::string layerName = GeometryTGeo::getFT3LayerPattern() + directionName + "_" + std::to_string(layerNumber);
127  mLayerName[direction][layerNumber] = layerName;
128  auto& z = layersConfig[layerNumber][0];
129 
130  auto& rIn = layersConfig[layerNumber][1];
131  auto& rOut = layersConfig[layerNumber][2];
132  auto& thickness = layersConfig[layerNumber][3];
133  auto& x0 = layersConfig[layerNumber][4];
134 
135  LOG(INFO) << "Adding Layer " << layerName << " at z = " << z;
136  // Add layers
137  auto& thisLayer = mLayers[direction].emplace_back(direction, layerNumber, layerName, z, rIn, rOut, thickness, x0);
138  }
139  }
140 }

218 Bool_t Detector::ProcessHits(FairVolume* vol)
219 {
220  // This method is called from the MC stepping
221  if (!(fMC->TrackCharge())) {
222  return kFALSE;
223  }
224 
225  Int_t lay = 0, volID = vol->getMCid();
226  while ((lay <= mNumberOfLayers * 2) && (volID != mLayerID[lay % 2][lay / 2])) {
227  ++lay;
228  }
229 
230  auto stack = (o2::data::Stack*)fMC->GetStack();
231 
232  bool startHit = false, stopHit = false;
233  unsigned char status = 0;
234  if (fMC->IsTrackEntering()) {
235  status |= Hit::kTrackEntering;
236  }
237  if (fMC->IsTrackInside()) {
238  status |= Hit::kTrackInside;
239  }
240  if (fMC->IsTrackExiting()) {
241  status |= Hit::kTrackExiting;
242  }
243  if (fMC->IsTrackOut()) {
244  status |= Hit::kTrackOut;
245  }
246  if (fMC->IsTrackStop()) {
247  status |= Hit::kTrackStopped;
248  }
249  if (fMC->IsTrackAlive()) {
250  status |= Hit::kTrackAlive;
251  }
252 
253  // track is entering or created in the volume
254  if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
255  startHit = true;
256  } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
257  stopHit = true;
258  }
259 
260  // increment energy loss at all steps except entrance
261  if (!startHit) {
262  mTrackData.mEnergyLoss += fMC->Edep();
263  }
264  if (!(startHit | stopHit)) {
265  return kFALSE; // do noting
266  }
267  if (startHit) {
268  mTrackData.mEnergyLoss = 0.;
269  fMC->TrackMomentum(mTrackData.mMomentumStart);
270  fMC->TrackPosition(mTrackData.mPositionStart);
271  mTrackData.mTrkStatusStart = status;
272  mTrackData.mHitStarted = true;
273  }
274  if (stopHit) {
275  TLorentzVector positionStop;
276  fMC->TrackPosition(positionStop);
277  // Retrieve the indices with the volume path
278  int chipindex = lay;
279 
280  Hit* p = addHit(stack->GetCurrentTrackNumber(), chipindex, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
281  mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
282  mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
283  // p->SetTotalEnergy(vmc->Etot());
284 
285  // RS: not sure this is needed
286  // Increment number of Detector det points in TParticle
287  stack->addHit(GetDetId());
288  }
289 
290  return kTRUE;
291 }

303  Float_t epsilSi = 1.0E-4; // .10000E+01;
304  Float_t stminSi = 0.0; // cm "Default value used"
305 
306  Float_t tmaxfdAir = 0.1; // .10000E+01; // Degree
307  Float_t stemaxAir = .10000E+01; // cm
308  Float_t deemaxAir = 0.1; // 0.30000E-02; // Fraction of particle's energy 0<deemax<=1
309  Float_t epsilAir = 1.0E-4; // .10000E+01;
310  Float_t stminAir = 0.0; // cm "Default value used"
311 
312  // AIR
313  Float_t aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
314  Float_t zAir[4] = {6., 7., 8., 18.};
315  Float_t wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
316  Float_t dAir = 1.20479E-3;
317 
318  o2::base::Detector::Mixture(1, "AIR$", aAir, zAir, dAir, 4, wAir);
319  o2::base::Detector::Medium(1, "AIR$", 1, 0, ifield, fieldm, tmaxfdAir, stemaxAir, deemaxAir, epsilAir, stminAir);
320 
321  o2::base::Detector::Material(3, "SI$", 0.28086E+02, 0.14000E+02, 0.23300E+01, 0.93600E+01, 0.99900E+03);
322  o2::base::Detector::Medium(3, "SI$", 3, 0, ifield, fieldm, tmaxfdSi, stemaxSi, deemaxSi, epsilSi, stminSi);
323 }

360 {
361 
362  mGeometryTGeo = GeometryTGeo::Instance();
363 
364  TGeoVolume* volFT3 = new TGeoVolumeAssembly(GeometryTGeo::getFT3VolPattern());
365 
366  LOG(INFO) << "GeometryBuilder::buildGeometry volume name = " << GeometryTGeo::getFT3VolPattern();
367 
368  TGeoVolume* vALIC = gGeoManager->GetVolume("barrel");

372 
373  LOG(DEBUG) << "buildGeometry: "

375 
376  for (Int_t direction : {0, 1}) {
377

379  mLayers[direction][iLayer].createLayer(volFT3);

383 
384  for (auto direction : {0, 1}) {
385  for (int iLayer = 0; iLayer < mNumberOfLayers; iLayer++) {
386  mLayerID[direction][iLayer] = gMC ? TVirtualMC::GetMC()->VolId(Form("%s_%d_%d", GeometryTGeo::getFT3SensorPattern(), direction, iLayer)) : 0;
387  std::string directionString = direction ? "Forward" : "Backward";
388  LOG(INFO) << "mLayerID for " << directionString << " layer " << iLayer << " = " << mLayerID[direction][iLayer];
389  }
