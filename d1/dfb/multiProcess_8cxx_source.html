91 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
92 {
93  if (!cfgc.options().get<bool>("doMC")) {
94  return WorkflowSpec{
95  // only use rec-level process when MC info is not there
96  adaptAnalysisTask<MultipleProcessExample>(cfgc, Processes{&MultipleProcessExample::processRec}),
97  };
98  }
99  return WorkflowSpec{
100  // use additional process functions when MC info is present
101  // functions will be executed in the sequence they are listed - allows to use, for example,
102  // histograms that were filled previously
103  // produced tables *cannot* be used
104  // filters will be applied *to all* processes
105  adaptAnalysisTask<MultipleProcessExample>(cfgc, Processes{&MultipleProcessExample::processRec,
106  &MultipleProcessExample::processGen,
107  &MultipleProcessExample::processResolution}),
108  };
